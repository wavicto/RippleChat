8pub fn do_derive_schema(item: proc_macro::TokenStream) -> proc_macro::TokenStream {
9    let input = parse_macro_input!(item as DeriveInput);
10
11    let span = input.span();
12    let name = &input.ident;
13
14    let generator = match Generator::new(&input) {
15        Ok(generator) => generator,
16        Err(err) => return err.into_compile_error().into(),
17    };
18
19    // Add a bound `T: Schema` to every type parameter T.
20    let generics = generator.add_trait_bounds(input.generics);
21    let (impl_generics, ty_generics, where_clause) = generics.split_for_impl();
22
23    let ty = generator
24        .generate_type(&input.data, span, name.to_string())
25        .unwrap_or_else(syn::Error::into_compile_error);
26
27    let postcard_schema = &generator.postcard_schema;
28    let expanded = quote! {
29        impl #impl_generics #postcard_schema::Schema for #name #ty_generics #where_clause {
30            const SCHEMA: &'static #postcard_schema::schema::NamedType = #ty;
31        }
32    };
33
34    expanded.into()
35}
36
37struct Generator {
38    postcard_schema: Path,
39}
40
41impl Generator {
42    fn new(input: &DeriveInput) -> syn::Result<Self> {
43        let mut generator = Self {
44            postcard_schema: parse_quote!(::postcard_schema),
45        };
46        for attr in &input.attrs {
47            if attr.path().is_ident("postcard") {
48                attr.parse_nested_meta(|meta| {
49                    // #[postcard(crate = path::to::postcard)]
50                    if meta.path.is_ident("crate") {
51                        generator.postcard_schema = meta.value()?.parse()?;
52                        return Ok(());
53                    }
54
55                    Err(meta.error("unsupported #[postcard] attribute"))
56                })?;
57            }
58        }
59        Ok(generator)
60    }
61
62    fn generate_type(
63        &self,
64        data: &Data,
65        span: Span,
66        name: String,
67    ) -> Result<TokenStream, syn::Error> {
68        let postcard_schema = &self.postcard_schema;
69        let ty = match data {
70            Data::Struct(data) => self.generate_struct(&data.fields),
71            Data::Enum(data) => {
72                let name = data.variants.iter().map(|v| v.ident.to_string());
73                let ty = data
74                    .variants
75                    .iter()
76                    .map(|v| self.generate_variants(&v.fields));
77
78                quote! {
79                    &#postcard_schema::schema::DataModelType::Enum(&[
80                        #( &#postcard_schema::schema::NamedVariant { name: #name, ty: #ty } ),*
81                    ])
82                }
83            }
84            Data::Union(_) => {
85                return Err(syn::Error::new(
86                    span,
87                    "unions are not supported by `postcard::experimental::schema`",
88                ))
89            }
90        };
91
92        Ok(quote! {
93            &#postcard_schema::schema::NamedType {
94                name: #name,
95                ty: #ty,
96            }
97        })
98    }
99
100    fn generate_struct(&self, fields: &Fields) -> TokenStream {
101        let postcard_schema = &self.postcard_schema;
102        match fields {
103            syn::Fields::Named(fields) => {
104                let fields = fields.named.iter().map(|f| {
105                let ty = &f.ty;
106                let name = f.ident.as_ref().unwrap().to_string();
107                quote_spanned!(f.span() => &#postcard_schema::schema::NamedValue { name: #name, ty: <#ty as #postcard_schema::Schema>::SCHEMA })
108            });
109                quote! { &#postcard_schema::schema::DataModelType::Struct(&[
110                    #( #fields ),*
111                ]) }
112            }
113            syn::Fields::Unnamed(fields) => {
114                if fields.unnamed.len() == 1 {
115                    let f = fields.unnamed[0].clone();
116                    let ty = &f.ty;
117                    let qs = quote_spanned!(f.span() => <#ty as #postcard_schema::Schema>::SCHEMA);
118
119                    quote! { &#postcard_schema::schema::DataModelType::NewtypeStruct(#qs) }
120                } else {
121                    let fields = fields.unnamed.iter().map(|f| {
122                        let ty = &f.ty;
123                        quote_spanned!(f.span() => <#ty as #postcard_schema::Schema>::SCHEMA)
124                    });
125                    quote! { &#postcard_schema::schema::DataModelType::TupleStruct(&[
126                        #( #fields ),*
127                    ]) }
128                }
129            }
130            syn::Fields::Unit => {
131                quote! { &#postcard_schema::schema::DataModelType::UnitStruct }
132            }
133        }
134    }
135
136    fn generate_variants(&self, fields: &Fields) -> TokenStream {
137        let postcard_schema = &self.postcard_schema;
138        match fields {
139            syn::Fields::Named(fields) => {
140                let fields = fields.named.iter().map(|f| {
141                let ty = &f.ty;
142                let name = f.ident.as_ref().unwrap().to_string();
143                quote_spanned!(f.span() => &#postcard_schema::schema::NamedValue { name: #name, ty: <#ty as #postcard_schema::Schema>::SCHEMA })
144            });
145                quote! { &#postcard_schema::schema::DataModelVariant::StructVariant(&[
146                    #( #fields ),*
147                ]) }
148            }
149            syn::Fields::Unnamed(fields) => {
150                if fields.unnamed.len() == 1 {
151                    let f = fields.unnamed[0].clone();
152                    let ty = &f.ty;
153                    let qs = quote_spanned!(f.span() => <#ty as #postcard_schema::Schema>::SCHEMA);
154
155                    quote! { &#postcard_schema::schema::DataModelVariant::NewtypeVariant(#qs) }
156                } else {
157                    let fields = fields.unnamed.iter().map(|f| {
158                        let ty = &f.ty;
159                        quote_spanned!(f.span() => <#ty as #postcard_schema::Schema>::SCHEMA)
160                    });
161                    quote! { &#postcard_schema::schema::DataModelVariant::TupleVariant(&[
162                        #( #fields ),*
163                    ]) }
164                }
165            }
166            syn::Fields::Unit => {
167                quote! { &#postcard_schema::schema::DataModelVariant::UnitVariant }
168            }
169        }
170    }
171
172    /// Add a bound `T: MaxSize` to every type parameter T.
173    fn add_trait_bounds(&self, mut generics: Generics) -> Generics {
174        let postcard_schema = &self.postcard_schema;
175        for param in &mut generics.params {
176            if let GenericParam::Type(ref mut type_param) = *param {
177                type_param
178                    .bounds
179                    .push(parse_quote!(#postcard_schema::Schema));
180            }
181        }
182        generics
183    }
184}
