12fn convert_hex_ipv4(hex_ip: &str) -> Ipv4Addr {
13    if hex_ip.len() != 8 {
14        return Ipv4Addr::UNSPECIFIED;
15    }
16    let o1: u8 = u8::from_str_radix(&hex_ip[6..8], 0x10).unwrap_or(0);
17    let o2: u8 = u8::from_str_radix(&hex_ip[4..6], 0x10).unwrap_or(0);
18    let o3: u8 = u8::from_str_radix(&hex_ip[2..4], 0x10).unwrap_or(0);
19    let o4: u8 = u8::from_str_radix(&hex_ip[0..2], 0x10).unwrap_or(0);
20    Ipv4Addr::new(o1, o2, o3, o4)
21}

23fn convert_hex_ipv6(hex_ip: &str) -> Ipv6Addr {
24    if hex_ip.len() != 32 {
25        return Ipv6Addr::UNSPECIFIED;
26    }
27    let h1: u16 = u16::from_str_radix(&hex_ip[0..4], 0x10).unwrap_or(0);
28    let h2: u16 = u16::from_str_radix(&hex_ip[4..8], 0x10).unwrap_or(0);
29    let h3: u16 = u16::from_str_radix(&hex_ip[8..12], 0x10).unwrap_or(0);
30    let h4: u16 = u16::from_str_radix(&hex_ip[12..16], 0x10).unwrap_or(0);
31    let h5: u16 = u16::from_str_radix(&hex_ip[16..20], 0x10).unwrap_or(0);
32    let h6: u16 = u16::from_str_radix(&hex_ip[20..24], 0x10).unwrap_or(0);
33    let h7: u16 = u16::from_str_radix(&hex_ip[24..28], 0x10).unwrap_or(0);
34    let h8: u16 = u16::from_str_radix(&hex_ip[28..32], 0x10).unwrap_or(0);
35    Ipv6Addr::new(h1, h2, h3, h4, h5, h6, h7, h8)
36}

38fn get_arp_map() -> HashMap<Ipv4Addr, MacAddr> {
39    let mut arp_map: HashMap<Ipv4Addr, MacAddr> = HashMap::new();
40    let arp_data = read_to_string(PATH_PROC_NET_ARP);
41    let arp_text = match arp_data {
42        Ok(content) => content,
43        Err(_) => String::new(),
44    };
45    let arp_table: Vec<&str> = arp_text.trim().split("\n").collect();
46    for row in arp_table {
47        let mut fields: Vec<&str> = row.split(" ").collect();
48        fields.retain(|value| *value != "");
49        if fields.len() >= 6 {
50            // fields[0]: IP Address
51            // fields[3]: MAC Address (colon-separated string of hex format)
52            // fields[5]: Interface Name
53            match Ipv4Addr::from_str(fields[0]) {
54                Ok(ipv4_addr) => {
55                    arp_map.insert(ipv4_addr, MacAddr::from_hex_format(fields[3]));
56                }
57                Err(_) => {}
58            }
59        }
60    }
61    arp_map
62}

64fn get_ipv4_gateway_map() -> HashMap<String, Ipv4Addr> {
65    let mut ipv4_gateway_map: HashMap<String, Ipv4Addr> = HashMap::new();
66    let route_data = read_to_string(PATH_PROC_NET_ROUTE);
67    let route_text = match route_data {
68        Ok(content) => content,
69        Err(_) => String::new(),
70    };
71    let route_table: Vec<&str> = route_text.trim().split("\n").collect();
72    for row in route_table {
73        let fields: Vec<&str> = row.split("\t").collect();
74        if fields.len() >= 3 {
75            // fields[0]: Interface Name
76            // fields[2]: IPv4 Address 8 hex chars
77            if fields[2] != "00000000" {
78                ipv4_gateway_map.insert(fields[0].to_string(), convert_hex_ipv4(fields[2]));
79            }
80        }
81    }
82    ipv4_gateway_map
83}

93    for row in route_table {
94        let fields: Vec<&str> = row.split("\t").collect();
95        if fields.len() >= 10 {
96            // fields[4]: IPv6 Address 32 hex chars without colons
97            // fields[9]: Interface Name
98            if fields[4] != "00000000000000000000000000000000" {
99                ipv6_gateway_map.insert(fields[9].to_string(), convert_hex_ipv6(fields[4]));
100            }
101        }
102    }
103    ipv6_gateway_map
104}

106pub fn get_gateway_map() -> HashMap<String, NetworkDevice> {
107    let mut gateway_map: HashMap<String, NetworkDevice> = HashMap::new();
108    let arp_map: HashMap<Ipv4Addr, MacAddr> = get_arp_map();
109    for (if_name, ipv4_addr) in get_ipv4_gateway_map() {
110        let gateway = gateway_map.entry(if_name).or_insert(NetworkDevice::new());
111        if let Some(mac_addr) = arp_map.get(&ipv4_addr) {
112            gateway.mac_addr = mac_addr.clone();
113        }
114        gateway.ipv4.push(ipv4_addr);
115    }
116    for (if_name, ipv6_addr) in get_ipv6_gateway_map() {
117        let gateway = gateway_map.entry(if_name).or_insert(NetworkDevice::new());
118        gateway.ipv6.push(ipv6_addr);
119    }
120    gateway_map
