14/// Get default Gateway
15pub fn get_default_gateway() -> Result<NetworkDevice, String> {
16    let local_ip: IpAddr = match interface::get_local_ipaddr() {
17        Some(local_ip) => local_ip,
18        None => return Err(String::from("Local IP address not found")),
19    };
20    let interfaces: Vec<Interface> = interface::get_interfaces();
21    for iface in interfaces {
22        match local_ip {
23            IpAddr::V4(local_ipv4) => {
24                if iface.ipv4.iter().any(|x| x.addr() == local_ipv4) {
25                    if let Some(gateway) = iface.gateway {
26                        return Ok(gateway);
27                    }
28                }
29            }
30            IpAddr::V6(local_ipv6) => {
31                if iface.ipv6.iter().any(|x| x.addr() == local_ipv6) {
32                    if let Some(gateway) = iface.gateway {
33                        return Ok(gateway);
34                    }
35                }
36            }
37        }
38    }
39    Err(String::from("Default Gateway not found"))
40}
