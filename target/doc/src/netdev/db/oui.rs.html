23/// Check if the MAC address is a Virtual Machine MAC address
24pub fn is_virtual_mac(mac: &MacAddr) -> bool {
25    let mac = mac.address();
26    let prefix = mac[0..8].to_uppercase();
27    KNOWN_VM_MAC_PREFIXES.contains(&prefix.as_str())
28}
29
30/// Check if the MAC address is a known Loopback MAC address
31pub fn is_known_loopback_mac(mac: &MacAddr) -> bool {
32    let mac = mac.address();
33    KNOWN_LOOPBACK_MAC_ADDRESSES.contains(&mac.to_uppercase().as_str())
