18/// [`ArcSlice`] is a fun optimisation. For `FuturesUnorderedBounded`, we have `n` slots for futures,
19/// and we create a separate context when polling each individual future to avoid having n^2 polling.
20///
21/// Originally, we pre-allocated `n` `Arc<Wake>` types and stored those along side the future slots.
22/// These wakers would have a `Weak` pointing to some shared state, as well as an index for which slot this
23/// waker is associated with.
24///
25/// [`RawWaker`] only gives us 1 pointer worth of data to play with - but we need 2. So unfortunately we needed
26/// the extra allocations here
27///
28/// ... unless we hack around a little bit!
29///
30/// [`ArcSlice`] represents the shared state, as well as having a long tail of indices. The layout is as follows
31/// ```text
32/// [ strong_count | waker | head | tail | len | slot 0 | slot 1 | slot 2 | slot 3 | ... ]
33/// ```
34///
35/// [`ArcSlot`] represents our [`RawWaker`]. It points to one of the numbers in the list.
36/// Since the layouts of the internals are fixed (`repr(C)`) - we can count back from the index to find
37/// the shared data.
38///
39/// For example, if we have an `ArcSlot` pointing at the number 2, we can count back to pointer 2 `usize`s + [`ArcSliceInnerMeta`] to find
40/// the start of the [`ArcSlice`], and then we can insert `2` into the list of futures to poll, finally calling `waker.wake()`.
41///
42/// Each slot also forms part of a linked list.
43pub(crate) struct ArcSlice {
44    ptr: NonNull<ArcSliceInnerMeta>,
45    phantom: PhantomData<ArcSliceInner>,
46}

103impl ArcSlice {
104    fn slice_start(&self) -> *mut ArcSlotInner {
105        unsafe {
106            let ptr = self.ptr.as_ptr().cast::<u8>();
107            ptr.add(slice_offset()).cast::<ArcSlotInner>()
108        }
109    }
110
111    /// The push function from the 1024cores intrusive MPSC queue algorithm.
112    /// <https://www.1024cores.net/home/lock-free-algorithms/queues/intrusive-mpsc-node-based-queue>
113    ///
114    /// Safety: index must be within capacity
115    pub(crate) unsafe fn push(&self, index: usize) {
116        let queue = unsafe { &*ptr::addr_of!((*self.ptr.as_ptr()).queue) };
117        let slot = unsafe { self.slice_start().add(index) };
118
119        let mut wake_lock = unsafe { &*slot }.wake_lock.lock();
120        let prev = core::mem::replace(&mut *wake_lock, true);
121
122        if !prev {
123            queue.enqueue(unsafe { NonNull::new_unchecked(slot) });
124        }
125    }
126
127    /// Register the waker
128    pub(crate) fn register(&mut self, waker: &Waker) {
129        // Safety:
130        // Diatomic waker requires we do not concurrently run
131        // "register", "unregister", and "wait_until".
132        // we only call register with mut access, thus we are safe.
133        let meta = unsafe { &*self.ptr.as_ptr() };
134        unsafe { meta.waker.register(waker) }
135    }
136
137    fn get(&self, index: usize) -> ManuallyDrop<Waker> {
138        // SAFETY: This cannot go through Deref::deref or RcBoxPtr::inner because
139        // this is required to retain raw/mut provenance such that e.g. `get_mut` can
140        // write through the pointer after the Rc is recovered through `from_raw`.
141        let slot = unsafe { self.slice_start().add(index) };
142
143        debug_assert_eq!(
144            unsafe { (*slot).index },
145            index,
146            "the slot should point at our index"
147        );
148        slot::waker(slot)
149    }
150
151    /// The pop function from the 1024cores intrusive MPSC queue algorithm
152    ///
153    /// Note that this is unsafe as it required mutual exclusion (only one
154    /// thread can call this) to be guaranteed elsewhere.
155    pub(crate) unsafe fn pop(&self) -> ReadySlot<(usize, ManuallyDrop<Waker>)> {
156        let queue = unsafe { &*ptr::addr_of!((*self.ptr.as_ptr()).queue) };
157        match unsafe { queue.try_dequeue_unchecked() } {
158            Ok(slot) => {
159                let slot = unsafe { &*slot.as_ptr() };
160                *slot.wake_lock.lock() = false;
161                ReadySlot::Ready((slot.index, self.get(slot.index)))
162            }
163            Err(TryDequeueError::Inconsistent) => ReadySlot::Inconsistent,
164            Err(TryDequeueError::Empty) => ReadySlot::None,
165            Err(TryDequeueError::Busy) => unreachable!(),
166        }
167    }
168}

185    /// Traverses back the [`ArcSlice`] to find the [`ArcSliceInnerMeta`] pointer
186    ///
187    /// # Safety:
188    /// `ptr` must be from an `ArcSlot` originally
189    unsafe fn meta_raw(ptr: *mut ArcSlotInner) -> *mut ArcSliceInnerMeta {
190        let index = unsafe { (*ptr).index };
191        let slice_start = unsafe { ptr.sub(index) };
192
193        unsafe { slice_start.cast::<u8>().sub(slice_offset()) }.cast::<ArcSliceInnerMeta>()
194    }
195
196    /// Traverses back the [`ArcSlice`] to find the [`ArcSliceInnerMeta`] pointer
197    ///
198    /// # Safety:
199    /// * `ptr` must be from an `ArcSlot` originally
200    /// * The original `ArcSlot` must outlive `'a`
201    unsafe fn meta_ref<'a>(ptr: *const ArcSlotInner) -> &'a ArcSliceInnerMeta {
202        unsafe { &*meta_raw(ptr.cast_mut()) }
203    }
204
205    pub(super) fn waker(ptr: *const ArcSlotInner) -> ManuallyDrop<Waker> {
206        static VTABLE: &RawWakerVTable =
207            &RawWakerVTable::new(clone_waker, wake, wake_by_ref, drop_waker);
208
209        // Increment the reference count of the arc to clone it.
210        unsafe fn clone_waker(waker: *const ()) -> RawWaker {
211            unsafe { meta_ref(waker.cast()).inc_strong() };
212            RawWaker::new(waker, VTABLE)
213        }
214
215        // We don't need ownership. Just wake_by_ref and drop the waker
216        unsafe fn wake(waker: *const ()) {
217            unsafe {
218                wake_by_ref(waker);
219                drop_waker(waker);
220            }
221        }
222
223        // Find the `ArcSliceInnerMeta` and push the current index value into it,
224        // then call the stored waker to trigger a poll
225        unsafe fn wake_by_ref(waker: *const ()) {
226            let slot = waker.cast::<ArcSlotInner>();
227
228            let node = unsafe { &*slot };
229
230            let mut wake_lock = node.wake_lock.lock();
231            let prev = core::mem::replace(&mut *wake_lock, true);
232
233            if !prev {
234                let meta = unsafe { meta_ref(slot) };
235                meta.queue
236                    .enqueue(unsafe { NonNull::new_unchecked(slot.cast_mut()) });
237                meta.waker.notify();
238            }
239        }
240
241        // Decrement the reference count of the Arc on drop
242        unsafe fn drop_waker(waker: *const ()) {
243            let meta = unsafe { meta_ref(waker.cast()) };
244            if meta.dec_strong() {
245                unsafe {
246                    super::drop_inner(meta_raw(waker.cast::<ArcSlotInner>().cast_mut()), meta.len);
247                }
248            }
249        }
250
251        let raw_waker = RawWaker::new(ptr.cast(), VTABLE);
252        unsafe { ManuallyDrop::new(Waker::from_raw(raw_waker)) }
253    }
254}

256impl ArcSliceInnerMeta {
257    fn inc_strong(&self) {
258        // Using a relaxed ordering is alright here, as knowledge of the
259        // original reference prevents other threads from erroneously deleting
260        // the object.
261        //
262        // As explained in the [Boost documentation][1], Increasing the
263        // reference counter can always be done with memory_order_relaxed: New
264        // references to an object can only be formed from an existing
265        // reference, and passing an existing reference from one thread to
266        // another must already provide any required synchronization.
267        //
268        // [1]: (www.boost.org/doc/libs/1_55_0/doc/html/atomic/usage_examples.html)
269        let old_size = self.strong.fetch_add(1, Ordering::Relaxed);
270
271        // However we need to guard against massive refcounts in case someone is `mem::forget`ing
272        // Arcs. If we don't do this the count can overflow and users will use-after free. This
273        // branch will never be taken in any realistic program. We abort because such a program is
274        // incredibly degenerate, and we don't care to support it.
275        //
276        // This check is not 100% water-proof: we error when the refcount grows beyond `isize::MAX`.
277        // But we do that check *after* having done the increment, so there is a chance here that
278        // the worst already happened and we actually do overflow the `usize` counter. However, that
279        // requires the counter to grow from `isize::MAX` to `usize::MAX` between the increment
280        // above and the `abort` below, which seems exceedingly unlikely.
281        if old_size > (isize::MAX) as usize {
282            abort("too many arc clones");
283        }
284    }
285    fn dec_strong(&self) -> bool {
286        // Because `fetch_sub` is already atomic, we do not need to synchronize
287        // with other threads unless we are going to delete the object. This
288        // same logic applies to the below `fetch_sub` to the `weak` count.
289        let old_size = self.strong.fetch_sub(1, Ordering::Release);
290        if old_size != 1 {
291            return false;
292        }
293
294        // This fence is needed to prevent reordering of use of the data and
295        // deletion of the data.  Because it is marked `Release`, the decreasing
296        // of the reference count synchronizes with this `Acquire` fence. This
297        // means that use of the data happens before decreasing the reference
298        // count, which happens before this fence, which happens before the
299        // deletion of the data.
300        //
301        // As explained in the [Boost documentation][1],
302        //
303        // > It is important to enforce any possible access to the object in one
304        // > thread (through an existing reference) to *happen before* deleting
305        // > the object in a different thread. This is achieved by a "release"
306        // > operation after dropping a reference (any access to the object
307        // > through this reference must obviously happened before), and an
308        // > "acquire" operation before deleting the object.
309        //
310        // In particular, while the contents of an Arc are usually immutable, it's
311        // possible to have interior writes to something like a Mutex<T>. Since a
312        // Mutex is not acquired when it is deleted, we can't rely on its
313        // synchronization logic to make writes in thread A visible to a destructor
314        // running in thread B.
315        //
316        // Also note that the Acquire fence here could probably be replaced with an
317        // Acquire load, which could improve performance in highly-contended
318        // situations. See [2].
319        //
320        // [1]: (www.boost.org/doc/libs/1_55_0/doc/html/atomic/usage_examples.html)
321        // [2]: (https://github.com/rust-lang/rust/pull/41714)
322        atomic::fence(Ordering::Acquire);
323        true
324    }
325}

327fn slice_offset() -> usize {
328    fn padding_needed_for(layout: &Layout, align: usize) -> usize {
329        let len = layout.size();
330
331        // Rounded up value is:
332        //   len_rounded_up = (len + align - 1) & !(align - 1);
333        // and then we return the padding difference: `len_rounded_up - len`.
334        //
335        // We use modular arithmetic throughout:
336        //
337        // 1. align is guaranteed to be > 0, so align - 1 is always
338        //    valid.
339        //
340        // 2. `len + align - 1` can overflow by at most `align - 1`,
341        //    so the &-mask with `!(align - 1)` will ensure that in the
342        //    case of overflow, `len_rounded_up` will itself be 0.
343        //    Thus the returned padding, when added to `len`, yields 0,
344        //    which trivially satisfies the alignment `align`.
345        //
346        // (Of course, attempts to allocate blocks of memory whose
347        // size and padding overflow in the above manner should cause
348        // the allocator to yield an error anyway.)
349
350        let len_rounded_up = len.wrapping_add(align).wrapping_sub(1) & !align.wrapping_sub(1);
351        len_rounded_up.wrapping_sub(len)
352    }
353
354    let layout = Layout::new::<ArcSliceInnerMeta>();
355    layout.size() + padding_needed_for(&layout, core::mem::align_of::<ArcSlotInner>())
356}

358/// Drops the internals of the [`ArcSlice`].
359///
360/// # Safety:
361/// The pointer must point to a currently allocated [`ArcSlice`].
362unsafe fn drop_inner(p: *mut ArcSliceInnerMeta, capacity: usize) {
363    let layout = ArcSlice::layout(capacity);
364
365    // SAFETY: the pointer points to an aligned and init instance of `ArcSliceInnerMeta`
366    unsafe { drop_in_place(p) };
367
368    // SAFETY: this pointer has been allocated in the global allocator with the given layout
369    unsafe { dealloc(p.cast(), layout) };
370}

372impl Drop for ArcSlice {
373    fn drop(&mut self) {
374        let meta = unsafe { &*self.ptr.as_ptr() };
375        if meta.dec_strong() {
376            unsafe { drop_inner(self.ptr.as_ptr().cast(), meta.len) }
377        }
378    }
379}

381impl ArcSlice {
382    /// Allocates an `ArcInner<T>` with sufficient space for
383    /// a possibly-unsized inner value where the value has the layout provided.
384    pub(crate) fn new(cap: usize) -> Self {
385        // code taken and modified from `Arc::allocate_for_layout`
386
387        let arc_slice_layout = Self::layout(cap);
388
389        // safety: layout size is > 0 because it has at least 7 usizes
390        // in the metadata alone
391        debug_assert!(arc_slice_layout.size() > 0);
392        let ptr = unsafe { alloc::alloc::alloc(arc_slice_layout) };
393        if ptr.is_null() {
394            handle_alloc_error(arc_slice_layout)
395        }
396
397        // meta should be the first item in the alloc
398        let meta = ptr.cast::<ArcSliceInnerMeta>();
399        let slice = unsafe { ptr.add(slice_offset()).cast::<ArcSlotInner>() };
400
401        // SAFETY:
402        // The inner pointer is allocated and aligned, they just need to be initialised
403        unsafe {
404            let stub = slice.add(cap);
405
406            for i in 0..cap {
407                ptr::write(
408                    slice.add(i),
409                    ArcSlotInner {
410                        index: i,
411                        wake_lock: SpinMutex::new(false),
412                        links: Links::new(),
413                    },
414                );
415            }
416            ptr::write(
417                slice.add(cap),
418                ArcSlotInner {
419                    index: cap,
420                    wake_lock: SpinMutex::new(false),
421                    links: Links::new_stub(),
422                },
423            );
424
425            ptr::write(
426                meta,
427                ArcSliceInnerMeta {
428                    strong: AtomicUsize::new(1),
429                    len: cap,
430                    waker: DiatomicWaker::new(),
431                    queue: MpscQueue::new_with_stub(NonNull::new_unchecked(stub)),
432                },
433            );
434        }
435
436        Self {
437            ptr: unsafe { NonNull::new_unchecked(meta) },
438            phantom: PhantomData,
439        }
440    }
441
442    fn layout(cap: usize) -> Layout {
443        let padded = Layout::new::<ArcSlotInner>().pad_to_align();
444        let alloc_size = padded.size().checked_mul(cap + 1).unwrap();
445        let slice_layout =
446            Layout::from_size_align(alloc_size, Layout::new::<ArcSlotInner>().align()).unwrap();
447
448        Layout::new::<ArcSliceInnerMeta>()
449            .extend(slice_layout)
450            .unwrap()
451            .0
452            .pad_to_align()
453    }
454}
