19// Returns true if the character is `CR`(0x0d), `LF`(0x0a), `SP`(0x20) or `HTAB`(0x09), `DQUOTE`(0x22)
20fn is_removable_character(c: char) -> bool {
21    let cp = c as u32;
22    cp == 0x0d || cp == 0x0a || cp == 0x20 || cp == 0x09 || cp == 0x22
23}
24
25fn skip_trailing_characteres(text: &str) -> Option<usize> {
26    for (index, c) in text.chars().rev().enumerate() {
27        if !is_removable_character(c) {
28            return Some(index);
29        }
30    }
31    None
32}
33
34fn skip_starting_characteres(text: &str) -> Option<usize> {
35    for (index, c) in text.chars().enumerate() {
36        if !is_removable_character(c) {
37            return Some(index);
38        }
39    }
40    None
41}

43fn formatted_quoted_string_from(s: &str) -> Result<&str, StunError> {
44    if !QuotedStringParser::validate(QuotedStringParseLevel::QuotedText, s)
45        && !QuotedStringParser::validate(QuotedStringParseLevel::QuotedString, s)
46    {
47        return Err(StunError::new(
48            StunErrorType::InvalidParam,
49            "The text does not meet the grammar for `quoted-string`",
50        ));
51    }
52
53    // the quoted text still can have surrounding white spaces because
54    // of the `LWS` (linear white space) rule, we have o trim.
55    let s = match skip_starting_characteres(s) {
56        Some(pos) => &s[pos..],
57        None => return Ok(&s[0..0]),
58    };
59
60    let mut res = s;
61    if let Some(pos) = skip_trailing_characteres(s) {
62        res = &s[..s.len() - pos];
63    }
64
65    Ok(res)
66}

68/// A string object that meets the grammar for "quoted-string"
69/// as described in SIP: Session Initiation Protocol.
70/// [`RFC3261`](https://www.rfc-editor.org/rfc/rfc3261)
71#[derive(Debug, PartialEq, Clone, Hash, Eq, PartialOrd, Ord)]
72pub struct QuotedString(String);
73
74impl QuotedString {
75    pub fn new<S>(value: S) -> Result<Self, StunError>
76    where
77        S: AsRef<str>,
78    {
79        let val = formatted_quoted_string_from(value.as_ref())?;
80        Ok(QuotedString(String::from(val)))
81    }
82
83    /// Returns a slice representation of the "quoted-string"
84    pub fn as_str(&self) -> &str {
85        self.0.as_str()
86    }
87}

115impl<'a> crate::Decode<'a> for QuotedString {
116    fn decode(raw_value: &'a [u8]) -> Result<(Self, usize), StunError> {
117        let str = std::str::from_utf8(raw_value)?;
118        let quoted = QuotedString::try_from(str)?;
119
120        if quoted.as_str() != str {
121            return Err(StunError::new(
122                StunErrorType::InvalidParam,
123                concat!(
124                    "The text must be an unquoted sequence of `qdtext` or `quoted-pair`,",
125                    " without the double quotes and their surrounding white spaces"
126                ),
127            ));
128        }
129
130        Ok((quoted, str.len()))
131    }
132}
133
134impl Encode for QuotedString {
135    fn encode(&self, raw_value: &mut [u8]) -> Result<usize, StunError> {
136        let len = self.as_str().len();
137        check_buffer_boundaries(raw_value, len)?;
138        raw_value[..len].clone_from_slice(self.as_str().as_bytes());
139
140        Ok(len)
141    }
142}

148    #[test]
149    fn skip_trailing() {
150        assert_eq!(skip_trailing_characteres(""), None);
151        assert_eq!(skip_trailing_characteres("\u{0d}"), None);
152        assert_eq!(skip_trailing_characteres("\u{0d}a"), Some(0usize));
153        assert_eq!(skip_trailing_characteres("a\u{0d}"), Some(1usize));
154        assert_eq!(skip_trailing_characteres("a\u{0d}\u{0d}"), Some(2usize));
155    }
156
157    #[test]
158    fn skip_starting() {
159        assert_eq!(skip_starting_characteres(""), None);
160        assert_eq!(skip_starting_characteres("\u{0d}"), None);
161        assert_eq!(skip_starting_characteres("a\u{0d}"), Some(0usize));
162        assert_eq!(skip_starting_characteres("\u{0d}a"), Some(1usize));
163    }
164
165    #[test]
166    fn formatted_quoted_string_ok() {
167        // Meets the grammar
168        assert_eq!(formatted_quoted_string_from(""), Ok(""));
169        assert_eq!(formatted_quoted_string_from("  "), Ok(""));
170        assert_eq!(formatted_quoted_string_from("\"    \""), Ok(""));
171        assert_eq!(
172            formatted_quoted_string_from("  Hello world!  "),
173            Ok("Hello world!")
174        );
175        assert_eq!(
176            formatted_quoted_string_from(" \" Hello world!  \""),
177            Ok("Hello world!")
178        );
179        assert_eq!(
180            formatted_quoted_string_from("\" Test \\quoted \""),
181            Ok("Test \\quoted")
182        );
183        assert_eq!(
184            formatted_quoted_string_from(
185                "\u{0d}\u{0a} \" \u{0d}\u{0a} Test \\quoted \u{0d}\u{0a} \""
186            ),
187            Ok("Test \\quoted")
188        );
189        assert_eq!(
190            formatted_quoted_string_from("\\abfg\\h\u{fd}\u{80}\u{81}\u{82}\u{83}\u{bf}"),
191            Ok("\\abfg\\h\u{fd}\u{80}\u{81}\u{82}\u{83}\u{bf}")
192        );
193    }
194
195    #[test]
