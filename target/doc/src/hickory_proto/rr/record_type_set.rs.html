20/// A collection of record types.
21///
22/// This represents the "type bit maps" field in various records.
23#[derive(Clone)]
24pub(crate) struct RecordTypeSet {
25    types: BTreeSet<RecordType>,
26    original_encoding: Option<Vec<u8>>,
27}
28
29impl RecordTypeSet {
30    /// Construct a new set of record types.
31    pub(crate) fn new(types: impl IntoIterator<Item = RecordType>) -> Self {
32        Self {
33            types: types.into_iter().collect(),
34            original_encoding: None,
35        }
36    }
37
38    pub(crate) fn iter(&self) -> impl Iterator<Item = RecordType> + '_ {
39        self.types.iter().copied()
40    }
41
42    #[cfg(feature = "__dnssec")]
43    pub(crate) fn contains(&self, r#type: RecordType) -> bool {
44        self.types.contains(&r#type)
45    }
46}
47
48impl PartialEq for RecordTypeSet {
49    fn eq(&self, other: &Self) -> bool {
50        self.types == other.types
51    }
52}
53
54impl Eq for RecordTypeSet {}
55
56impl Hash for RecordTypeSet {
57    fn hash<H: Hasher>(&self, state: &mut H) {
58        self.types.hash(state);
59    }
60}
61
62impl fmt::Debug for RecordTypeSet {
63    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
64        let original_encoding = if self.original_encoding.is_some() {
65            &"Some(...)"
66        } else {
67            &"None"
68        };
69        f.debug_struct("RecordTypeSet")
70            .field("types", &self.types)
71            .field("original_encoding", original_encoding)
72            .finish()
73    }
74}

76impl BinEncodable for RecordTypeSet {
77    fn emit(&self, encoder: &mut BinEncoder<'_>) -> ProtoResult<()> {
78        if let Some(encoded_bytes) = &self.original_encoding {
79            return encoder.emit_vec(encoded_bytes);
80        }
81
82        let mut hash: BTreeMap<u8, Vec<u8>> = BTreeMap::new();
83
84        // collect the bitmaps
85        for rr_type in self.types.iter() {
86            let code = u16::from(*rr_type);
87            let window = (code >> 8) as u8;
88            let low = (code & 0x00FF) as u8;
89
90            let bit_map = hash.entry(window).or_default();
91            // len + left is the block in the bitmap, divided by 8 for the bits, + the bit in the current_byte
92            let index = low / 8;
93            let bit = 0b1000_0000 >> (low % 8);
94
95            // adding necessary space to the vector
96            if bit_map.len() < (index as usize + 1) {
97                bit_map.resize(index as usize + 1, 0_u8);
98            }
99
100            bit_map[index as usize] |= bit;
101        }
102
103        // output bitmaps
104        for (window, bitmap) in hash {
105            encoder.emit(window)?;
106            // the hashset should never be larger that 255 based on above logic.
107            encoder.emit(bitmap.len() as u8)?;
108            for bits in bitmap {
109                encoder.emit(bits)?;
110            }
111        }
112
113        Ok(())
114    }
115}

117impl RecordDataDecodable<'_> for RecordTypeSet {
118    fn read_data(decoder: &mut BinDecoder<'_>, length: Restrict<u16>) -> ProtoResult<Self> {
119        // 3.2.1.  Type Bit Maps Encoding
120        //
121        //  The encoding of the Type Bit Maps field is the same as that used by
122        //  the NSEC RR, described in [RFC4034].  It is explained and clarified
123        //  here for clarity.
124        //
125        //  The RR type space is split into 256 window blocks, each representing
126        //  the low-order 8 bits of the 16-bit RR type space.  Each block that
127        //  has at least one active RR type is encoded using a single octet
128        //  window number (from 0 to 255), a single octet bitmap length (from 1
129        //  to 32) indicating the number of octets used for the bitmap of the
130        //  window block, and up to 32 octets (256 bits) of bitmap.
131        //
132        //  Blocks are present in the NSEC3 RR RDATA in increasing numerical
133        //  order.
134        //
135        //     Type Bit Maps Field = ( Window Block # | Bitmap Length | Bitmap )+
136        //
137        //     where "|" denotes concatenation.
138        //
139        //  Each bitmap encodes the low-order 8 bits of RR types within the
140        //  window block, in network bit order.  The first bit is bit 0.  For
141        //  window block 0, bit 1 corresponds to RR type 1 (A), bit 2 corresponds
142        //  to RR type 2 (NS), and so forth.  For window block 1, bit 1
143        //  corresponds to RR type 257, bit 2 to RR type 258.  If a bit is set to
144        //  1, it indicates that an RRSet of that type is present for the
145        //  original owner name of the NSEC3 RR.  If a bit is set to 0, it
146        //  indicates that no RRSet of that type is present for the original
147        //  owner name of the NSEC3 RR.
148        //
149        //  Since bit 0 in window block 0 refers to the non-existing RR type 0,
150        //  it MUST be set to 0.  After verification, the validator MUST ignore
151        //  the value of bit 0 in window block 0.
152        //
153        //  Bits representing Meta-TYPEs or QTYPEs as specified in Section 3.1 of
154        //  [RFC2929] or within the range reserved for assignment only to QTYPEs
155        //  and Meta-TYPEs MUST be set to 0, since they do not appear in zone
156        //  data.  If encountered, they must be ignored upon reading.
157        //
158        //  Blocks with no types present MUST NOT be included.  Trailing zero
159        //  octets in the bitmap MUST be omitted.  The length of the bitmap of
160        //  each block is determined by the type code with the largest numerical
161        //  value, within that block, among the set of RR types present at the
162        //  original owner name of the NSEC3 RR.  Trailing octets not specified
163        //  MUST be interpreted as zero octets.
164        let mut types = BTreeSet::new();
165        let mut state = BitMapReadState::Window;
166
167        // loop through all the bytes in the bitmap
168        let bit_map_len = length.unverified();
169        let bytes = decoder.read_vec(bit_map_len as usize)?.unverified();
170        for current_byte in bytes.iter() {
171            state = match state {
172                BitMapReadState::Window => BitMapReadState::Len {
173                    window: *current_byte,
174                },
175                BitMapReadState::Len { window } => BitMapReadState::RecordType {
176                    window,
177                    len: Restrict::new(*current_byte),
178                    left: Restrict::new(*current_byte),
179                },
180                BitMapReadState::RecordType { window, len, left } => {
181                    // window is the Window Block # from above
182                    // len is the Bitmap Length
183                    // current_byte is the Bitmap
184                    let mut bit_map = *current_byte;
185
186                    // for all the bits in the current_byte
187                    for i in 0..8 {
188                        // if the current_bytes most significant bit is set
189                        if bit_map & 0b1000_0000 == 0b1000_0000 {
190                            // len - left is the block in the bitmap, times 8 for the bits, + the bit in the current_byte
191                            let low_byte: u8 = len
192                            .checked_sub(left.unverified(/*will fail as param in this call if invalid*/))
193                            .checked_mul(8)
194                            .checked_add(i)
195                            .map_err(|_| "block len or left out of bounds in NSEC(3)")?
196                            .unverified(/*any u8 is valid at this point*/);
197                            let rr_type: u16 = (u16::from(window) << 8) | u16::from(low_byte);
198                            types.insert(RecordType::from(rr_type));
199                        }
200                        // shift left and look at the next bit
201                        bit_map <<= 1;
202                    }
203
204                    // move to the next section of the bit_map
205                    let left = left
206                        .checked_sub(1)
207                        .map_err(|_| ProtoError::from("block left out of bounds in NSEC(3)"))?;
208                    if left.unverified(/*comparison is safe*/) == 0 {
209                        // we've exhausted this Window, move to the next
210                        BitMapReadState::Window
211                    } else {
212                        // continue reading this Window
213                        BitMapReadState::RecordType { window, len, left }
214                    }
215                }
216            };
217        }
218
219        Ok(Self {
220            types,
221            original_encoding: Some(bytes),
222        })
223    }
224}
225
226enum BitMapReadState {
227    Window,
228    Len {
229        window: u8,
230    },
231    RecordType {
232        window: u8,
233        len: Restrict<u8>,
234        left: Restrict<u8>,
235    },
236}

262#[cfg(test)]
263mod tests {
264    #![allow(clippy::dbg_macro, clippy::print_stdout)]
265
266    use super::*;
267
268    #[test]
269    fn test_encode_decode() {
270        let types = RecordTypeSet::new([RecordType::A, RecordType::NS]);
271
272        let mut bytes = Vec::new();
273        let mut encoder: BinEncoder<'_> = BinEncoder::new(&mut bytes);
274        types.emit(&mut encoder).expect("Encoding error");
275        let bytes = encoder.into_bytes();
276
277        let mut decoder: BinDecoder<'_> = BinDecoder::new(bytes);
278        let restrict = Restrict::new(bytes.len() as u16);
279        let read_bit_map =
280            RecordTypeSet::read_data(&mut decoder, restrict).expect("Decoding error");
281        assert_eq!(types, read_bit_map);
282    }
