40/// If a port mapping service has been seen within the last [`AVAILABILITY_TRUST_DURATION`] it will
41/// not be probed again.
42const AVAILABILITY_TRUST_DURATION: Duration = Duration::from_secs(60 * 10); // 10 minutes
43
44/// Capacity of the channel to communicate with the long-running service.
45const SERVICE_CHANNEL_CAPACITY: usize = 32; // should be plenty
46
47/// If a port mapping service has not been seen within the last [`UNAVAILABILITY_TRUST_DURATION`]
48/// we allow trying a mapping using said protocol.
49const UNAVAILABILITY_TRUST_DURATION: Duration = Duration::from_secs(5);

70// Cannot have backtrace due to Clone bound
71// #[nested_enum_utils::common_fields({
72//     backtrace: Option<snafu::Backtrace>,
73// })]
74#[allow(missing_docs)]
75#[derive(Debug, Clone, Snafu)]
76#[non_exhaustive]
77pub enum ProbeError {
78    #[snafu(display("Mapping channel is full"))]
79    ChannelFull,
80    #[snafu(display("Mapping channel is closed"))]
81    ChannelClosed,
82    #[snafu(display("No gateway found for probe"))]
83    NoGateway,
84    #[snafu(display("gateway found is ipv6, ignoring"))]
85    Ipv6Gateway,
86    #[snafu(display("Probe task stopped. is_panic: {is_panic}, is_cancelled: {is_cancelled}"))]
87    Join { is_panic: bool, is_cancelled: bool },
88}

90#[derive(derive_more::Debug)]
91enum Message {
92    /// Attempt to get a mapping if the local port is set but there is no mapping.
93    ProcureMapping,
94    /// Request to update the local port.
95    ///
96    /// The resulting external address can be obtained subscribing using
97    /// [`Client::watch_external_address`].
98    /// A value of `None` will deactivate port mapping.
99    UpdateLocalPort { local_port: Option<NonZeroU16> },
100    /// Request to probe the port mapping protocols.
101    ///
102    /// The requester should wait for the result at the [`oneshot::Receiver`] counterpart of the
103    /// [`oneshot::Sender`].
104    Probe {
105        /// Sender side to communicate the result of the probe.
106        #[debug("_")]
107        result_tx: oneshot::Sender<Result<ProbeOutput, ProbeError>>,
108    },
109}

160    /// Creates a new port mapping client with a previously created metrics collector.
161    pub fn with_metrics(config: Config, metrics: Arc<Metrics>) -> Self {
162        let (service_tx, service_rx) = mpsc::channel(SERVICE_CHANNEL_CAPACITY);
163
164        let (service, watcher) = Service::new(config, service_rx, metrics.clone());
165
166        let handle = AbortOnDropHandle::new(tokio::spawn(
167            async move { service.run().await }.instrument(info_span!("portmapper.service")),
168        ));
169
170        Client {
171            port_mapping: watcher,
172            service_tx,
173            metrics,
174            _service_handle: std::sync::Arc::new(handle),
175        }
176    }

178    /// Request a probe to the port mapping protocols.
179    ///
180    /// Returns the [`oneshot::Receiver`] used to obtain the result of the probe.
181    pub fn probe(&self) -> oneshot::Receiver<Result<ProbeOutput, ProbeError>> {
182        let (result_tx, result_rx) = oneshot::channel();
183
184        if let Err(e) = self.service_tx.try_send(Message::Probe { result_tx }) {
185            use mpsc::error::TrySendError::*;
186
187            // recover the sender and return the error there
188            let (result_tx, e) = match e {
189                Full(Message::Probe { result_tx }) => (result_tx, ChannelFullSnafu.build()),
190                Closed(Message::Probe { result_tx }) => (result_tx, ChannelClosedSnafu.build()),
191                Full(_) | Closed(_) => unreachable!("Sent value is a probe."),
192            };
193
194            // sender was just created. If it's dropped we have two send error and are likely
195            // shutting down
196            // NOTE: second Err is infallible match due to being the sent value
197            if let Err(Err(e)) = result_tx.send(Err(e)) {
198                trace!("Failed to request probe: {e}")
199            }
200        }
201        result_rx
202    }
203
204    /// Try to get a mapping for the last local port if there isn't one already.
205    pub fn procure_mapping(&self) {
206        // requester can't really do anything with this error if returned, so we log it
207        if let Err(e) = self.service_tx.try_send(Message::ProcureMapping) {
208            trace!("Failed to request mapping {e}")
209        }
210    }
211
212    /// Update the local port.
213    ///
214    /// If the port changes, this will trigger a port mapping attempt.
215    pub fn update_local_port(&self, local_port: NonZeroU16) {
216        let local_port = Some(local_port);
217        // requester can't really do anything with this error if returned, so we log it
218        if let Err(e) = self
219            .service_tx
220            .try_send(Message::UpdateLocalPort { local_port })
221        {
222            trace!("Failed to update local port {e}")
223        }
224    }
225
226    /// Deactivate port mapping.
227    pub fn deactivate(&self) {
228        // requester can't really do anything with this error if returned, so we log it
229        if let Err(e) = self
230            .service_tx
231            .try_send(Message::UpdateLocalPort { local_port: None })
232        {
233            trace!("Failed to deactivate port mapping {e}")
234        }
235    }

239        self.port_mapping.clone()
240    }

244        &self.metrics
245    }

248/// Port mapping protocol information obtained during a probe.
249#[derive(Debug)]
250struct Probe {
251    /// When was the probe last updated.
252    last_probe: Instant,
253    /// The last [`upnp::Gateway`] and when was it last seen.
254    last_upnp_gateway_addr: Option<(upnp::Gateway, Instant)>,
255    /// Last time PCP was seen.
256    last_pcp: Option<Instant>,
257    /// Last time NAT-PMP was seen.
258    last_nat_pmp: Option<Instant>,
259}
260
261impl Probe {
262    /// An empty probe set to `now`.
263    fn empty() -> Self {
264        Self {
265            last_probe: Instant::now(),
266            last_upnp_gateway_addr: None,
267            last_pcp: None,
268            last_nat_pmp: None,
269        }
270    }
271    /// Create a new probe based on a previous output.
272    async fn from_output(
273        config: Config,
274        output: ProbeOutput,
275        local_ip: Ipv4Addr,
276        gateway: Ipv4Addr,
277        metrics: Arc<Metrics>,
278    ) -> Probe {
279        let ProbeOutput { upnp, pcp, nat_pmp } = output;
280        let Config {
281            enable_upnp,
282            enable_pcp,
283            enable_nat_pmp,
284        } = config;
285        let mut upnp_probing_task = util::MaybeFuture {
286            inner: (enable_upnp && !upnp).then(|| {
287                let metrics = metrics.clone();
288                Box::pin(async move {
289                    upnp::probe_available(&metrics)
290                        .await
291                        .map(|addr| (addr, Instant::now()))
292                })
293            }),
294        };
295
296        let mut pcp_probing_task = util::MaybeFuture {
297            inner: (enable_pcp && !pcp).then(|| {
298                let metrics = metrics.clone();
299                Box::pin(async move {
300                    metrics.pcp_probes.inc();
301                    pcp::probe_available(local_ip, gateway)
302                        .await
303                        .then(Instant::now)
304                })
305            }),
306        };
307
308        let mut nat_pmp_probing_task = util::MaybeFuture {
309            inner: (enable_nat_pmp && !nat_pmp).then(|| {
310                Box::pin(async {
311                    nat_pmp::probe_available(local_ip, gateway)
312                        .await
313                        .then(Instant::now)
314                })
315            }),
316        };
317
318        if upnp_probing_task.inner.is_some() {
319            metrics.upnp_probes.inc();
320        }
321
322        let mut upnp_done = upnp_probing_task.inner.is_none();
323        let mut pcp_done = pcp_probing_task.inner.is_none();
324        let mut nat_pmp_done = nat_pmp_probing_task.inner.is_none();
325
326        let mut probe = Probe::empty();
327
328        while !upnp_done || !pcp_done || !nat_pmp_done {
329            tokio::select! {
330                last_upnp_gateway_addr = &mut upnp_probing_task, if !upnp_done => {
331                    trace!("tick: upnp probe ready");
332                    probe.last_upnp_gateway_addr = last_upnp_gateway_addr;
333                    upnp_done = true;
334                },
335                last_nat_pmp = &mut nat_pmp_probing_task, if !nat_pmp_done => {
336                    trace!("tick: nat_pmp probe ready");
337                    probe.last_nat_pmp = last_nat_pmp;
338                    nat_pmp_done = true;
339                },
340                last_pcp = &mut pcp_probing_task, if !pcp_done => {
341                    trace!("tick: pcp probe ready");
342                    probe.last_pcp = last_pcp;
343                    pcp_done = true;
344                },
345            }
346        }
347
348        probe
349    }
350
351    /// Returns a [`ProbeOutput`] indicating which services can be considered available.
352    fn output(&self) -> ProbeOutput {
353        let now = Instant::now();
354
355        // check if the last UPnP gateway is valid
356        let upnp = self
357            .last_upnp_gateway_addr
358            .as_ref()
359            .map(|(_gateway_addr, last_probed)| *last_probed + AVAILABILITY_TRUST_DURATION > now)
360            .unwrap_or_default();
361
362        let pcp = self
363            .last_pcp
364            .as_ref()
365            .map(|last_probed| *last_probed + AVAILABILITY_TRUST_DURATION > now)
366            .unwrap_or_default();
367
368        let nat_pmp = self
369            .last_nat_pmp
370            .as_ref()
371            .map(|last_probed| *last_probed + AVAILABILITY_TRUST_DURATION > now)
372            .unwrap_or_default();
373
374        ProbeOutput { upnp, pcp, nat_pmp }
375    }
376
377    /// Updates a probe with the `Some` values of another probe that is _assumed_ newer.
378    fn update(&mut self, probe: Probe, metrics: &Arc<Metrics>) {
379        let Probe {
380            last_probe,
381            last_upnp_gateway_addr,
382            last_pcp,
383            last_nat_pmp,
384        } = probe;
385        if last_upnp_gateway_addr.is_some() {
386            metrics.upnp_available.inc();
387            let new_gateway = last_upnp_gateway_addr
388                .as_ref()
389                .map(|(addr, _last_seen)| addr);
390            let old_gateway = self
391                .last_upnp_gateway_addr
392                .as_ref()
393                .map(|(addr, _last_seen)| addr);
394            if new_gateway != old_gateway {
395                metrics.upnp_gateway_updated.inc();
396                debug!(
397                    "upnp gateway changed {:?} -> {:?}",
398                    old_gateway
399                        .map(|gw| gw.to_string())
400                        .unwrap_or("None".into()),
401                    new_gateway
402                        .map(|gw| gw.to_string())
403                        .unwrap_or("None".into())
404                )
405            };
406            self.last_upnp_gateway_addr = last_upnp_gateway_addr;
407        }
408        if last_pcp.is_some() {
409            metrics.pcp_available.inc();
410            self.last_pcp = last_pcp;
411        }
412        if last_nat_pmp.is_some() {
413            self.last_nat_pmp = last_nat_pmp;
414        }
415
416        self.last_probe = last_probe;
417    }
418}

423/// A port mapping client.
424#[derive(Debug)]
425pub struct Service {
426    config: Config,
427    /// Local port to map.
428    local_port: Option<NonZeroU16>,
429    /// Channel over which the service is informed of messages.
430    ///
431    /// The service will stop when all senders are gone.
432    rx: mpsc::Receiver<Message>,
433    /// Currently active mapping.
434    current_mapping: CurrentMapping,
435    /// Last updated probe.
436    full_probe: Probe,
437    /// Task attempting to get a port mapping.
438    ///
439    /// This task will be cancelled if a request to set the local port arrives before it's
440    /// finished.
441    mapping_task: Option<AbortOnDropHandle<Result<mapping::Mapping, mapping::Error>>>,
442    /// Task probing the necessary protocols.
443    ///
444    /// Requests for a probe that arrive while this task is still in progress will receive the same
445    /// result.
446    probing_task: Option<(AbortOnDropHandle<Probe>, Vec<oneshot::Sender<ProbeResult>>)>,
447    metrics: Arc<Metrics>,
448}
449
450impl Service {
451    fn new(
452        config: Config,
453        rx: mpsc::Receiver<Message>,
454        metrics: Arc<Metrics>,
455    ) -> (Self, watch::Receiver<Option<SocketAddrV4>>) {
456        let (current_mapping, watcher) = CurrentMapping::new(metrics.clone());
457        let mut full_probe = Probe::empty();
458        if let Some(in_the_past) = full_probe
459            .last_probe
460            .checked_sub(AVAILABILITY_TRUST_DURATION)
461        {
462            // we want to do a first full probe, so set is as expired on start-up
463            full_probe.last_probe = in_the_past;
464        }
465        let service = Service {
466            config,
467            local_port: None,
468            rx,
469            current_mapping,
470            full_probe,
471            mapping_task: None,
472            probing_task: None,
473            metrics,
474        };
475
476        (service, watcher)
477    }
478
479    /// Clears the current mapping and releases it.
480    async fn invalidate_mapping(&mut self) {
481        if let Some(old_mapping) = self.current_mapping.update(None) {
482            if let Err(e) = old_mapping.release().await {
483                debug!("failed to release mapping {e}");
484            }
485        }
486    }
487
488    async fn run(mut self) {
489        debug!("portmap starting");
490        loop {
491            tokio::select! {
492                msg = self.rx.recv() => {
493                    trace!("tick: msg {msg:?}");
494                    match msg {
495                        Some(msg) => {
496                            self.handle_msg(msg).await;
497                        },
498                        None => {
499                            debug!("portmap service channel dropped. Likely shutting down.");
500                            break;
501                        }
502                    }
503                }
504                mapping_result = util::MaybeFuture{ inner: self.mapping_task.as_mut() } => {
505                    trace!("tick: mapping ready");
506                    // regardless of outcome, the task is finished, clear it
507                    self.mapping_task = None;
508                    // there isn't really a way to react to a join error here. Flatten it to make
509                    // it easier to work with
510                    self.on_mapping_result(mapping_result);
511                }
512                probe_result = util::MaybeFuture{ inner: self.probing_task.as_mut().map(|(fut, _rec)| fut) } => {
513                    trace!("tick: probe ready");
514                    // retrieve the receivers and clear the task
515                    let receivers = self.probing_task.take().expect("is some").1;
516                    let probe_result = probe_result.map_err(|e| JoinSnafu { is_panic: e.is_panic(), is_cancelled: e.is_cancelled() }.build());
517                    self.on_probe_result(probe_result, receivers);
518                }
519                Some(event) = self.current_mapping.next() => {
520                    trace!("tick: mapping event {event:?}");
521                    match event {
522                        current_mapping::Event::Renew { external_ip, external_port } | current_mapping::Event::Expired { external_ip, external_port } => {
523                            self.get_mapping(Some((external_ip, external_port)));
524                        },
525                    }
526
527                }
528            }
529        }
530    }
531
532    fn on_probe_result(
533        &mut self,
534        result: Result<Probe, ProbeError>,
535        receivers: Vec<oneshot::Sender<ProbeResult>>,
536    ) {
537        let result = result.map(|probe| {
538            self.full_probe.update(probe, &self.metrics);
539            // TODO(@divma): the gateway of the current mapping could have changed. Tailscale
540            // still assumes the current mapping is valid/active and will return it even after
541            // this
542            let output = self.full_probe.output();
543            trace!(?output, "probe output");
544            output
545        });
546        for tx in receivers {
547            // ignore the error. If the receiver is no longer there we don't really care
548            let _ = tx.send(result.clone());
549        }
550    }
551
552    fn on_mapping_result(
553        &mut self,
554        result: Result<Result<mapping::Mapping, mapping::Error>, tokio::task::JoinError>,
555    ) {
556        match result {
557            Ok(Ok(mapping)) => {
558                self.current_mapping.update(Some(mapping));
559            }
560            Ok(Err(e)) => {
561                debug!("failed to get a port mapping {e}");
562                self.metrics.mapping_failures.inc();
563            }
564            Err(e) => {
565                debug!("failed to get a port mapping {e}");
566                self.metrics.mapping_failures.inc();
567            }
568        }
569    }
570
571    async fn handle_msg(&mut self, msg: Message) {
572        match msg {
573            Message::ProcureMapping => self.update_local_port(self.local_port).await,
574            Message::UpdateLocalPort { local_port } => self.update_local_port(local_port).await,
575            Message::Probe { result_tx } => self.probe_request(result_tx),
576        }
577    }
578
579    /// Updates the local port of the port mapping service.
580    ///
581    /// If the port changed, any port mapping task is cancelled. If the new port is some, it will
582    /// start a new port mapping task.
583    async fn update_local_port(&mut self, local_port: Option<NonZeroU16>) {
584        // ignore requests to update the local port in a way that does not produce a change
585        if local_port != self.local_port {
586            self.metrics.local_port_updates.inc();
587            let old_port = std::mem::replace(&mut self.local_port, local_port);
588
589            // clear the current mapping task if any
590
591            let dropped_task = self.mapping_task.take();
592            // check if the dropped task had finished to reduce log noise
593            let did_cancel = dropped_task
594                .map(|task| !task.is_finished())
595                .unwrap_or_default();
596
597            if did_cancel {
598                debug!(
599                    "canceled mapping task due to local port update. Old: {:?} New: {:?}",
600                    old_port, self.local_port
601                )
602            }
603
604            // get the current external port if any to try to get it again
605            let external_addr = self.current_mapping.external();
606
607            // since the port has changed, the current mapping is no longer valid and should be
608            // released
609
610            if external_addr.is_some() {
611                self.invalidate_mapping().await;
612            }
613
614            // start a new mapping task to account for the new port if necessary
615            self.get_mapping(external_addr);
616        } else if self.current_mapping.external().is_none() {
617            // if the local port has not changed, but there is no active mapping try to get one
618            self.get_mapping(None)
619        }
620    }
621
622    fn get_mapping(&mut self, external_addr: Option<(Ipv4Addr, NonZeroU16)>) {
623        if let Some(local_port) = self.local_port {
624            self.metrics.mapping_attempts.inc();
625
626            let (local_ip, gateway) = match ip_and_gateway() {
627                Ok(ip_and_gw) => ip_and_gw,
628                Err(e) => return debug!("can't get mapping: {e}"),
629            };
630
631            let ProbeOutput { upnp, pcp, nat_pmp } = self.full_probe.output();
632
633            debug!("getting a port mapping for {local_ip}:{local_port} -> {external_addr:?}");
634            let recently_probed =
635                self.full_probe.last_probe + UNAVAILABILITY_TRUST_DURATION > Instant::now();
636            // strategy:
637            // 1. check the available services and prefer pcp, then nat_pmp then upnp since it's
638            //    the most unreliable, but possibly the most deployed one
639            // 2. if no service was available, fallback to upnp if enabled, followed by pcp and
640            //    nat_pmp
641            self.mapping_task = if pcp {
642                // try pcp if available first
643                let task = mapping::Mapping::new_pcp(local_ip, local_port, gateway, external_addr);
644                Some(AbortOnDropHandle::new(tokio::spawn(
645                    task.instrument(info_span!("pcp")),
646                )))
647            } else if nat_pmp {
648                // next nat_pmp if available
649                let task =
650                    mapping::Mapping::new_nat_pmp(local_ip, local_port, gateway, external_addr);
651                Some(AbortOnDropHandle::new(tokio::spawn(
652                    task.instrument(info_span!("pmp")),
653                )))
654            } else if upnp || self.config.enable_upnp {
655                // next upnp if available or enabled
656                let external_port = external_addr.map(|(_addr, port)| port);
657                let gateway = self
658                    .full_probe
659                    .last_upnp_gateway_addr
660                    .as_ref()
661                    .map(|(gateway, _last_seen)| gateway.clone());
662                let task = mapping::Mapping::new_upnp(local_ip, local_port, gateway, external_port);
663
664                Some(AbortOnDropHandle::new(tokio::spawn(
665                    task.instrument(info_span!("upnp")),
666                )))
667            } else if !recently_probed && self.config.enable_pcp {
668                // if no service is available and the default fallback (upnp) is disabled, try pcp
669                // first
670                let task = mapping::Mapping::new_pcp(local_ip, local_port, gateway, external_addr);
671
672                Some(AbortOnDropHandle::new(tokio::spawn(
673                    task.instrument(info_span!("pcp")),
674                )))
675            } else if !recently_probed && self.config.enable_nat_pmp {
676                // finally try nat_pmp if enabled
677                let task =
678                    mapping::Mapping::new_nat_pmp(local_ip, local_port, gateway, external_addr);
679                Some(AbortOnDropHandle::new(tokio::spawn(
680                    task.instrument(info_span!("pmp")),
681                )))
682            } else {
683                // give up
684                return;
685            }
686        }
687    }
688
689    /// Handles a probe request.
690    ///
691    /// If there is a task getting a probe, the receiver will be added with any other waiting for a
692    /// result. If no probe is underway, a result can be returned immediately if it's still
693    /// considered valid. Otherwise, a new probe task will be started.
694    fn probe_request(&mut self, result_tx: oneshot::Sender<Result<ProbeOutput, ProbeError>>) {
695        match self.probing_task.as_mut() {
696            Some((_task_handle, receivers)) => receivers.push(result_tx),
697            None => {
698                let probe_output = self.full_probe.output();
699                if probe_output.all_available() {
700                    // we don't care if the requester is no longer there
701                    let _ = result_tx.send(Ok(probe_output));
702                } else {
703                    self.metrics.probes_started.inc();
704
705                    let (local_ip, gateway) = match ip_and_gateway() {
706                        Ok(ip_and_gw) => ip_and_gw,
707                        Err(e) => {
708                            // there is no guarantee this will be displayed, so log it anyway
709                            debug!("could not start probe: {e}");
710                            let _ = result_tx.send(Err(e));
711                            return;
712                        }
713                    };
714
715                    let config = self.config.clone();
716                    let metrics = self.metrics.clone();
717                    let handle = tokio::spawn(
718                        async move {
719                            Probe::from_output(config, probe_output, local_ip, gateway, metrics)
720                                .await
721                        }
722                        .instrument(info_span!("portmapper.probe")),
723                    );
724                    let receivers = vec![result_tx];
725                    self.probing_task = Some((AbortOnDropHandle::new(handle), receivers));
726                }
727            }
728        }
729    }
730}
731
732/// Gets the local ip and gateway address for port mapping.
733fn ip_and_gateway() -> Result<(Ipv4Addr, Ipv4Addr), ProbeError> {
734    let Some(HomeRouter { gateway, my_ip }) = HomeRouter::new() else {
735        return Err(NoGatewaySnafu.build());
736    };
737
738    let local_ip = match my_ip {
739        Some(std::net::IpAddr::V4(ip))
740            if !ip.is_unspecified() && !ip.is_loopback() && !ip.is_multicast() =>
741        {
742            ip
743        }
744        other => {
745            debug!("no address suitable for port mapping found ({other:?}), using localhost");
746            Ipv4Addr::LOCALHOST
747        }
748    };
749
750    let std::net::IpAddr::V4(gateway) = gateway else {
751        return Err(Ipv6GatewaySnafu.build());
752    };
753
754    Ok((local_ip, gateway))
755}
