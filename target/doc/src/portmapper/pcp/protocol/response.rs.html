10/// ResultCode in a [`Response`] when it's successful.
11#[derive(Debug, Clone, Copy, PartialEq, Eq, TryFromPrimitive, IntoPrimitive)]
12#[repr(u8)]
13pub enum SuccessCode {
14    /// Result code indicating a successful response.
15    Success = 0,
16}

18/// ResultCode in a [`Response`], when said code is an error.
19/// [`SuccessCode`] handles the success case.
20///
21/// Refer to [RFC 6887 Result Codes](https://datatracker.ietf.org/doc/html/rfc6887#section-7.4)
22// NOTE: docs for each variant are largely adapted from the RFC's description of each code.
23#[derive(Debug, Clone, Copy, PartialEq, Eq, TryFromPrimitive, IntoPrimitive, Display)]
24#[repr(u8)]
25pub enum ErrorCode {
26    /// The version number at the start of the PCP Request header is not recognized by the PCP
27    /// server.
28    #[display("sent version is not supported")]
29    UnsuppVersion = 1,
30    /// The requested operation is disabled for this PCP client, or the PCP client requested an
31    /// operation that cannot be fulfilled by the PCP server's security policy.
32    #[display("operation not authorized")]
33    NotAuthorized = 2,
34    /// The request could not be successfully parsed.
35    #[display("could not parse the request")]
36    MalformedRequest = 3,
37    /// Unsupported Opcode.
38    #[display("opcode is not supported")]
39    UnsuppOpcode = 4,
40    /// Unsupported option. This error only occurs if the option is in the mandatory-to-process
41    /// range.
42    #[display("option is not supported")]
43    UnsuppOption = 5,
44    /// Malformed option (e.g., appears too many times, invalid length).
45    #[display("option could not be parsed")]
46    MalformedOption = 6,
47    /// The PCP server or the device it controls is experiencing a network failure of some sort
48    /// (e.g., has not yet obtained an external IP address). This is a short lifetime error.
49    #[display("spurious network failure")]
50    NetworkFailure = 7,
51    /// Request is well-formed and valid, but the server has insufficient resources to complete the
52    /// requested operation at this time. This is a short lifetime error.
53    #[display("not enough resources for this request")]
54    NoResources = 8,
55    /// Unsupported transport protocol, e.g., SCTP in a NAT that handles only UDP and TCP. This is
56    /// a long lifetime error.
57    #[display("unsupported protocol")]
58    UnsuppProtocol = 9,
59    /// This attempt to create a new mapping would exceed this subscriber's port quota. This is a
60    /// short lifetime error.
61    #[display("quota exceeded")]
62    UserExQuota = 10,
63    /// The suggested external port and/or external address cannot be provided.
64    #[display("requested external address cannot be provided")]
65    CannotProvideExternal = 11,
66    /// The source IP address of the request packet does not match the contents of the PCP Client's
67    /// IP Address field.
68    #[display("sender and declared ip do not match")]
69    AddressMismatch = 12,
70    /// The PCP server was not able to create the filters in this request.
71    #[display("excessive reporte peers in filter option")]
72    ExcessiveRemotePeers = 13,
73}

77/// Result code of a PCP response.
78#[derive(Debug)]
79pub enum ResultCode {
80    /// A success result code. See [`SuccessCode`].
81    Success,
82    /// An error code. See [`ErrorCode`].
83    Error(ErrorCode),
84}
85
86impl TryFrom<u8> for ResultCode {
87    type Error = TryFromPrimitiveError<ErrorCode>;
88
89    fn try_from(value: u8) -> Result<Self, TryFromPrimitiveError<ErrorCode>> {
90        if let Ok(SuccessCode::Success) = SuccessCode::try_from(value) {
91            Ok(ResultCode::Success)
92        } else {
93            ErrorCode::try_from(value).map(ResultCode::Error)
94        }
95    }
96}
97
98impl From<ResultCode> for u8 {
99    fn from(value: ResultCode) -> Self {
100        match value {
101            ResultCode::Success => SuccessCode::Success.into(),
102            ResultCode::Error(e) => e.into(),
103        }
104    }
105}

107/// A PCP successful Response/Notification.
108///
109/// See [RFC 6887 Response Header](https://datatracker.ietf.org/doc/html/rfc6887#section-7.2)
110// NOTE: first two fields are *currently* not used, but are useful for debug purposes
111#[allow(unused)]
112#[derive(Debug, PartialEq, Eq)]
113pub struct Response {
114    /// Lifetime in seconds that can be assumed by this response.
115    ///
116    /// For map requests, this lifetime is how long to assume a mapping will last.
117    pub lifetime_seconds: u32,
118    /// Epoch time of the server.
119    pub epoch_time: u32,
120    /// Data of the resoponse.
121    pub data: OpcodeData,
122}
123
124/// Errors that can occur when decoding a [`Response`] from a server.
125#[allow(missing_docs)]
126#[non_exhaustive]
127#[common_fields({
128    backtrace: Option<Backtrace>,
129})]
130#[derive(Debug, Snafu)]
131pub enum DecodeError {
132    /// Request is too short or is otherwise malformed.
133    #[snafu(display("Response is malformed"))]
134    Malformed {},
135    /// The [`Response::RESPONSE_INDICATOR`] is not present.
136    #[snafu(display("Packet does not appear to be a response"))]
137    NotAResponse {},
138    /// The received opcode is not recognized.
139    #[snafu(display("Invalid Opcode received"))]
140    InvalidOpcode {},
141    /// The received version is not recognized.
142    #[snafu(display("Invalid version received"))]
143    InvalidVersion {},
144    /// The received result code is not recognized.
145    #[snafu(display("Invalid result code received"))]
146    InvalidResultCode {},
147    /// The received opcode data could not be decoded.
148    #[snafu(display("Invalid opcode data received"))]
149    InvalidOpcodeData {},
150}
151
152#[derive(Debug, Snafu)]
153pub enum Error {
154    #[snafu(transparent)]
155    DecodeError { source: DecodeError },
156    #[snafu(transparent)]
157    ErrorCode { source: ErrorCode },
158}
159
160impl Response {
161    /// Max size of a PCP packet as indicated in
162    /// [RFC 6887 Common Request and Response Header Format](https://datatracker.ietf.org/doc/html/rfc6887#section-7)
163    pub const MAX_SIZE: usize = 1100;
164
165    /// Minimum size of an encoded [`Response`] sent by a server to this client.
166    pub const MIN_SIZE: usize = // parts
167        1 + // version
168        1 + // opcode ORd with [`Response::RESPONSE_INDICATOR`]
169        1 + // reserved
170        1 + // result code
171        4 + // lifetime
172        4 + // epoch time
173        12; // reserved
174
175    /// Indicator ORd into the [`Opcode`] to indicate a response packet.
176    pub const RESPONSE_INDICATOR: u8 = 1u8 << 7;
177
178    /// Decode a response.
179    pub fn decode(buf: &[u8]) -> Result<Self, Error> {
180        snafu::ensure!(
181            Self::MIN_SIZE <= buf.len() && buf.len() <= Self::MAX_SIZE,
182            MalformedSnafu
183        );
184
185        let _version: Version = buf[0].try_into().map_err(|_| InvalidVersionSnafu.build())?;
186
187        let opcode = buf[1];
188        snafu::ensure!(
189            opcode & Self::RESPONSE_INDICATOR == Self::RESPONSE_INDICATOR,
190            NotAResponseSnafu
191        );
192        let opcode: Opcode = (opcode & !Self::RESPONSE_INDICATOR)
193            .try_into()
194            .map_err(|_| InvalidOpcodeSnafu.build())?;
195
196        // buf[2] reserved
197
198        // return early if the result code is an error
199        let result_code: ResultCode = buf[3]
200            .try_into()
201            .map_err(|_| InvalidResultCodeSnafu.build())?;
202        match result_code {
203            ResultCode::Success => {}
204            ResultCode::Error(error_code) => return Err(error_code.into()),
205        }
206
207        let lifetime_bytes = buf[4..8].try_into().expect("slice has the right len");
208        let lifetime_seconds = u32::from_be_bytes(lifetime_bytes);
209
210        let epoch_bytes = buf[8..12].try_into().expect("slice has the right len");
211        let epoch_time = u32::from_be_bytes(epoch_bytes);
212
213        // buf[12..24] reserved
214
215        let data =
216            OpcodeData::decode(opcode, &buf[24..]).map_err(|_| InvalidOpcodeDataSnafu.build())?;
217
218        Ok(Response {
219            lifetime_seconds,
220            epoch_time,
221            data,
222        })
223    }
224
225    #[cfg(test)]
226    fn random<R: rand::Rng>(opcode: Opcode, rng: &mut R) -> Self {
227        let data = OpcodeData::random(opcode, rng);
228        Self {
229            lifetime_seconds: rng.gen(),
230            epoch_time: rng.gen(),
231            data,
232        }
233    }
234
235    #[cfg(test)]
236    fn encode(&self) -> Vec<u8> {
237        let Response {
238            lifetime_seconds,
239            epoch_time,
240            data,
241        } = self;
242        let mut buf = Vec::with_capacity(Self::MIN_SIZE);
243        // version
244        buf.push(Version::Pcp.into());
245        // response indicator and opcode
246        let opcode: u8 = data.opcode().into();
247        buf.push(Response::RESPONSE_INDICATOR | opcode);
248        // reserved
249        buf.push(0);
250        // result code
251        buf.push(ResultCode::Success.into());
252        // lifetime
253        for b in lifetime_seconds.to_be_bytes() {
254            buf.push(b);
255        }
256        // epoch
257        for b in epoch_time.to_be_bytes() {
258            buf.push(b);
259        }
260        // reserved
261        for _ in 12..Response::MIN_SIZE {
262            buf.push(0)
263        }
264        data.encode_into(&mut buf);
265        buf
266    }
267}
