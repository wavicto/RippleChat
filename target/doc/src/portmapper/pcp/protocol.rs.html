13// PCP and NAT-PMP share same ports, reassigned by IANA from the older version to the new one. See
14// <https://datatracker.ietf.org/doc/html/rfc6887#section-19>
15
16/// Port to use when acting as a server. This is the one we direct requests to.
17pub const SERVER_PORT: u16 = 5351;
18
19/// NAT-PMP/PCP Version
20#[derive(Debug, Clone, Copy, PartialEq, Eq, IntoPrimitive, TryFromPrimitive)]
21#[repr(u8)]
22pub enum Version {
23    /// PCP Version according to [RFC 6887 Version Negotiation](https://datatracker.ietf.org/doc/html/rfc6887#section-9)
24    // Version 2
25    Pcp = 2,
26}

28/// Opcode as defined in [RFC 6887 IANA Considerations](https://datatracker.ietf.org/doc/html/rfc6887#section-19)
29// NOTE: PEER is not used, therefore not implemented.
30#[derive(Debug, Clone, Copy, PartialEq, Eq, IntoPrimitive, TryFromPrimitive)]
31#[repr(u8)]
32pub enum Opcode {
33    /// Announce Opcode.
34    ///
35    /// Used by the server to announce changes to clients. These include restarts
36    /// (indicating loss of state) and changes to mappings and external ip addresses.
37    ///
38    /// See [RFC 6887 ANNOUNCE Opcode](https://datatracker.ietf.org/doc/html/rfc6887#section-14.1)
39    Announce = 0,
40    /// Map Opcode,
41    ///
42    /// Used to deal with endpoint-idependent mappings.
43    ///
44    /// See [RFC 6887 MAP Opcode](https://datatracker.ietf.org/doc/html/rfc6887#section-11)
45    Map = 1,
46}
