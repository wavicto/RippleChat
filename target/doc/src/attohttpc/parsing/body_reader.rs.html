9#[derive(Debug)]
10pub enum BodyReader {
11    Chunked(ChunkedReader<BaseStream>),
12    Length(Take<BufReader<BaseStream>>),
13    Close(BufReader<BaseStream>),
14}

16impl Read for BodyReader {
17    #[inline]
18    fn read(&mut self, buf: &mut [u8]) -> io::Result<usize> {
19        match self {
20            BodyReader::Chunked(r) => r.read(buf),
21            BodyReader::Length(r) => r.read(buf),
22            BodyReader::Close(r) => r.read(buf),
23        }
24    }
25}

27impl BufRead for BodyReader {
28    #[inline]

37    #[inline]

47fn is_chunked(headers: &HeaderMap) -> bool {
48    headers
49        .get_all(TRANSFER_ENCODING)
50        .into_iter()
51        .filter_map(|val| val.to_str().ok())
52        .any(|val| {
53            val.split(',')
54                .map(|s| s.trim())
55                .any(|s| s.eq_ignore_ascii_case("chunked"))
56        })
57}

59fn parse_content_length(val: &HeaderValue) -> Result<u64> {
60    let val = val.to_str().map_err(|_| InvalidResponseKind::ContentLength)?;
61    let val = val.parse::<u64>().map_err(|_| InvalidResponseKind::ContentLength)?;
62    Ok(val)
63}
64
65fn is_content_length(headers: &HeaderMap) -> Result<Option<u64>> {
66    let mut last = None;
67    for val in headers.get_all(CONTENT_LENGTH) {
68        let val = parse_content_length(val)?;
69        last = Some(match last {
70            None => val,
71            Some(last) if last == val => val,
72            _ => {
73                return Err(InvalidResponseKind::ContentLength.into());
74            }
75        });
76    }
77    Ok(last)
78}
79
80impl BodyReader {
81    pub fn new(headers: &HeaderMap, reader: BufReader<BaseStream>) -> Result<BodyReader> {
82        if is_chunked(headers) {
83            debug!("creating a chunked body reader");
84            Ok(BodyReader::Chunked(ChunkedReader::new(reader)))
85        } else if let Some(val) = is_content_length(headers)? {
86            debug!("creating a length body reader");
87            Ok(BodyReader::Length(reader.take(val)))
88        } else {
89            debug!("creating close reader");
90            Ok(BodyReader::Close(reader))
91        }
92    }
93}
