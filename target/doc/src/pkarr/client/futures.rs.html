11/// Publish to both DHT and configured Relays, resolve after queries on both network are complete.
12///
13/// If Relays network completes succeeds first, it is possible for the query to the
14/// DHT to fail because CAS is outdated, in that case, we should ignore that error.
15pub async fn publish_both_networks(
16    dht_future: impl Future<Output = Result<(), PublishError>> + Send + 'static,
17    relays_future: impl Future<Output = Result<(), PublishError>> + Send + 'static,
18) -> Result<(), PublishError> {
19    SelectFuture {
20        first_result: None,
21        dht_future: dht_future.boxed(),
22        relays_future: relays_future.boxed(),
23    }
24    .await
25}

27pub struct SelectFuture {
28    first_result: Option<(Network, Result<(), PublishError>)>,
29    dht_future: Pin<Box<dyn Future<Output = Result<(), PublishError>> + Send>>,
30    relays_future: Pin<Box<dyn Future<Output = Result<(), PublishError>> + Send>>,
31}
32
33impl SelectFuture {
34    /// Process a completed future result based on what we've seen before
35    fn process_result(
36        &mut self,
37        network: Network,
38        result: Result<(), PublishError>,
39    ) -> Poll<Result<(), PublishError>> {
40        match self.first_result {
41            // We already have a success, ignore CAS failures
42            Some((_, Ok(()))) => Poll::Ready(match result {
43                Err(PublishError::Concurrency(ConcurrencyError::CasFailed)) => Ok(()),
44                _ => result,
45            }),
46            // We already have a failure, return the later network's result
47            Some(_) => Poll::Ready(result),
48            // This is our first result, store it and continue polling
49            None => {
50                self.first_result = Some((network, result));
51
52                Poll::Pending
53            }
54        }
55    }
56}
57
58impl Future for SelectFuture {
59    type Output = Result<(), PublishError>;
60
61    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
62        let this = self.get_mut();
63
64        let done_network = this.first_result.as_ref().map(|r| r.0);
65
66        // Poll DHT future if not completed yet
67        if !matches!(done_network, Some(Network::Dht)) {
68            if let Poll::Ready(result) = this.dht_future.as_mut().poll(cx) {
69                return this.process_result(Network::Dht, result);
70            }
71        }
72
73        // Poll Relays future if not completed yet
74        if !matches!(done_network, Some(Network::Relays)) {
75            if let Poll::Ready(result) = this.relays_future.as_mut().poll(cx) {
76                return this.process_result(Network::Relays, result);
77            }
78        }
79
80        Poll::Pending
81    }
82}

86/// Returns a stream combinator that yields items from two inner streams in a round-robin fashion.
87///
88/// This combinator will:
89/// - Alternate between the two streams on each poll.
90/// - Continue polling a stream even after the other is exhausted.
91/// - Only terminate when **both** streams have returned `None`.
92pub fn select_stream(
93    dht_stream: Pin<Box<dyn Stream<Item = SignedPacket> + Send>>,
94    relays_stream: Pin<Box<dyn Stream<Item = SignedPacket> + Send>>,
95) -> SelectStream {
96    SelectStream {
97        mode: Mode::RoundRobin(Network::Dht),
98        dht_stream,
99        relays_stream,
100    }
101}
102
103pub struct SelectStream {
104    mode: Mode,
105    dht_stream: Pin<Box<dyn Stream<Item = SignedPacket> + Send>>,
106    relays_stream: Pin<Box<dyn Stream<Item = SignedPacket> + Send>>,
107}
108
109#[derive(Clone, Debug)]
110enum Mode {
111    RoundRobin(Network),
112    Exhausted(Network),
113}
114
115#[derive(Clone, Copy, Debug)]
116enum Network {
117    Dht,
118    Relays,
119}
120
121impl Stream for SelectStream {
122    type Item = SignedPacket;
123
124    fn poll_next(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
125        let this = self.get_mut();
126
127        match this.mode {
128            Mode::RoundRobin(current) => {
129                // Alternate which stream to poll next
130                let (primary, secondary) = match current {
131                    Network::Dht => (this.dht_stream.as_mut(), this.relays_stream.as_mut()),
132                    Network::Relays => (this.relays_stream.as_mut(), this.dht_stream.as_mut()),
133                };
134
135                // Update the Network for next poll
136                this.mode = Mode::RoundRobin(match current {
137                    Network::Dht => Network::Relays,
138                    Network::Relays => Network::Dht,
139                });
140
141                // Try polling the current primary stream
142                match primary.poll_next(cx) {
143                    Poll::Ready(Some(item)) => Poll::Ready(Some(item)),
144                    Poll::Ready(None) => {
145                        // Primary is exhausted, now only poll the remaining one
146                        this.mode = Mode::Exhausted(current);
147                        secondary.poll_next(cx)
148                    }
149                    Poll::Pending => Poll::Pending,
150                }
151            }
152            Mode::Exhausted(exhausted) => match exhausted {
153                Network::Dht => this.relays_stream.poll_next(cx),
154                Network::Relays => this.dht_stream.poll_next(cx),
155            },
156        }
157    }
158}
