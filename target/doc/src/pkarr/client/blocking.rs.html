35    // === Publish ===
36
37    /// Publishes a [SignedPacket] to the [mainline] Dht and or [Relays](https://pkarr.org/relays).
38    ///
39    /// # Lost Update Problem
40    ///
41    /// Mainline DHT and remote relays form a distributed network, and like all distributed networks,
42    /// it is vulnerable to [Writeâ€“write conflict](https://en.wikipedia.org/wiki/Write-write_conflict).
43    ///
44    /// ## Read first
45    ///
46    /// To mitigate the risk of lost updates, you should call the [Self::resolve_most_recent] method
47    /// then start authoring the new [SignedPacket] based on the most recent as in the following example:
48    ///
49    ///```rust
50    /// use pkarr::{Client, SignedPacket, Keypair};
51    /// // For local testing
52    /// use pkarr::mainline::Testnet;
53    ///
54    /// fn run() -> anyhow::Result<()> {
55    ///     let testnet = Testnet::new(3)?;
56    ///     let client = Client::builder()
57    ///         // Disable the default network settings (builtin relays and mainline bootstrap nodes).
58    ///         .no_default_network()
59    ///         .bootstrap(&testnet.bootstrap)
60    ///         .build()?
61    ///         .as_blocking();
62    ///
63    ///     let keypair = Keypair::random();
64    ///
65    ///     let (signed_packet, cas) = if let Some(most_recent) = client
66    ///         .resolve_most_recent(&keypair.public_key())
67    ///     {
68    ///
69    ///         let mut builder = SignedPacket::builder();
70    ///
71    ///         // 1. Optionally inherit all or some of the existing records.
72    ///         for record in most_recent.all_resource_records() {
73    ///             let name = record.name.to_string();
74    ///
75    ///             if name != "foo" && name != "sercert" {
76    ///                 builder = builder.record(record.clone());
77    ///             }
78    ///         };
79    ///
80    ///         // 2. Optionally add more new records.
81    ///         let signed_packet = builder
82    ///             .txt("foo".try_into()?, "bar".try_into()?, 30)
83    ///             .a("secret".try_into()?, 42.into(), 30)
84    ///             .sign(&keypair)?;
85    ///
86    ///         (
87    ///             signed_packet,
88    ///             // 3. Use the most recent [SignedPacket::timestamp] as a `CAS`.
89    ///             Some(most_recent.timestamp())
90    ///         )
91    ///     } else {
92    ///         (
93    ///             SignedPacket::builder()
94    ///                 .txt("foo".try_into()?, "bar".try_into()?, 30)
95    ///                 .a("secret".try_into()?, 42.into(), 30)
96    ///                 .sign(&keypair)?,
97    ///             None
98    ///         )
99    ///     };
100    ///
101    ///     client.publish(&signed_packet, cas)?;
102    ///
103    ///     Ok(())
104    /// }
105    /// ```
106    ///
107    /// ## Errors
108    ///
109    /// This method may return on of these errors:
110    ///
111    /// 1. [super::QueryError]: when the query fails, and you need to retry or debug the network.
112    /// 2. [super::ConcurrencyError]: when an write conflict (or the risk of it) is detedcted.
113    ///
114    /// If you get a [super::ConcurrencyError]; you should resolver the most recent packet again,
115    /// and repeat the steps in the previous example.
116    pub fn publish(
117        &self,
118        signed_packet: &SignedPacket,
119        cas: Option<Timestamp>,
120    ) -> Result<(), PublishError> {
121        futures_lite::future::block_on(self.0.publish(signed_packet, cas))
122    }
123
124    // === Resolve ===
125
126    /// Returns a [SignedPacket] from the cache even if it is expired.
127    /// If there is no packet in the cache, or if the cached packet is expired,
128    /// it will make a DHT query in a background query and caches any more recent packets it receives.
129    ///
130    /// If you want to get the most recent version of a [SignedPacket],
131    /// you should use [Self::resolve_most_recent].
132    pub fn resolve(&self, public_key: &PublicKey) -> Option<SignedPacket> {
133        futures_lite::future::block_on(self.0.resolve(public_key))
134    }
135
136    /// Returns the most recent [SignedPacket] found after querying all
137    /// [mainline] Dht nodes and or [Relays](https:://pkarr.org/relays).
138    ///
139    /// Useful if you want to read the most recent packet before publishing
140    /// a new packet.
141    ///
142    /// This is a best effort, and doesn't guarantee consistency.
143    pub fn resolve_most_recent(&self, public_key: &PublicKey) -> Option<SignedPacket> {
144        futures_lite::future::block_on(self.0.resolve_most_recent(public_key))
145    }
146}
