105    /// Queue data for a tail loss probe (or anti-amplification deadlock prevention) packet
106    ///
107    /// Probes are sent similarly to normal packets when an expected ACK has not arrived. We never
108    /// deem a packet lost until we receive an ACK that should have included it, but if a trailing
109    /// run of packets (or their ACKs) are lost, this might not happen in a timely fashion. We send
110    /// probe packets to force an ACK, and exempt them from congestion control to prevent a deadlock
111    /// when the congestion window is filled with lost tail packets.
112    ///
113    /// We prefer to send new data, to make the most efficient use of bandwidth. If there's no data
114    /// waiting to be sent, then we retransmit in-flight data to reduce odds of loss. If there's no
115    /// in-flight data either, we're probably a client guarding against a handshake
116    /// anti-amplification deadlock and we just make something up.
117    pub(super) fn maybe_queue_probe(
118        &mut self,
119        request_immediate_ack: bool,
120        streams: &StreamsState,
121    ) {
122        if self.loss_probes == 0 {
123            return;
124        }
125
126        if request_immediate_ack {
127            // The probe should be ACKed without delay (should only be used in the Data space and
128            // when the peer supports the acknowledgement frequency extension)
129            self.immediate_ack_pending = true;
130        }
131
132        // Retransmit the data of the oldest in-flight packet
133        if !self.pending.is_empty(streams) {
134            // There's real data to send here, no need to make something up
135            return;
136        }
137
138        for packet in self.sent_packets.values_mut() {
139            if !packet.retransmits.is_empty(streams) {
140                // Remove retransmitted data from the old packet so we don't end up retransmitting
141                // it *again* even if the copy we're sending now gets acknowledged.
142                self.pending |= mem::take(&mut packet.retransmits);
143                return;
144            }
145        }
146
147        // Nothing new to send and nothing to retransmit, so fall back on a ping. This should only
148        // happen in rare cases during the handshake when the server becomes blocked by
149        // anti-amplification.
150        self.ping_pending = true;
151    }
152

174    /// Verifies sanity of an ECN block and returns whether congestion was encountered.
175    pub(super) fn detect_ecn(
176        &mut self,
177        newly_acked: u64,
178        ecn: frame::EcnCounts,
179    ) -> Result<bool, &'static str> {
180        let ect0_increase = ecn
181            .ect0
182            .checked_sub(self.ecn_feedback.ect0)
183            .ok_or("peer ECT(0) count regression")?;
184        let ect1_increase = ecn
185            .ect1
186            .checked_sub(self.ecn_feedback.ect1)
187            .ok_or("peer ECT(1) count regression")?;
188        let ce_increase = ecn
189            .ce
190            .checked_sub(self.ecn_feedback.ce)
191            .ok_or("peer CE count regression")?;
192        let total_increase = ect0_increase + ect1_increase + ce_increase;
193        if total_increase < newly_acked {
194            return Err("ECN bleaching");
195        }
196        if (ect0_increase + ce_increase) < newly_acked || ect1_increase != 0 {
197            return Err("ECN corruption");
198        }
199        // If total_increase > newly_acked (which happens when ACKs are lost), this is required by
200        // the draft so that long-term drift does not occur. If =, then the only question is whether
201        // to count CE packets as CE or ECT0. Recording them as CE is more consistent and keeps the
202        // congestion check obvious.
203        self.ecn_feedback = ecn;
204        Ok(ce_increase != 0)
205    }
206
207    /// Stop tracking sent packet `number`, and return what we knew about it
208    pub(super) fn take(&mut self, number: u64) -> Option<SentPacket> {
209        let packet = self.sent_packets.remove(&number)?;
210        self.in_flight -= u64::from(packet.size);
211        if !packet.ack_eliciting && number > self.largest_ack_eliciting_sent {
212            self.unacked_non_ack_eliciting_tail =
213                self.unacked_non_ack_eliciting_tail.checked_sub(1).unwrap();
214        }
215        Some(packet)
216    }
217
218    /// Returns the number of bytes to *remove* from the connection's in-flight count
219    pub(super) fn sent(&mut self, number: u64, packet: SentPacket) -> u64 {
220        // Retain state for at most this many non-ACK-eliciting packets sent after the most recently
221        // sent ACK-eliciting packet. We're never guaranteed to receive an ACK for those, and we
222        // can't judge them as lost without an ACK, so to limit memory in applications which receive
223        // packets but don't send ACK-eliciting data for long periods use we must eventually start
224        // forgetting about them, although it might also be reasonable to just kill the connection
225        // due to weird peer behavior.
226        const MAX_UNACKED_NON_ACK_ELICTING_TAIL: u64 = 1_000;
227
228        let mut forgotten_bytes = 0;
229        if packet.ack_eliciting {
230            self.unacked_non_ack_eliciting_tail = 0;
231            self.largest_ack_eliciting_sent = number;
232        } else if self.unacked_non_ack_eliciting_tail > MAX_UNACKED_NON_ACK_ELICTING_TAIL {
233            let oldest_after_ack_eliciting = *self
234                .sent_packets
235                .range((
236                    Bound::Excluded(self.largest_ack_eliciting_sent),
237                    Bound::Unbounded,
238                ))
239                .next()
240                .unwrap()
241                .0;
242            // Per https://www.rfc-editor.org/rfc/rfc9000.html#name-frames-and-frame-types,
243            // non-ACK-eliciting packets must only contain PADDING, ACK, and CONNECTION_CLOSE
244            // frames, which require no special handling on ACK or loss beyond removal from
245            // in-flight counters if padded.
246            let packet = self
247                .sent_packets
248                .remove(&oldest_after_ack_eliciting)
249                .unwrap();
250            forgotten_bytes = u64::from(packet.size);
251            self.in_flight -= forgotten_bytes;
252        } else {
253            self.unacked_non_ack_eliciting_tail += 1;
254        }
255
256        self.in_flight += u64::from(packet.size);
257        self.sent_packets.insert(number, packet);
258        forgotten_bytes
259    }
260}

298/// Retransmittable data queue
299#[allow(unreachable_pub)] // fuzzing only
300#[derive(Debug, Default, Clone)]
301pub struct Retransmits {
302    pub(super) max_data: bool,
303    pub(super) max_stream_id: [bool; 2],
304    pub(super) reset_stream: Vec<(StreamId, VarInt)>,
305    pub(super) stop_sending: Vec<frame::StopSending>,
306    pub(super) max_stream_data: FxHashSet<StreamId>,
307    pub(super) crypto: VecDeque<frame::Crypto>,
308    pub(super) new_cids: Vec<IssuedCid>,
309    pub(super) retire_cids: Vec<u64>,
310    pub(super) ack_frequency: bool,
311    pub(super) handshake_done: bool,
312    pub(super) observed_addr: bool,
313    /// For each enqueued NEW_TOKEN frame, a copy of the path's remote address
314    ///
315    /// There are 2 reasons this is unusual:
316    ///
317    /// - If the path changes, NEW_TOKEN frames bound for the old path are not retransmitted on the
318    ///   new path. That is why this field stores the remote address: so that ones for old paths
319    ///   can be filtered out.
320    /// - If a token is lost, a new randomly generated token is re-transmitted, rather than the
321    ///   original. This is so that if both transmissions are received, the client won't risk
322    ///   sending the same token twice. That is why this field does _not_ store any actual token.
323    ///
324    /// It is true that a QUIC endpoint will only want to effectively have NEW_TOKEN frames
325    /// enqueued for its current path at a given point in time. Based on that, we could conceivably
326    /// change this from a vector to an `Option<(SocketAddr, usize)>` or just a `usize` or
327    /// something. However, due to the architecture of Quinn, it is considerably simpler to not do
328    /// that; consider what such a change would mean for implementing `BitOrAssign` on Self.
329    pub(super) new_tokens: Vec<SocketAddr>,
330}

344            && self.retire_cids.is_empty()

346            && !self.handshake_done

348            && self.new_tokens.is_empty()
349    }

352impl ::std::ops::BitOrAssign for Retransmits {
353    fn bitor_assign(&mut self, rhs: Self) {
354        // We reduce in-stream head-of-line blocking by queueing retransmits before other data for
355        // STREAM and CRYPTO frames.
356        self.max_data |= rhs.max_data;
357        for dir in Dir::iter() {
358            self.max_stream_id[dir as usize] |= rhs.max_stream_id[dir as usize];
359        }
360        self.reset_stream.extend_from_slice(&rhs.reset_stream);
361        self.stop_sending.extend_from_slice(&rhs.stop_sending);
362        self.max_stream_data.extend(&rhs.max_stream_data);
363        for crypto in rhs.crypto.into_iter().rev() {
364            self.crypto.push_front(crypto);
365        }
366        self.new_cids.extend(&rhs.new_cids);
367        self.retire_cids.extend(rhs.retire_cids);
368        self.ack_frequency |= rhs.ack_frequency;
369        self.handshake_done |= rhs.handshake_done;
370        self.observed_addr |= rhs.observed_addr;
371        self.new_tokens.extend_from_slice(&rhs.new_tokens);
372    }
373}

427/// RFC4303-style sliding window packet number deduplicator.
428///
429/// A contiguous bitfield, where each bit corresponds to a packet number and the rightmost bit is
430/// always set. A set bit represents a packet that has been successfully authenticated. Bits left of
431/// the window are assumed to be set.
432///
433/// ```text
434/// ...xxxxxxxxx 1 0
435///     ^        ^ ^
436/// window highest next
437/// ```
438pub(super) struct Dedup {
439    window: Window,
440    /// Lowest packet number higher than all yet authenticated.
441    next: u64,
442}
443
444/// Inner bitfield type.
445///
446/// Because QUIC never reuses packet numbers, this only needs to be large enough to deal with
447/// packets that are reordered but still delivered in a timely manner.
448type Window = u128;
449
450/// Number of packets tracked by `Dedup`.
451const WINDOW_SIZE: u64 = 1 + mem::size_of::<Window>() as u64 * 8;
452
453impl Dedup {
454    /// Construct an empty window positioned at the start.
455    pub(super) fn new() -> Self {
456        Self { window: 0, next: 0 }
457    }
458
459    /// Highest packet number authenticated.
460    fn highest(&self) -> u64 {
461        self.next - 1
462    }
463
464    /// Record a newly authenticated packet number.
465    ///
466    /// Returns whether the packet might be a duplicate.
467    pub(super) fn insert(&mut self, packet: u64) -> bool {
468        if let Some(diff) = packet.checked_sub(self.next) {
469            // Right of window
470            self.window = (self.window << 1 | 1)
471                .checked_shl(cmp::min(diff, u64::from(u32::MAX)) as u32)
472                .unwrap_or(0);
473            self.next = packet + 1;
474            false
475        } else if self.highest() - packet < WINDOW_SIZE {
476            // Within window
477            if let Some(bit) = (self.highest() - packet).checked_sub(1) {
478                // < highest
479                let mask = 1 << bit;
480                let duplicate = self.window & mask != 0;
481                self.window |= mask;
482                duplicate
483            } else {
484                // == highest
485                true
486            }
487        } else {
488            // Left of window
489            true
490        }
491    }
492
493    /// Returns the packet number of the smallest packet missing between the provided interval
494    ///
495    /// If there are no missing packets, returns `None`
496    fn smallest_missing_in_interval(&self, lower_bound: u64, upper_bound: u64) -> Option<u64> {
497        debug_assert!(lower_bound <= upper_bound);
498        debug_assert!(upper_bound <= self.highest());
499        const BITFIELD_SIZE: u64 = (mem::size_of::<Window>() * 8) as u64;
500
501        // Since we already know the packets at the boundaries have been received, we only need to
502        // check those in between them (this removes the necessity of extra logic to deal with the
503        // highest packet, which is stored outside the bitfield)
504        let lower_bound = lower_bound + 1;
505        let upper_bound = upper_bound.saturating_sub(1);
506
507        // Note: the offsets are counted from the right
508        // The highest packet is not included in the bitfield, so we subtract 1 to account for that
509        let start_offset = (self.highest() - upper_bound).max(1) - 1;
510        if start_offset >= BITFIELD_SIZE {
511            // The start offset is outside of the window. All packets outside of the window are
512            // considered to be received.
513            return None;
514        }
515
516        let end_offset_exclusive = self.highest().saturating_sub(lower_bound);
517
518        // The range is clamped at the edge of the window, because any earlier packets are
519        // considered to be received
520        let range_len = end_offset_exclusive
521            .saturating_sub(start_offset)
522            .min(BITFIELD_SIZE);
523        if range_len == 0 {
524            return None;
525        }
526
527        // Ensure the shift is within bounds (we already know start_offset < BITFIELD_SIZE,
528        // because of the early return)
529        let mask = if range_len == BITFIELD_SIZE {
530            u128::MAX
531        } else {
532            ((1u128 << range_len) - 1) << start_offset
533        };
534        let gaps = !self.window & mask;
535
536        let smallest_missing_offset = 128 - gaps.leading_zeros() as u64;
537        let smallest_missing_packet = self.highest() - smallest_missing_offset;
538
539        if smallest_missing_packet <= upper_bound {
540            Some(smallest_missing_packet)
541        } else {
542            None
543        }
544    }
545
546    /// Returns true if there are any missing packets between the provided interval
547    ///
548    /// The provided packet numbers must have been received before calling this function
549    fn missing_in_interval(&self, lower_bound: u64, upper_bound: u64) -> bool {
550        self.smallest_missing_in_interval(lower_bound, upper_bound)
551            .is_some()
552    }
553}

577#[derive(Debug)]
578pub(super) struct PendingAcks {
579    /// Whether we should send an ACK immediately, even if that means sending an ACK-only packet
580    ///
581    /// When `immediate_ack_required` is false, the normal behavior is to send ACK frames only when
582    /// there is other data to send, or when the `MaxAckDelay` timer expires.
583    immediate_ack_required: bool,
584    /// The number of ack-eliciting packets received since the last ACK frame was sent
585    ///
586    /// Once the count _exceeds_ `ack_eliciting_threshold`, an immediate ACK is required
587    ack_eliciting_since_last_ack_sent: u64,
588    non_ack_eliciting_since_last_ack_sent: u64,
589    ack_eliciting_threshold: u64,
590    /// The reordering threshold, controlling how we respond to out-of-order ack-eliciting packets
591    ///
592    /// Different values enable different behavior:
593    ///
594    /// * `0`: no special action is taken
595    /// * `1`: an ACK is immediately sent if it is out-of-order according to RFC 9000
596    /// * `>1`: an ACK is immediately sent if it is out-of-order according to the ACK frequency draft
597    reordering_threshold: u64,
598    /// The earliest ack-eliciting packet since the last ACK was sent, used to calculate the moment
599    /// upon which `max_ack_delay` elapses
600    earliest_ack_eliciting_since_last_ack_sent: Option<Instant>,
601    /// The packet number ranges of ack-eliciting packets the peer hasn't confirmed receipt of ACKs
602    /// for
603    ranges: ArrayRangeSet,
604    /// The packet with the largest packet number, and the time upon which it was received (used to
605    /// calculate ACK delay in [`PendingAcks::ack_delay`])
606    largest_packet: Option<(u64, Instant)>,
607    /// The ack-eliciting packet we have received with the largest packet number
608    largest_ack_eliciting_packet: Option<u64>,
609    /// The largest acknowledged packet number sent in an ACK frame
610    largest_acked: Option<u64>,
611}
612
613impl PendingAcks {
614    fn new() -> Self {
615        Self {
616            immediate_ack_required: false,
617            ack_eliciting_since_last_ack_sent: 0,
618            non_ack_eliciting_since_last_ack_sent: 0,
619            ack_eliciting_threshold: 1,
620            reordering_threshold: 1,
621            earliest_ack_eliciting_since_last_ack_sent: None,
622            ranges: ArrayRangeSet::default(),
623            largest_packet: None,
624            largest_ack_eliciting_packet: None,
625            largest_acked: None,
626        }
627    }
628
629    pub(super) fn set_ack_frequency_params(&mut self, frame: &frame::AckFrequency) {
630        self.ack_eliciting_threshold = frame.ack_eliciting_threshold.into_inner();
631        self.reordering_threshold = frame.reordering_threshold.into_inner();
632    }
633
634    pub(super) fn set_immediate_ack_required(&mut self) {
635        self.immediate_ack_required = true;
636    }
637
638    pub(super) fn on_max_ack_delay_timeout(&mut self) {
639        self.immediate_ack_required = self.ack_eliciting_since_last_ack_sent > 0;
640    }
641
642    pub(super) fn max_ack_delay_timeout(&self, max_ack_delay: Duration) -> Option<Instant> {
643        self.earliest_ack_eliciting_since_last_ack_sent
644            .map(|earliest_unacked| earliest_unacked + max_ack_delay)
645    }
646
647    /// Whether any ACK frames can be sent
648    pub(super) fn can_send(&self) -> bool {
649        self.immediate_ack_required && !self.ranges.is_empty()
650    }
651
652    /// Returns the delay since the packet with the largest packet number was received
653    pub(super) fn ack_delay(&self, now: Instant) -> Duration {
654        self.largest_packet
655            .map_or(Duration::default(), |(_, received)| now - received)
656    }
657
658    /// Handle receipt of a new packet
659    ///
660    /// Returns true if the max ack delay timer should be armed
661    pub(super) fn packet_received(
662        &mut self,
663        now: Instant,
664        packet_number: u64,
665        ack_eliciting: bool,
666        dedup: &Dedup,
667    ) -> bool {
668        if !ack_eliciting {
669            self.non_ack_eliciting_since_last_ack_sent += 1;
670            return false;
671        }
672
673        let prev_largest_ack_eliciting = self.largest_ack_eliciting_packet.unwrap_or(0);
674
675        // Track largest ack-eliciting packet
676        self.largest_ack_eliciting_packet = self
677            .largest_ack_eliciting_packet
678            .map(|pn| pn.max(packet_number))
679            .or(Some(packet_number));
680
681        // Handle ack_eliciting_threshold
682        self.ack_eliciting_since_last_ack_sent += 1;
683        self.immediate_ack_required |=
684            self.ack_eliciting_since_last_ack_sent > self.ack_eliciting_threshold;
685
686        // Handle out-of-order packets
687        self.immediate_ack_required |=
688            self.is_out_of_order(packet_number, prev_largest_ack_eliciting, dedup);
689
690        // Arm max_ack_delay timer if necessary
691        if self.earliest_ack_eliciting_since_last_ack_sent.is_none() && !self.can_send() {
692            self.earliest_ack_eliciting_since_last_ack_sent = Some(now);
693            return true;
694        }
695
696        false
697    }
698
699    fn is_out_of_order(
700        &self,
701        packet_number: u64,
702        prev_largest_ack_eliciting: u64,
703        dedup: &Dedup,
704    ) -> bool {
705        match self.reordering_threshold {
706            0 => false,
707            1 => {
708                // From https://www.rfc-editor.org/rfc/rfc9000#section-13.2.1-7
709                packet_number < prev_largest_ack_eliciting
710                    || dedup.missing_in_interval(prev_largest_ack_eliciting, packet_number)
711            }
712            _ => {
713                // From acknowledgement frequency draft, section 6.1: send an ACK immediately if
714                // doing so would cause the sender to detect a new packet loss
715                let Some((largest_acked, largest_unacked)) =
716                    self.largest_acked.zip(self.largest_ack_eliciting_packet)
717                else {
718                    return false;
719                };
720                if self.reordering_threshold > largest_acked {
721                    return false;
722                }
723                // The largest packet number that could be declared lost without a new ACK being
724                // sent
725                let largest_reported = largest_acked - self.reordering_threshold + 1;
726                let Some(smallest_missing_unreported) =
727                    dedup.smallest_missing_in_interval(largest_reported, largest_unacked)
728                else {
729                    return false;
730                };
731                largest_unacked - smallest_missing_unreported >= self.reordering_threshold
732            }
733        }
734    }
735
736    /// Should be called whenever ACKs have been sent
737    ///
738    /// This will suppress sending further ACKs until additional ACK eliciting frames arrive
739    pub(super) fn acks_sent(&mut self) {
740        // It is possible (though unlikely) that the ACKs we just sent do not cover all the
741        // ACK-eliciting packets we have received (e.g. if there is not enough room in the packet to
742        // fit all the ranges). To keep things simple, however, we assume they do. If there are
743        // indeed some ACKs that weren't covered, the packets might be ACKed later anyway, because
744        // they are still contained in `self.ranges`. If we somehow fail to send the ACKs at a later
745        // moment, the peer will assume the packets got lost and will retransmit their frames in a
746        // new packet, which is suboptimal, because we already received them. Our assumption here is
747        // that simplicity results in code that is more performant, even in the presence of
748        // occasional redundant retransmits.
749        self.immediate_ack_required = false;
750        self.ack_eliciting_since_last_ack_sent = 0;
751        self.non_ack_eliciting_since_last_ack_sent = 0;
752        self.earliest_ack_eliciting_since_last_ack_sent = None;
753        self.largest_acked = self.largest_ack_eliciting_packet;
754    }
755
756    /// Insert one packet that needs to be acknowledged
757    pub(super) fn insert_one(&mut self, packet: u64, now: Instant) {
758        self.ranges.insert_one(packet);
759
760        if self.largest_packet.map_or(true, |(pn, _)| packet > pn) {
761            self.largest_packet = Some((packet, now));
762        }
763
764        if self.ranges.len() > MAX_ACK_BLOCKS {
765            self.ranges.pop_min();
766        }
767    }
768
769    /// Remove ACKs of packets numbered at or below `max` from the set of pending ACKs
770    pub(super) fn subtract_below(&mut self, max: u64) {
771        self.ranges.remove(0..(max + 1));
772    }
773
774    /// Returns the set of currently pending ACK ranges
775    pub(super) fn ranges(&self) -> &ArrayRangeSet {
776        &self.ranges
777    }
778
779    /// Queue an ACK if a significant number of non-ACK-eliciting packets have not yet been
780    /// acknowledged
781    ///
782    /// Should be called immediately before a non-probing packet is composed, when we've already
783    /// committed to sending a packet regardless.
784    pub(super) fn maybe_ack_non_eliciting(&mut self) {
785        // If we're going to send a packet anyway, and we've received a significant number of
786        // non-ACK-eliciting packets, then include an ACK to help the peer perform timely loss
787        // detection even if they're not sending any ACK-eliciting packets themselves. Exact
788        // threshold chosen somewhat arbitrarily.
789        const LAZY_ACK_THRESHOLD: u64 = 10;
790        if self.non_ack_eliciting_since_last_ack_sent > LAZY_ACK_THRESHOLD {
791            self.immediate_ack_required = true;
792        }
793    }
794}

796/// Helper for mitigating [optimistic ACK attacks]
797///
798/// A malicious peer could prompt the local application to begin a large data transfer, and then
799/// send ACKs without first waiting for data to be received. This could defeat congestion control,
800/// allowing the connection to consume disproportionate resources. We therefore occasionally skip
801/// packet numbers, and classify any ACK referencing a skipped packet number as a transport error.
802///
803/// Skipped packet numbers occur only in the application data space (where costly transfers might
804/// take place) and are distributed exponentially to reflect the reduced likelihood and impact of
805/// bad behavior from a peer that has been well-behaved for an extended period.
806///
807/// ACKs for packet numbers that have not yet been allocated are also a transport error, but an
808/// attacker with knowledge of the congestion control algorithm in use could time falsified ACKs to
809/// arrive after the packets they reference are sent.
810///
811/// [optimistic ACK attacks]: https://www.rfc-editor.org/rfc/rfc9000.html#name-optimistic-ack-attack
812pub(super) struct PacketNumberFilter {
813    /// Next outgoing packet number to skip
814    next_skipped_packet_number: u64,
815    /// Most recently skipped packet number
816    prev_skipped_packet_number: Option<u64>,
817    /// Next packet number to skip is randomly selected from 2^n..2^n+1
818    exponent: u32,
819}
820
821impl PacketNumberFilter {
822    pub(super) fn new(rng: &mut (impl Rng + ?Sized)) -> Self {
823        // First skipped PN is in 0..64
824        let exponent = 6;
825        Self {
826            next_skipped_packet_number: rng.gen_range(0..2u64.saturating_pow(exponent)),
827            prev_skipped_packet_number: None,
828            exponent,
829        }
830    }
831
832    #[cfg(test)]
833    pub(super) fn disabled() -> Self {
834        Self {
835            next_skipped_packet_number: u64::MAX,
836            prev_skipped_packet_number: None,
837            exponent: u32::MAX,
838        }
839    }
840
841    pub(super) fn peek(&self, space: &PacketSpace) -> u64 {
842        let n = space.next_packet_number;
843        if n != self.next_skipped_packet_number {
844            return n;
845        }
846        n + 1
847    }
848
849    pub(super) fn allocate(
850        &mut self,
851        rng: &mut (impl Rng + ?Sized),
852        space: &mut PacketSpace,
853    ) -> u64 {
854        let n = space.get_tx_number();
855        if n != self.next_skipped_packet_number {
856            return n;
857        }
858
859        trace!("skipping pn {n}");
860        // Skip this packet number, and choose the next one to skip
861        self.prev_skipped_packet_number = Some(self.next_skipped_packet_number);
862        let next_exponent = self.exponent.saturating_add(1);
863        self.next_skipped_packet_number =
864            rng.gen_range(2u64.saturating_pow(self.exponent)..2u64.saturating_pow(next_exponent));
865        self.exponent = next_exponent;
866
867        space.get_tx_number()
868    }
869
870    pub(super) fn check_ack(
871        &self,
872        space_id: SpaceId,
873        range: std::ops::RangeInclusive<u64>,
874    ) -> Result<(), TransportError> {
875        if space_id == SpaceId::Data
876            && self
877                .prev_skipped_packet_number
878                .is_some_and(|x| range.contains(&x))
879        {
880            return Err(TransportError::PROTOCOL_VIOLATION("unsent packet acked"));
881        }
882        Ok(())
883    }
884}
