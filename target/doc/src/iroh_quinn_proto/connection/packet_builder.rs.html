31    /// Write a new packet header to `buffer` and determine the packet's properties
32    ///
33    /// Marks the connection drained and returns `None` if the confidentiality limit would be
34    /// violated.
35    pub(super) fn new(
36        now: Instant,
37        space_id: SpaceId,
38        dst_cid: ConnectionId,
39        buffer: &mut Vec<u8>,
40        buffer_capacity: usize,
41        datagram_start: usize,
42        ack_eliciting: bool,
43        conn: &mut Connection,
44    ) -> Option<Self> {
45        let version = conn.version;
46        // Initiate key update if we're approaching the confidentiality limit
47        let sent_with_keys = conn.spaces[space_id].sent_with_keys;
48        if space_id == SpaceId::Data {
49            if sent_with_keys >= conn.key_phase_size {
50                conn.force_key_update();
51            }
52        } else {
53            let confidentiality_limit = conn.spaces[space_id]
54                .crypto
55                .as_ref()
56                .map_or_else(
57                    || &conn.zero_rtt_crypto.as_ref().unwrap().packet,
58                    |keys| &keys.packet.local,
59                )
60                .confidentiality_limit();
61            if sent_with_keys.saturating_add(1) == confidentiality_limit {
62                // We still have time to attempt a graceful close
63                conn.close_inner(
64                    now,
65                    Close::Connection(frame::ConnectionClose {
66                        error_code: TransportErrorCode::AEAD_LIMIT_REACHED,
67                        frame_type: None,
68                        reason: Bytes::from_static(b"confidentiality limit reached"),
69                    }),
70                )
71            } else if sent_with_keys > confidentiality_limit {
72                // Confidentiality limited violated and there's nothing we can do
73                conn.kill(
74                    TransportError::AEAD_LIMIT_REACHED("confidentiality limit reached").into(),
75                );
76                return None;
77            }
78        }
79
80        let space = &mut conn.spaces[space_id];
81        let exact_number = match space_id {
82            SpaceId::Data => conn.packet_number_filter.allocate(&mut conn.rng, space),
83            _ => space.get_tx_number(),
84        };
85
86        let span = trace_span!("send", space = ?space_id, pn = exact_number).entered();
87
88        let number = PacketNumber::new(exact_number, space.largest_acked_packet.unwrap_or(0));
89        let header = match space_id {
90            SpaceId::Data if space.crypto.is_some() => Header::Short {
91                dst_cid,
92                number,
93                spin: if conn.spin_enabled {
94                    conn.spin
95                } else {
96                    conn.rng.gen()
97                },
98                key_phase: conn.key_phase,
99            },
100            SpaceId::Data => Header::Long {
101                ty: LongType::ZeroRtt,
102                src_cid: conn.handshake_cid,
103                dst_cid,
104                number,
105                version,
106            },
107            SpaceId::Handshake => Header::Long {
108                ty: LongType::Handshake,
109                src_cid: conn.handshake_cid,
110                dst_cid,
111                number,
112                version,
113            },
114            SpaceId::Initial => Header::Initial(InitialHeader {
115                src_cid: conn.handshake_cid,
116                dst_cid,
117                token: match &conn.side {
118                    ConnectionSide::Client { token, .. } => token.clone(),
119                    ConnectionSide::Server { .. } => Bytes::new(),
120                },
121                number,
122                version,
123            }),
124        };
125        let partial_encode = header.encode(buffer);
126        if conn.peer_params.grease_quic_bit && conn.rng.gen() {
127            buffer[partial_encode.start] ^= FIXED_BIT;
128        }
129
130        let (sample_size, tag_len) = if let Some(ref crypto) = space.crypto {
131            (
132                crypto.header.local.sample_size(),
133                crypto.packet.local.tag_len(),
134            )
135        } else if space_id == SpaceId::Data {
136            let zero_rtt = conn.zero_rtt_crypto.as_ref().unwrap();
137            (zero_rtt.header.sample_size(), zero_rtt.packet.tag_len())
138        } else {
139            unreachable!();
140        };
141
142        // Each packet must be large enough for header protection sampling, i.e. the combined
143        // lengths of the encoded packet number and protected payload must be at least 4 bytes
144        // longer than the sample required for header protection. Further, each packet should be at
145        // least tag_len + 6 bytes larger than the destination CID on incoming packets so that the
146        // peer may send stateless resets that are indistinguishable from regular traffic.
147
148        // pn_len + payload_len + tag_len >= sample_size + 4
149        // payload_len >= sample_size + 4 - pn_len - tag_len
150        let min_size = Ord::max(
151            buffer.len() + (sample_size + 4).saturating_sub(number.len() + tag_len),
152            partial_encode.start + dst_cid.len() + 6,
153        );
154        let max_size = buffer_capacity - tag_len;
155        debug_assert!(max_size >= min_size);
156
157        Some(Self {
158            datagram_start,
159            space: space_id,
160            partial_encode,
161            exact_number,
162            short_header: header.is_short(),
163            min_size,
164            max_size,
165            tag_len,
166            ack_eliciting,
167            _span: span,
168        })
169    }

171    /// Append the minimum amount of padding to the packet such that, after encryption, the
172    /// enclosing datagram will occupy at least `min_size` bytes
173    pub(super) fn pad_to(&mut self, min_size: u16) {
174        // The datagram might already have a larger minimum size than the caller is requesting, if
175        // e.g. we're coalescing packets and have populated more than `min_size` bytes with packets
176        // already.
177        self.min_size = Ord::max(
178            self.min_size,
179            self.datagram_start + (min_size as usize) - self.tag_len,
180        );
181    }

183    pub(super) fn finish_and_track(
184        self,
185        now: Instant,
186        conn: &mut Connection,
187        sent: Option<SentFrames>,
188        buffer: &mut Vec<u8>,
189    ) {
190        let ack_eliciting = self.ack_eliciting;
191        let exact_number = self.exact_number;
192        let space_id = self.space;
193        let (size, padded) = self.finish(conn, buffer);
194        let sent = match sent {
195            Some(sent) => sent,
196            None => return,
197        };
198
199        let size = match padded || ack_eliciting {
200            true => size as u16,
201            false => 0,
202        };
203
204        let packet = SentPacket {
205            largest_acked: sent.largest_acked,
206            time_sent: now,
207            size,
208            ack_eliciting,
209            retransmits: sent.retransmits,
210            stream_frames: sent.stream_frames,
211        };
212
213        conn.path
214            .sent(exact_number, packet, &mut conn.spaces[space_id]);
215        conn.stats.path.sent_packets += 1;
216        conn.reset_keep_alive(now);
217        if size != 0 {
218            if ack_eliciting {
219                conn.spaces[space_id].time_of_last_ack_eliciting_packet = Some(now);
220                if conn.permit_idle_reset {
221                    conn.reset_idle_timeout(now, space_id);
222                }
223                conn.permit_idle_reset = false;
224            }
225            conn.set_loss_detection_timer(now);
226            conn.path.pacing.on_transmit(size);
227        }
228    }

230    /// Encrypt packet, returning the length of the packet and whether padding was added
231    pub(super) fn finish(self, conn: &mut Connection, buffer: &mut Vec<u8>) -> (usize, bool) {
232        let pad = buffer.len() < self.min_size;
233        if pad {
234            trace!("PADDING * {}", self.min_size - buffer.len());
235            buffer.resize(self.min_size, 0);
236        }
237
238        let space = &conn.spaces[self.space];
239        let (header_crypto, packet_crypto) = if let Some(ref crypto) = space.crypto {
240            (&*crypto.header.local, &*crypto.packet.local)
241        } else if self.space == SpaceId::Data {
242            let zero_rtt = conn.zero_rtt_crypto.as_ref().unwrap();
243            (&*zero_rtt.header, &*zero_rtt.packet)
244        } else {
245            unreachable!("tried to send {:?} packet without keys", self.space);
246        };
247
248        debug_assert_eq!(
249            packet_crypto.tag_len(),
250            self.tag_len,
251            "Mismatching crypto tag len"
252        );
253
254        buffer.resize(buffer.len() + packet_crypto.tag_len(), 0);
255        let encode_start = self.partial_encode.start;
256        let packet_buf = &mut buffer[encode_start..];
257        self.partial_encode.finish(
258            packet_buf,
259            header_crypto,
260            Some((self.exact_number, packet_crypto)),
261        );
262
263        (buffer.len() - encode_start, pad)
264    }
