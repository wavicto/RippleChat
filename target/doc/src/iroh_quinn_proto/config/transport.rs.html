8/// Parameters governing the core QUIC state machine
9///
10/// Default values should be suitable for most internet applications. Applications protocols which
11/// forbid remotely-initiated streams should set `max_concurrent_bidi_streams` and
12/// `max_concurrent_uni_streams` to zero.
13///
14/// In some cases, performance or resource requirements can be improved by tuning these values to
15/// suit a particular application and/or network connection. In particular, data window sizes can be
16/// tuned for a particular expected round trip time, link capacity, and memory availability. Tuning
17/// for higher bandwidths and latencies increases worst-case memory consumption, but does not impair
18/// performance at lower bandwidths and latencies. The default configuration is tuned for a 100Mbps
19/// link with a 100ms round trip time.
20pub struct TransportConfig {
21    pub(crate) max_concurrent_bidi_streams: VarInt,
22    pub(crate) max_concurrent_uni_streams: VarInt,

25    pub(crate) receive_window: VarInt,
26    pub(crate) send_window: u64,
27    pub(crate) send_fairness: bool,
28
29    pub(crate) packet_threshold: u32,
30    pub(crate) time_threshold: f32,
31    pub(crate) initial_rtt: Duration,
32    pub(crate) initial_mtu: u16,
33    pub(crate) min_mtu: u16,
34    pub(crate) mtu_discovery_config: Option<MtuDiscoveryConfig>,
35    pub(crate) ack_frequency_config: Option<AckFrequencyConfig>,

72    ///
73    /// The true idle timeout is the minimum of this and the peer's own max idle timeout. `None`

75    ///

449/// Parameters for controlling the peer's acknowledgement frequency
450///
451/// The parameters provided in this config will be sent to the peer at the beginning of the
452/// connection, so it can take them into account when sending acknowledgements (see each parameter's
453/// description for details on how it influences acknowledgement frequency).
454///
455/// Quinn's implementation follows the fourth draft of the
456/// [QUIC Acknowledgement Frequency extension](https://datatracker.ietf.org/doc/html/draft-ietf-quic-ack-frequency-04).
457/// The defaults produce behavior slightly different than the behavior without this extension,
458/// because they change the way reordered packets are handled (see
459/// [`AckFrequencyConfig::reordering_threshold`] for details).
460#[derive(Clone, Debug)]
461pub struct AckFrequencyConfig {
462    pub(crate) ack_eliciting_threshold: VarInt,
463    pub(crate) max_ack_delay: Option<Duration>,
464    pub(crate) reordering_threshold: VarInt,
465}
466
467impl AckFrequencyConfig {
468    /// The ack-eliciting threshold we will request the peer to use
469    ///
470    /// This threshold represents the number of ack-eliciting packets an endpoint may receive
471    /// without immediately sending an ACK.
472    ///
473    /// The remote peer should send at least one ACK frame when more than this number of
474    /// ack-eliciting packets have been received. A value of 0 results in a receiver immediately
475    /// acknowledging every ack-eliciting packet.
476    ///
477    /// Defaults to 1, which sends ACK frames for every other ack-eliciting packet.
478    pub fn ack_eliciting_threshold(&mut self, value: VarInt) -> &mut Self {
479        self.ack_eliciting_threshold = value;
480        self
481    }
482
483    /// The `max_ack_delay` we will request the peer to use
484    ///
485    /// This parameter represents the maximum amount of time that an endpoint waits before sending
486    /// an ACK when the ack-eliciting threshold hasn't been reached.
487    ///
488    /// The effective `max_ack_delay` will be clamped to be at least the peer's `min_ack_delay`
489    /// transport parameter, and at most the greater of the current path RTT or 25ms.
490    ///
491    /// Defaults to `None`, in which case the peer's original `max_ack_delay` will be used, as
492    /// obtained from its transport parameters.
493    pub fn max_ack_delay(&mut self, value: Option<Duration>) -> &mut Self {
494        self.max_ack_delay = value;
495        self
496    }
497
498    /// The reordering threshold we will request the peer to use
499    ///
500    /// This threshold represents the amount of out-of-order packets that will trigger an endpoint
501    /// to send an ACK, without waiting for `ack_eliciting_threshold` to be exceeded or for
502    /// `max_ack_delay` to be elapsed.
503    ///
504    /// A value of 0 indicates out-of-order packets do not elicit an immediate ACK. A value of 1
505    /// immediately acknowledges any packets that are received out of order (this is also the
506    /// behavior when the extension is disabled).
507    ///
508    /// It is recommended to set this value to [`TransportConfig::packet_threshold`] minus one.
509    /// Since the default value for [`TransportConfig::packet_threshold`] is 3, this value defaults
510    /// to 2.
511    pub fn reordering_threshold(&mut self, value: VarInt) -> &mut Self {
512        self.reordering_threshold = value;
513        self
514    }
515}
516
517impl Default for AckFrequencyConfig {
518    fn default() -> Self {
519        Self {
520            ack_eliciting_threshold: VarInt(1),
521            max_ack_delay: None,
522            reordering_threshold: VarInt(2),
523        }
524    }

527/// Parameters governing MTU discovery.
528///
529/// # The why of MTU discovery
530///
531/// By design, QUIC ensures during the handshake that the network path between the client and the
532/// server is able to transmit unfragmented UDP packets with a body of 1200 bytes. In other words,
533/// once the connection is established, we know that the network path's maximum transmission unit
534/// (MTU) is of at least 1200 bytes (plus IP and UDP headers). Because of this, a QUIC endpoint can
535/// split outgoing data in packets of 1200 bytes, with confidence that the network will be able to
536/// deliver them (if the endpoint were to send bigger packets, they could prove too big and end up
537/// being dropped).
538///
539/// There is, however, a significant overhead associated to sending a packet. If the same
540/// information can be sent in fewer packets, that results in higher throughput. The amount of
541/// packets that need to be sent is inversely proportional to the MTU: the higher the MTU, the
542/// bigger the packets that can be sent, and the fewer packets that are needed to transmit a given
543/// amount of bytes.
544///
545/// Most networks have an MTU higher than 1200. Through MTU discovery, endpoints can detect the
546/// path's MTU and, if it turns out to be higher, start sending bigger packets.
547///
548/// # MTU discovery internals
549///
550/// Quinn implements MTU discovery through DPLPMTUD (Datagram Packetization Layer Path MTU
551/// Discovery), described in [section 14.3 of RFC
552/// 9000](https://www.rfc-editor.org/rfc/rfc9000.html#section-14.3). This method consists of sending
553/// QUIC packets padded to a particular size (called PMTU probes), and waiting to see if the remote
554/// peer responds with an ACK. If an ACK is received, that means the probe arrived at the remote
555/// peer, which in turn means that the network path's MTU is of at least the packet's size. If the
556/// probe is lost, it is sent another 2 times before concluding that the MTU is lower than the
557/// packet's size.
558///
559/// MTU discovery runs on a schedule (e.g. every 600 seconds) specified through
560/// [`MtuDiscoveryConfig::interval`]. The first run happens right after the handshake, and
561/// subsequent discoveries are scheduled to run when the interval has elapsed, starting from the
562/// last time when MTU discovery completed.
563///
564/// Since the search space for MTUs is quite big (the smallest possible MTU is 1200, and the highest
565/// is 65527), Quinn performs a binary search to keep the number of probes as low as possible. The
566/// lower bound of the search is equal to [`TransportConfig::initial_mtu`] in the
567/// initial MTU discovery run, and is equal to the currently discovered MTU in subsequent runs. The
568/// upper bound is determined by the minimum of [`MtuDiscoveryConfig::upper_bound`] and the
569/// `max_udp_payload_size` transport parameter received from the peer during the handshake.
570///
571/// # Black hole detection
572///
573/// If, at some point, the network path no longer accepts packets of the detected size, packet loss
574/// will eventually trigger black hole detection and reset the detected MTU to 1200. In that case,
575/// MTU discovery will be triggered after [`MtuDiscoveryConfig::black_hole_cooldown`] (ignoring the
576/// timer that was set based on [`MtuDiscoveryConfig::interval`]).
577///
578/// # Interaction between peers
579///
580/// There is no guarantee that the MTU on the path between A and B is the same as the MTU of the
581/// path between B and A. Therefore, each peer in the connection needs to run MTU discovery
582/// independently in order to discover the path's MTU.
583#[derive(Clone, Debug)]
584pub struct MtuDiscoveryConfig {
585    pub(crate) interval: Duration,
586    pub(crate) upper_bound: u16,
587    pub(crate) minimum_change: u16,
588    pub(crate) black_hole_cooldown: Duration,
589}
590
591impl MtuDiscoveryConfig {
592    /// Specifies the time to wait after completing MTU discovery before starting a new MTU
593    /// discovery run.
594    ///
595    /// Defaults to 600 seconds, as recommended by [RFC
596    /// 8899](https://www.rfc-editor.org/rfc/rfc8899).
597    pub fn interval(&mut self, value: Duration) -> &mut Self {
598        self.interval = value;
599        self
600    }
601
602    /// Specifies the upper bound to the max UDP payload size that MTU discovery will search for.
603    ///
604    /// Defaults to 1452, to stay within Ethernet's MTU when using IPv4 and IPv6. The highest
605    /// allowed value is 65527, which corresponds to the maximum permitted UDP payload on IPv6.
606    ///
607    /// It is safe to use an arbitrarily high upper bound, regardless of the network path's MTU. The
608    /// only drawback is that MTU discovery might take more time to finish.
609    pub fn upper_bound(&mut self, value: u16) -> &mut Self {
610        self.upper_bound = value.min(MAX_UDP_PAYLOAD);
611        self
612    }
613
614    /// Specifies the amount of time that MTU discovery should wait after a black hole was detected
615    /// before running again. Defaults to one minute.
616    ///
617    /// Black hole detection can be spuriously triggered in case of congestion, so it makes sense to
618    /// try MTU discovery again after a short period of time.
619    pub fn black_hole_cooldown(&mut self, value: Duration) -> &mut Self {
620        self.black_hole_cooldown = value;
621        self
622    }
623
624    /// Specifies the minimum MTU change to stop the MTU discovery phase.
625    /// Defaults to 20.
626    pub fn minimum_change(&mut self, value: u16) -> &mut Self {
627        self.minimum_change = value;
628        self
629    }
630}
631
632impl Default for MtuDiscoveryConfig {
633    fn default() -> Self {
634        Self {
635            interval: Duration::from_secs(600),
636            upper_bound: 1452,
637            black_hole_cooldown: Duration::from_secs(60),
638            minimum_change: 20,
639        }
640    }
641}

643/// Maximum duration of inactivity to accept before timing out the connection
644///
645/// This wraps an underlying [`VarInt`], representing the duration in milliseconds. Values can be
646/// constructed by converting directly from `VarInt`, or using `TryFrom<Duration>`.
647///
648/// ```
649/// # use std::{convert::TryFrom, time::Duration};
650/// # use iroh_quinn_proto::{IdleTimeout, VarIntBoundsExceeded, VarInt};
651/// # fn main() -> Result<(), VarIntBoundsExceeded> {
652/// // A `VarInt`-encoded value in milliseconds
653/// let timeout = IdleTimeout::from(VarInt::from_u32(10_000));
654///
655/// // Try to convert a `Duration` into a `VarInt`-encoded timeout
656/// let timeout = IdleTimeout::try_from(Duration::from_secs(10))?;
657/// # Ok(())
658/// # }
659/// ```
660#[derive(Default, Copy, Clone, Eq, Hash, Ord, PartialEq, PartialOrd)]
661pub struct IdleTimeout(VarInt);
662
663impl From<VarInt> for IdleTimeout {
664    fn from(inner: VarInt) -> Self {
665        Self(inner)
666    }
667}
668
669impl std::convert::TryFrom<Duration> for IdleTimeout {
670    type Error = VarIntBoundsExceeded;
671
672    fn try_from(timeout: Duration) -> Result<Self, Self::Error> {
673        let inner = VarInt::try_from(timeout.as_millis())?;
674        Ok(Self(inner))
675    }
676}
677
678impl fmt::Debug for IdleTimeout {
679    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
680        self.0.fmt(f)
681    }
