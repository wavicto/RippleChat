1use alloc::sync::Arc;
2use core::array;
3use core::task::Waker;
4use std::sync::{Mutex, MutexGuard};
5
6use super::{InlineWakerArray, ReadinessArray};
7
8/// A collection of wakers which delegate to an in-line waker.
9pub(crate) struct WakerArray<const N: usize> {
10    wakers: [Waker; N],
11    readiness: Arc<Mutex<ReadinessArray<N>>>,
12}
13
14impl<const N: usize> WakerArray<N> {
15    /// Create a new instance of `WakerArray`.
16    pub(crate) fn new() -> Self {
17        let readiness = Arc::new(Mutex::new(ReadinessArray::new()));
18        Self {
19            wakers: array::from_fn(|i| {
20                Arc::new(InlineWakerArray::new(i, readiness.clone())).into()
21            }),
22            readiness,
23        }
24    }
25
26    pub(crate) fn get(&self, index: usize) -> Option<&Waker> {
27        self.wakers.get(index)
28    }
29
30    /// Access the `Readiness`.
31    pub(crate) fn readiness(&mut self) -> MutexGuard<'_, ReadinessArray<N>> {
32        self.readiness.as_ref().lock().unwrap()
33    }
34}
