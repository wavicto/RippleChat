12const NDTPA_REFCNT: u16 = 2;
13const NDTPA_REACHABLE_TIME: u16 = 3;
14const NDTPA_BASE_REACHABLE_TIME: u16 = 4;

16const NDTPA_GC_STALETIME: u16 = 6;
17const NDTPA_DELAY_PROBE_TIME: u16 = 7;

31#[derive(Debug, PartialEq, Eq, Clone)]
32#[non_exhaustive]
33pub enum NeighbourTableParameter {
34    Ifindex(u32),
35    ReferenceCount(u32),
36    ReachableTime(u64),
37    BaseReachableTime(u64),
38    RetransTime(u64),
39    GcStaletime(u64),
40    DelayProbeTime(u64),
41    QueueLen(u32),
42    AppProbes(u32),
43    UcastProbes(u32),
44    McastProbes(u32),
45    AnycastDelay(u64),
46    ProxyDelay(u64),
47    ProxyQlen(u32),
48    Locktime(u64),
49    QueueLenbytes(u32),
50    McastReprobes(u32),
51    IntervalProbeTimeMs(u64),
52    Other(DefaultNla),
53}
54
55impl Nla for NeighbourTableParameter {
56    fn value_len(&self) -> usize {
57        match self {
58            Self::Ifindex(_)
59            | Self::ReferenceCount(_)
60            | Self::QueueLen(_)
61            | Self::AppProbes(_)
62            | Self::UcastProbes(_)
63            | Self::McastProbes(_)
64            | Self::ProxyQlen(_)
65            | Self::QueueLenbytes(_)
66            | Self::McastReprobes(_) => 4,
67
68            Self::ReachableTime(_)
69            | Self::BaseReachableTime(_)
70            | Self::RetransTime(_)
71            | Self::GcStaletime(_)
72            | Self::DelayProbeTime(_)
73            | Self::AnycastDelay(_)
74            | Self::ProxyDelay(_)
75            | Self::Locktime(_)
76            | Self::IntervalProbeTimeMs(_) => 8,
77
78            Self::Other(nla) => nla.value_len(),
79        }
80    }
81
82    fn emit_value(&self, buffer: &mut [u8]) {
83        match self {
84            Self::Ifindex(v)
85            | Self::ReferenceCount(v)
86            | Self::QueueLen(v)
87            | Self::AppProbes(v)
88            | Self::UcastProbes(v)
89            | Self::McastProbes(v)
90            | Self::ProxyQlen(v)
91            | Self::QueueLenbytes(v)
92            | Self::McastReprobes(v) => NativeEndian::write_u32(buffer, *v),
93
94            Self::ReachableTime(v)
95            | Self::BaseReachableTime(v)
96            | Self::RetransTime(v)
97            | Self::GcStaletime(v)
98            | Self::DelayProbeTime(v)
99            | Self::AnycastDelay(v)
100            | Self::ProxyDelay(v)
101            | Self::Locktime(v)
102            | Self::IntervalProbeTimeMs(v) => {
103                NativeEndian::write_u64(buffer, *v)
104            }
105
106            Self::Other(nla) => nla.emit_value(buffer),
107        }
108    }
109
110    fn kind(&self) -> u16 {
111        match self {
112            Self::Ifindex(_) => NDTPA_IFINDEX,
113            Self::ReferenceCount(_) => NDTPA_REFCNT,
114            Self::ReachableTime(_) => NDTPA_REACHABLE_TIME,
115            Self::BaseReachableTime(_) => NDTPA_BASE_REACHABLE_TIME,
116            Self::RetransTime(_) => NDTPA_RETRANS_TIME,
117            Self::GcStaletime(_) => NDTPA_GC_STALETIME,
118            Self::DelayProbeTime(_) => NDTPA_DELAY_PROBE_TIME,
119            Self::QueueLen(_) => NDTPA_QUEUE_LEN,
120            Self::AppProbes(_) => NDTPA_APP_PROBES,
121            Self::UcastProbes(_) => NDTPA_UCAST_PROBES,
122            Self::McastProbes(_) => NDTPA_MCAST_PROBES,
123            Self::AnycastDelay(_) => NDTPA_ANYCAST_DELAY,
124            Self::ProxyDelay(_) => NDTPA_PROXY_DELAY,
125            Self::ProxyQlen(_) => NDTPA_PROXY_QLEN,
126            Self::Locktime(_) => NDTPA_LOCKTIME,
127            Self::QueueLenbytes(_) => NDTPA_QUEUE_LENBYTES,
128            Self::McastReprobes(_) => NDTPA_MCAST_REPROBES,
129            Self::IntervalProbeTimeMs(_) => NDTPA_INTERVAL_PROBE_TIME_MS,
130            Self::Other(nla) => nla.kind(),
131        }
132    }
133}

135impl<'a, T: AsRef<[u8]> + ?Sized> Parseable<NlaBuffer<&'a T>>
136    for NeighbourTableParameter
137{
138    fn parse(buf: &NlaBuffer<&'a T>) -> Result<Self, DecodeError> {
139        let payload = buf.value();
140        Ok(match buf.kind() {
141            NDTPA_IFINDEX => {
142                Self::Ifindex(parse_u32(payload).context(format!(
143                    "invalid NDTPA_IFINDEX value {payload:?}"
144                ))?)
145            }
146            NDTPA_REFCNT => {
147                Self::ReferenceCount(parse_u32(payload).context(format!(
148                    "invalid NDTPA_REFCNT value {payload:?}"
149                ))?)
150            }
151            NDTPA_REACHABLE_TIME => {
152                Self::ReachableTime(parse_u64(payload).context(format!(
153                    "invalid NDTPA_REACHABLE_TIME value {payload:?}"
154                ))?)
155            }
156            NDTPA_BASE_REACHABLE_TIME => {
157                Self::BaseReachableTime(parse_u64(payload).context(format!(
158                    "invalid NDTPA_BASE_REACHABLE_TIME value {payload:?}"
159                ))?)
160            }
161            NDTPA_RETRANS_TIME => {
162                Self::RetransTime(parse_u64(payload).context(format!(
163                    "invalid NDTPA_RETRANS_TIME value {payload:?}"
164                ))?)
165            }
166            NDTPA_GC_STALETIME => {
167                Self::GcStaletime(parse_u64(payload).context(format!(
168                    "invalid NDTPA_GC_STALE_TIME value {payload:?}"
169                ))?)
170            }
171            NDTPA_DELAY_PROBE_TIME => {
172                Self::DelayProbeTime(parse_u64(payload).context(format!(
173                    "invalid NDTPA_DELAY_PROBE_TIME value {payload:?}"
174                ))?)
175            }
176            NDTPA_QUEUE_LEN => Self::QueueLen(parse_u32(payload).context(
177                format!("invalid NDTPA_QUEUE_LEN value {payload:?}"),
178            )?),
179            NDTPA_APP_PROBES => Self::AppProbes(parse_u32(payload).context(
180                format!("invalid NDTPA_APP_PROBES value {payload:?}"),
181            )?),
182            NDTPA_UCAST_PROBES => {
183                Self::UcastProbes(parse_u32(payload).context(format!(
184                    "invalid NDTPA_UCAST_PROBES value {payload:?}"
185                ))?)
186            }
187            NDTPA_MCAST_PROBES => {
188                Self::McastProbes(parse_u32(payload).context(format!(
189                    "invalid NDTPA_MCAST_PROBES value {payload:?}"
190                ))?)
191            }
192            NDTPA_ANYCAST_DELAY => {
193                Self::AnycastDelay(parse_u64(payload).context(format!(
194                    "invalid NDTPA_ANYCAST_DELAY value {payload:?}"
195                ))?)
196            }
197            NDTPA_PROXY_DELAY => Self::ProxyDelay(parse_u64(payload).context(
198                format!("invalid NDTPA_PROXY_DELAY value {payload:?}"),
199            )?),
200            NDTPA_PROXY_QLEN => Self::ProxyQlen(parse_u32(payload).context(
201                format!("invalid NDTPA_PROXY_QLEN value {payload:?}"),
202            )?),
203            NDTPA_LOCKTIME => Self::Locktime(parse_u64(payload).context(
204                format!("invalid NDTPA_LOCKTIME value {payload:?}"),
205            )?),
206            NDTPA_QUEUE_LENBYTES => {
207                Self::QueueLenbytes(parse_u32(payload).context(format!(
208                    "invalid NDTPA_QUEUE_LENBYTES value {payload:?}"
209                ))?)
210            }
211            NDTPA_MCAST_REPROBES => {
212                Self::McastReprobes(parse_u32(payload).context(format!(
213                    "invalid NDTPA_MCAST_PROBES value {payload:?}"
214                ))?)
215            }
216            NDTPA_INTERVAL_PROBE_TIME_MS => Self::IntervalProbeTimeMs(
217                parse_u64(payload).context(format!(
218                    "invalid NDTPA_INTERVAL_PROBE_TIME_MS value {payload:?}"
219                ))?,
220            ),
221            _ => Self::Other(DefaultNla::parse(buf).context(format!(
222                "invalid NDTA_PARMS attribute {payload:?}"
223            ))?),
224        })
225    }
226}

228#[derive(Debug, PartialEq, Eq, Clone)]
229pub(crate) struct VecNeighbourTableParameter(
230    pub(crate) Vec<NeighbourTableParameter>,
231);
232
233impl<'a, T: AsRef<[u8]> + ?Sized> Parseable<NlaBuffer<&'a T>>
234    for VecNeighbourTableParameter
235{
236    fn parse(buf: &NlaBuffer<&'a T>) -> Result<Self, DecodeError> {
237        let mut nlas = vec![];
238        let err = "invalid NDTA_PARMS attribute";
239        for nla in NlasIterator::new(buf.into_inner()) {
240            let nla = nla.context(err)?;
241            nlas.push(NeighbourTableParameter::parse(&nla).context(err)?);
242        }
243        Ok(Self(nlas))
244    }
