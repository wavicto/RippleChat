10use crate::{
11    address::{AddressAttribute, AddressHeaderFlags, AddressScope},

15const ADDRESS_HEADER_LEN: usize = 8;
16
17buffer!(AddressMessageBuffer(ADDRESS_HEADER_LEN) {
18    family: (u8, 0),
19    prefix_len: (u8, 1),
20    flags: (u8, 2),
21    scope: (u8, 3),
22    index: (u32, 4..ADDRESS_HEADER_LEN),
23    payload: (slice, ADDRESS_HEADER_LEN..),
24});
25
26impl<'a, T: AsRef<[u8]> + ?Sized> AddressMessageBuffer<&'a T> {
27    pub fn attributes(
28        &self,
29    ) -> impl Iterator<Item = Result<NlaBuffer<&'a [u8]>, DecodeError>> {
30        NlasIterator::new(self.payload())
31    }
32}
33
34#[derive(Debug, PartialEq, Eq, Clone, Default)]
35#[non_exhaustive]
36pub struct AddressMessage {
37    pub header: AddressHeader,
38    pub attributes: Vec<AddressAttribute>,
39}
40
41#[derive(Debug, PartialEq, Eq, Clone, Default)]
42pub struct AddressHeader {
43    pub family: AddressFamily,
44    pub prefix_len: u8,
45    pub flags: AddressHeaderFlags,
46    pub scope: AddressScope,
47    pub index: u32,
48}
49
50impl Emitable for AddressHeader {
51    fn buffer_len(&self) -> usize {
52        ADDRESS_HEADER_LEN
53    }
54
55    fn emit(&self, buffer: &mut [u8]) {
56        let mut packet = AddressMessageBuffer::new(buffer);
57        packet.set_family(self.family.into());
58        packet.set_prefix_len(self.prefix_len);
59        packet.set_flags(self.flags.bits());
60        packet.set_scope(self.scope.into());
61        packet.set_index(self.index);
62    }
63}
64
65impl Emitable for AddressMessage {
66    fn buffer_len(&self) -> usize {
67        self.header.buffer_len() + self.attributes.as_slice().buffer_len()
68    }
69
70    fn emit(&self, buffer: &mut [u8]) {
71        self.header.emit(buffer);
72        self.attributes
73            .as_slice()
74            .emit(&mut buffer[self.header.buffer_len()..]);
75    }
76}
77
78impl<T: AsRef<[u8]>> Parseable<AddressMessageBuffer<T>> for AddressHeader {
79    fn parse(buf: &AddressMessageBuffer<T>) -> Result<Self, DecodeError> {
80        Ok(Self {
81            family: buf.family().into(),
82            prefix_len: buf.prefix_len(),
83            flags: AddressHeaderFlags::from_bits_retain(buf.flags()),
84            scope: buf.scope().into(),
85            index: buf.index(),
86        })
87    }
88}
89
90impl<'a, T: AsRef<[u8]> + 'a> Parseable<AddressMessageBuffer<&'a T>>
91    for AddressMessage
92{
93    fn parse(buf: &AddressMessageBuffer<&'a T>) -> Result<Self, DecodeError> {
94        Ok(AddressMessage {
95            header: AddressHeader::parse(buf)
96                .context("failed to parse address message header")?,
97            attributes: Vec::<AddressAttribute>::parse(buf)
98                .context("failed to parse address message NLAs")?,
99        })
100    }
101}
102
103impl<'a, T: AsRef<[u8]> + 'a> Parseable<AddressMessageBuffer<&'a T>>
104    for Vec<AddressAttribute>
105{
106    fn parse(buf: &AddressMessageBuffer<&'a T>) -> Result<Self, DecodeError> {
107        let mut attributes = vec![];
108        for nla_buf in buf.attributes() {
109            attributes.push(AddressAttribute::parse(&nla_buf?)?);
110        }
111        Ok(attributes)
112    }
113}
