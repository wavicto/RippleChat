121    NewNetnsId(i32),

177impl Nla for LinkAttribute {
178    fn value_len(&self) -> usize {
179        match self {
180            Self::VfInfoList(v) => v.as_slice().buffer_len(),
181            Self::VfPorts(v) => v.as_slice().buffer_len(),
182            Self::PortSelf(v) => v.buffer_len(),
183            Self::PhysPortId(v) => v.buffer_len(),
184            Self::PhysSwitchId(v) => v.buffer_len(),
185            Self::Event(v) => v.buffer_len(),
186            Self::Wireless(v) => v.buffer_len(),
187            Self::ProtoInfoBridge(v) => v.as_slice().buffer_len(),
188            Self::ProtoInfoInet6(v) => v.as_slice().buffer_len(),
189            Self::ProtoDownReason(v) => v.as_slice().buffer_len(),
190
191            Self::Address(bytes)
192            | Self::Broadcast(bytes)
193            | Self::PermAddress(bytes)
194            | Self::AfSpecUnknown(bytes) => bytes.len(),
195
196            Self::IfName(string)
197            | Self::Qdisc(string)
198            | Self::IfAlias(string)
199            | Self::PhysPortName(string) => string.len() + 1,
200
201            Self::Mode(_) | Self::Carrier(_) | Self::ProtoDown(_) => 1,
202
203            Self::Mtu(_)
204            | Self::NewNetnsId(_)
205            | Self::IfNetnsId(_)
206            | Self::Link(_)
207            | Self::Controller(_)
208            | Self::TxQueueLen(_)
209            | Self::NetNsPid(_)
210            | Self::NumVf(_)
211            | Self::Group(_)
212            | Self::NetNsFd(_)
213            | Self::ExtMask(_)
214            | Self::Promiscuity(_)
215            | Self::NumTxQueues(_)
216            | Self::NumRxQueues(_)
217            | Self::CarrierChanges(_)
218            | Self::GsoMaxSegs(_)
219            | Self::GsoMaxSize(_)
220            | Self::LinkNetNsId(_)
221            | Self::MinMtu(_)
222            | Self::CarrierUpCount(_)
223            | Self::CarrierDownCount(_)
224            | Self::NewIfIndex(_)
225            | Self::MaxMtu(_) => 4,
226
227            Self::OperState(_) => 1,
228            Self::Stats(_) => LINK_STATS_LEN,
229            Self::Stats64(_) => LINK_STATS64_LEN,
230            Self::Map(nla) => nla.buffer_len(),
231            Self::LinkInfo(nlas) => nlas.as_slice().buffer_len(),
232            Self::Xdp(nlas) => nlas.as_slice().buffer_len(),
233            Self::PropList(nlas) => nlas.as_slice().buffer_len(),
234            Self::AfSpecUnspec(nlas) => nlas.as_slice().buffer_len(),
235            Self::AfSpecBridge(nlas) => nlas.as_slice().buffer_len(),
236            Self::ProtoInfoUnknown(attr) => attr.value_len(),
237            Self::Other(attr) => attr.value_len(),
238        }
239    }

241    fn emit_value(&self, buffer: &mut [u8]) {
242        match self {
243            Self::VfInfoList(v) => v.as_slice().emit(buffer),
244            Self::VfPorts(v) => v.as_slice().emit(buffer),
245            Self::PortSelf(v) => v.emit(buffer),
246            Self::PhysPortId(v) => v.emit(buffer),
247            Self::PhysSwitchId(v) => v.emit(buffer),
248            Self::Event(v) => v.emit(buffer),
249            Self::Wireless(v) => v.emit(buffer),
250            Self::ProtoInfoBridge(v) => v.as_slice().emit(buffer),
251            Self::ProtoInfoInet6(v) => v.as_slice().emit(buffer),
252            Self::ProtoDownReason(v) => v.as_slice().emit(buffer),
253            Self::Address(bytes)
254            | Self::Broadcast(bytes)
255            | Self::PermAddress(bytes)
256            | Self::AfSpecUnknown(bytes) => {
257                buffer.copy_from_slice(bytes.as_slice())
258            }
259
260            Self::IfName(string)
261            | Self::Qdisc(string)
262            | Self::IfAlias(string)
263            | Self::PhysPortName(string) => {
264                buffer[..string.len()].copy_from_slice(string.as_bytes());
265                buffer[string.len()] = 0;
266            }
267
268            Self::Mode(val) | Self::Carrier(val) | Self::ProtoDown(val) => {
269                buffer[0] = *val
270            }
271
272            Self::Mtu(value)
273            | Self::Link(value)
274            | Self::Controller(value)
275            | Self::TxQueueLen(value)
276            | Self::NetNsPid(value)
277            | Self::NumVf(value)
278            | Self::Group(value)
279            | Self::Promiscuity(value)
280            | Self::NumTxQueues(value)
281            | Self::NumRxQueues(value)
282            | Self::CarrierChanges(value)
283            | Self::CarrierUpCount(value)
284            | Self::CarrierDownCount(value)
285            | Self::GsoMaxSegs(value)
286            | Self::GsoMaxSize(value)
287            | Self::MinMtu(value)
288            | Self::MaxMtu(value) => NativeEndian::write_u32(buffer, *value),
289
290            Self::ExtMask(value) => NativeEndian::write_u32(
291                buffer,
292                u32::from(&VecLinkExtentMask(value.to_vec())),
293            ),
294
295            Self::LinkNetNsId(v)
296            | Self::NetNsFd(v)
297            | Self::NewNetnsId(v)
298            | Self::NewIfIndex(v)
299            | Self::IfNetnsId(v) => NativeEndian::write_i32(buffer, *v),
300            Self::Stats(nla) => nla.emit(buffer),
301            Self::Map(nla) => nla.emit(buffer),
302            Self::Stats64(nla) => nla.emit(buffer),
303            Self::OperState(state) => buffer[0] = (*state).into(),
304            Self::LinkInfo(nlas) => nlas.as_slice().emit(buffer),
305            Self::Xdp(nlas) => nlas.as_slice().emit(buffer),
306            Self::PropList(nlas) => nlas.as_slice().emit(buffer),
307            Self::AfSpecUnspec(nlas) => nlas.as_slice().emit(buffer),
308            Self::AfSpecBridge(nlas) => nlas.as_slice().emit(buffer),
309            Self::ProtoInfoUnknown(attr) | Self::Other(attr) => {
310                attr.emit_value(buffer)
311            }
312        }
313    }

315    fn kind(&self) -> u16 {
316        match self {
317            Self::VfInfoList(_) => IFLA_VFINFO_LIST,
318            Self::VfPorts(_) => IFLA_VF_PORTS,
319            Self::PortSelf(_) => IFLA_PORT_SELF,
320            Self::PhysPortId(_) => IFLA_PHYS_PORT_ID,
321            Self::PhysSwitchId(_) => IFLA_PHYS_SWITCH_ID,
322            Self::LinkInfo(_) => IFLA_LINKINFO,
323            Self::Wireless(_) => IFLA_WIRELESS,
324            Self::ProtoInfoBridge(_) | Self::ProtoInfoInet6(_) => IFLA_PROTINFO,
325            Self::ProtoInfoUnknown(attr) => attr.kind(),
326            Self::Xdp(_) => IFLA_XDP,
327            Self::Event(_) => IFLA_EVENT,
328            Self::NewNetnsId(_) => IFLA_NEW_NETNSID,
329            Self::IfNetnsId(_) => IFLA_IF_NETNSID,
330            Self::CarrierUpCount(_) => IFLA_CARRIER_UP_COUNT,
331            Self::CarrierDownCount(_) => IFLA_CARRIER_DOWN_COUNT,
332            Self::NewIfIndex(_) => IFLA_NEW_IFINDEX,
333            Self::PropList(_) => IFLA_PROP_LIST | NLA_F_NESTED,
334            Self::ProtoDownReason(_) => IFLA_PROTO_DOWN_REASON,
335            Self::Address(_) => IFLA_ADDRESS,
336            Self::Broadcast(_) => IFLA_BROADCAST,
337            Self::PermAddress(_) => IFLA_PERM_ADDRESS,
338            Self::IfName(_) => IFLA_IFNAME,
339            Self::Qdisc(_) => IFLA_QDISC,
340            Self::IfAlias(_) => IFLA_IFALIAS,
341            Self::PhysPortName(_) => IFLA_PHYS_PORT_NAME,
342            Self::Mode(_) => IFLA_LINKMODE,
343            Self::Carrier(_) => IFLA_CARRIER,
344            Self::ProtoDown(_) => IFLA_PROTO_DOWN,
345            Self::Mtu(_) => IFLA_MTU,
346            Self::Link(_) => IFLA_LINK,
347            Self::Controller(_) => IFLA_MASTER,
348            Self::TxQueueLen(_) => IFLA_TXQLEN,
349            Self::NetNsPid(_) => IFLA_NET_NS_PID,
350            Self::NumVf(_) => IFLA_NUM_VF,
351            Self::Group(_) => IFLA_GROUP,
352            Self::NetNsFd(_) => IFLA_NET_NS_FD,
353            Self::ExtMask(_) => IFLA_EXT_MASK,
354            Self::Promiscuity(_) => IFLA_PROMISCUITY,
355            Self::NumTxQueues(_) => IFLA_NUM_TX_QUEUES,
356            Self::NumRxQueues(_) => IFLA_NUM_RX_QUEUES,
357            Self::CarrierChanges(_) => IFLA_CARRIER_CHANGES,
358            Self::GsoMaxSegs(_) => IFLA_GSO_MAX_SEGS,
359            Self::GsoMaxSize(_) => IFLA_GSO_MAX_SIZE,
360            Self::MinMtu(_) => IFLA_MIN_MTU,
361            Self::MaxMtu(_) => IFLA_MAX_MTU,
362            Self::LinkNetNsId(_) => IFLA_LINK_NETNSID,
363            Self::OperState(_) => IFLA_OPERSTATE,
364            Self::Map(_) => IFLA_MAP,
365            Self::Stats(_) => IFLA_STATS,
366            Self::Stats64(_) => IFLA_STATS64,
367            Self::AfSpecUnspec(_)
368            | Self::AfSpecBridge(_)
369            | Self::AfSpecUnknown(_) => IFLA_AF_SPEC,
370            Self::Other(attr) => attr.kind(),
371        }
372    }

375impl<'a, T: AsRef<[u8]> + ?Sized>
376    ParseableParametrized<NlaBuffer<&'a T>, AddressFamily> for LinkAttribute
377{
378    fn parse_with_param(
379        buf: &NlaBuffer<&'a T>,
380        interface_family: AddressFamily,
381    ) -> Result<Self, DecodeError> {
382        let payload = buf.value();
383        Ok(match buf.kind() {
384            IFLA_VFINFO_LIST => {
385                let err =
386                    |payload| format!("invalid IFLA_VFINFO_LIST {payload:?}");
387                if !payload.is_empty() {
388                    Self::VfInfoList(
389                        VecLinkVfInfo::parse(
390                            &NlaBuffer::new_checked(payload)
391                                .context(err(payload))?,
392                        )
393                        .context(err(payload))?
394                        .0,
395                    )
396                } else {
397                    // Empty IFLA_VFINFO_LIST, this is likely a netdevsim device
398                    // no need to parse it, it is empty
399                    Self::VfInfoList(vec![])
400                }
401            }
402            IFLA_VF_PORTS => {
403                let err =
404                    |payload| format!("invalid IFLA_VF_PORTS {payload:?}");
405                Self::VfPorts(
406                    VecLinkVfPort::parse(
407                        &NlaBuffer::new_checked(payload)
408                            .context(err(payload))?,
409                    )
410                    .context(err(payload))?
411                    .0,
412                )
413            }
414            IFLA_PORT_SELF => {
415                let err =
416                    |payload| format!("invalid IFLA_PORT_SELF {payload:?}");
417                Self::PortSelf(
418                    LinkVfPort::parse(
419                        &NlaBuffer::new_checked(payload)
420                            .context(err(payload))?,
421                    )
422                    .context(err(payload))?,
423                )
424            }
425            IFLA_PHYS_PORT_ID => {
426                Self::PhysPortId(LinkPhysId::parse(payload).context(
427                    format!("invalid IFLA_PHYS_PORT_ID value {payload:?}"),
428                )?)
429            }
430            IFLA_PHYS_SWITCH_ID => {
431                Self::PhysSwitchId(LinkPhysId::parse(payload).context(
432                    format!("invalid IFLA_PHYS_SWITCH_ID value {payload:?}"),
433                )?)
434            }
435            IFLA_WIRELESS => Self::Wireless(
436                LinkWirelessEvent::parse(payload)
437                    .context(format!("invalid IFLA_WIRELESS {payload:?}"))?,
438            ),
439            IFLA_PROTINFO => {
440                let err = |payload| {
441                    format!("invalid IFLA_PROTINFO for AF_INET6 {payload:?}")
442                };
443                match interface_family {
444                    AddressFamily::Inet6 => Self::ProtoInfoInet6(
445                        VecLinkProtoInfoInet6::parse(
446                            &NlaBuffer::new_checked(payload)
447                                .context(err(payload))?,
448                        )
449                        .context(err(payload))?
450                        .0,
451                    ),
452                    #[cfg(any(
453                        target_os = "linux",
454                        target_os = "fuchsia",
455                        target_os = "android"
456                    ))]
457                    AddressFamily::Bridge => Self::ProtoInfoBridge(
458                        VecLinkProtoInfoBridge::parse(&NlaBuffer::new_checked(
459                            payload,
460                        )?)
461                        .context(format!(
462                            "invalid IFLA_PROTINFO for AF_INET6 {payload:?}"
463                        ))?
464                        .0,
465                    ),
466                    _ => Self::ProtoInfoUnknown(
467                        DefaultNla::parse(buf).context(format!(
468                            "invalid IFLA_PROTINFO for {interface_family:?}: \
469                             {payload:?}"
470                        ))?,
471                    ),
472                }
473            }
474            IFLA_EVENT => Self::Event(
475                LinkEvent::parse(payload)
476                    .context(format!("invalid IFLA_EVENT {payload:?}"))?,
477            ),
478            IFLA_NEW_NETNSID => Self::NewNetnsId(
479                parse_i32(payload).context("invalid IFLA_NEW_NETNSID value")?,
480            ),
481            IFLA_IF_NETNSID => Self::IfNetnsId(
482                parse_i32(payload).context("invalid IFLA_IF_NETNSID value")?,
483            ),
484            IFLA_CARRIER_UP_COUNT => Self::CarrierUpCount(
485                parse_u32(payload)
486                    .context("invalid IFLA_CARRIER_UP_COUNT value")?,
487            ),
488            IFLA_CARRIER_DOWN_COUNT => Self::CarrierDownCount(
489                parse_u32(payload)
490                    .context("invalid IFLA_CARRIER_DOWN_COUNT value")?,
491            ),
492            IFLA_NEW_IFINDEX => Self::NewIfIndex(
493                parse_i32(payload).context("invalid IFLA_NEW_IFINDEX value")?,
494            ),
495
496            IFLA_PROP_LIST => {
497                let error_msg = "invalid IFLA_PROP_LIST value";
498                let mut nlas = vec![];
499                for nla in NlasIterator::new(payload) {
500                    let nla = &nla.context(error_msg)?;
501                    let parsed = Prop::parse(nla).context(error_msg)?;
502                    nlas.push(parsed);
503                }
504                Self::PropList(nlas)
505            }
506            IFLA_PROTO_DOWN_REASON => {
507                let mut nlas = vec![];
508                for nla in NlasIterator::new(payload) {
509                    let nla =
510                        &nla.context("invalid IFLA_PROTO_DOWN_REASON value")?;
511                    let parsed = LinkProtocolDownReason::parse(nla)?;
512                    nlas.push(parsed);
513                }
514                Self::ProtoDownReason(nlas)
515            }
516            // HW address (we parse them as Vec for now, because for IP over
517            // GRE, the HW address is an IP instead of a MAC for
518            // example
519            IFLA_ADDRESS => Self::Address(payload.to_vec()),
520            IFLA_BROADCAST => Self::Broadcast(payload.to_vec()),
521            IFLA_PERM_ADDRESS => Self::PermAddress(payload.to_vec()),
522            // String
523            IFLA_IFNAME => Self::IfName(
524                parse_string(payload).context("invalid IFLA_IFNAME value")?,
525            ),
526            IFLA_QDISC => Self::Qdisc(
527                parse_string(payload).context("invalid IFLA_QDISC value")?,
528            ),
529            IFLA_IFALIAS => Self::IfAlias(
530                parse_string(payload).context("invalid IFLA_IFALIAS value")?,
531            ),
532            IFLA_PHYS_PORT_NAME => Self::PhysPortName(
533                parse_string(payload)
534                    .context("invalid IFLA_PHYS_PORT_NAME value")?,
535            ),
536            IFLA_LINKMODE => Self::Mode(
537                parse_u8(payload).context("invalid IFLA_LINKMODE value")?,
538            ),
539            IFLA_CARRIER => Self::Carrier(
540                parse_u8(payload).context("invalid IFLA_CARRIER value")?,
541            ),
542            IFLA_PROTO_DOWN => Self::ProtoDown(
543                parse_u8(payload).context("invalid IFLA_PROTO_DOWN value")?,
544            ),
545
546            IFLA_MTU => {
547                Self::Mtu(parse_u32(payload).context("invalid IFLA_MTU value")?)
548            }
549            IFLA_LINK => Self::Link(
550                parse_u32(payload).context("invalid IFLA_LINK value")?,
551            ),
552            IFLA_MASTER => Self::Controller(
553                parse_u32(payload).context("invalid IFLA_MASTER value")?,
554            ),
555            IFLA_TXQLEN => Self::TxQueueLen(
556                parse_u32(payload).context("invalid IFLA_TXQLEN value")?,
557            ),
558            IFLA_NET_NS_PID => Self::NetNsPid(
559                parse_u32(payload).context("invalid IFLA_NET_NS_PID value")?,
560            ),
561            IFLA_NUM_VF => Self::NumVf(
562                parse_u32(payload).context("invalid IFLA_NUM_VF value")?,
563            ),
564            IFLA_GROUP => Self::Group(
565                parse_u32(payload).context("invalid IFLA_GROUP value")?,
566            ),
567            IFLA_NET_NS_FD => Self::NetNsFd(
568                parse_i32(payload).context("invalid IFLA_NET_NS_FD value")?,
569            ),
570            IFLA_EXT_MASK => Self::ExtMask(
571                VecLinkExtentMask::from(
572                    parse_u32(payload)
573                        .context("invalid IFLA_EXT_MASK value")?,
574                )
575                .0,
576            ),
577            IFLA_PROMISCUITY => Self::Promiscuity(
578                parse_u32(payload).context("invalid IFLA_PROMISCUITY value")?,
579            ),
580            IFLA_NUM_TX_QUEUES => Self::NumTxQueues(
581                parse_u32(payload)
582                    .context("invalid IFLA_NUM_TX_QUEUES value")?,
583            ),
584            IFLA_NUM_RX_QUEUES => Self::NumRxQueues(
585                parse_u32(payload)
586                    .context("invalid IFLA_NUM_RX_QUEUES value")?,
587            ),
588            IFLA_CARRIER_CHANGES => Self::CarrierChanges(
589                parse_u32(payload)
590                    .context("invalid IFLA_CARRIER_CHANGES value")?,
591            ),
592            IFLA_GSO_MAX_SEGS => Self::GsoMaxSegs(
593                parse_u32(payload)
594                    .context("invalid IFLA_GSO_MAX_SEGS value")?,
595            ),
596            IFLA_GSO_MAX_SIZE => Self::GsoMaxSize(
597                parse_u32(payload)
598                    .context("invalid IFLA_GSO_MAX_SIZE value")?,
599            ),
600            IFLA_MIN_MTU => Self::MinMtu(
601                parse_u32(payload).context("invalid IFLA_MIN_MTU value")?,
602            ),
603            IFLA_MAX_MTU => Self::MaxMtu(
604                parse_u32(payload).context("invalid IFLA_MAX_MTU value")?,
605            ),
606            IFLA_LINK_NETNSID => Self::LinkNetNsId(
607                parse_i32(payload)
608                    .context("invalid IFLA_LINK_NETNSID value")?,
609            ),
610            IFLA_OPERSTATE => Self::OperState(
611                parse_u8(payload)
612                    .context("invalid IFLA_OPERSTATE value")?
613                    .into(),
614            ),
615            IFLA_MAP => {
616                let err =
617                    |payload| format!("Invalid IFLA_MAP value {payload:?}");
618                Self::Map(
619                    super::Map::parse(
620                        &MapBuffer::new_checked(payload)
621                            .context(err(payload))?,
622                    )
623                    .context(err(payload))?,
624                )
625            }
626            IFLA_STATS => Self::Stats(
627                super::Stats::parse(&StatsBuffer::new(
628                    expand_buffer_if_small(
629                        payload,
630                        LINK_STATS_LEN,
631                        "IFLA_STATS",
632                    )
633                    .as_slice(),
634                ))
635                .context(format!("Invalid IFLA_STATS value {payload:?}"))?,
636            ),
637            IFLA_STATS64 => {
638                let payload = expand_buffer_if_small(
639                    payload,
640                    LINK_STATS64_LEN,
641                    "IFLA_STATS64",
642                );
643                Self::Stats64(
644                    super::Stats64::parse(&Stats64Buffer::new(
645                        payload.as_slice(),
646                    ))
647                    .context(format!(
648                        "Invalid IFLA_STATS64 value {payload:?}"
649                    ))?,
650                )
651            }
652            IFLA_AF_SPEC => match interface_family {
653                AddressFamily::Unspec => {
654                    let err = "invalid IFLA_AF_SPEC value for AF_UNSPEC";
655                    Self::AfSpecUnspec(
656                        VecAfSpecUnspec::parse(
657                            &NlaBuffer::new_checked(&buf.value())
658                                .context(err)?,
659                        )
660                        .context(err)?
661                        .0,
662                    )
663                }
664                #[cfg(any(
665                    target_os = "linux",
666                    target_os = "fuchsia",
667                    target_os = "android"
668                ))]
669                AddressFamily::Bridge => {
670                    let err = "invalid IFLA_AF_SPEC value for AF_BRIDGE";
671                    Self::AfSpecBridge(
672                        VecAfSpecBridge::parse(
673                            &NlaBuffer::new_checked(&buf.value())
674                                .context(err)?,
675                        )
676                        .context(err)?
677                        .0,
678                    )
679                }
680                _ => Self::AfSpecUnknown(payload.to_vec()),
681            },
682            IFLA_LINKINFO => {
683                let err = "invalid IFLA_LINKINFO value";
684                Self::LinkInfo(
685                    VecLinkInfo::parse(
686                        &NlaBuffer::new_checked(&buf.value()).context(err)?,
687                    )
688                    .context(err)?
689                    .0,
690                )
691            }
692            IFLA_XDP => {
693                let err = "invalid IFLA_XDP value";
694                let buf = NlaBuffer::new_checked(payload).context(err)?;
695                Self::Xdp(VecLinkXdp::parse(&buf).context(err)?.0)
696            }
697            kind => Self::Other(
698                DefaultNla::parse(buf)
699                    .context(format!("unknown NLA type {kind}"))?,
700            ),
701        })
702    }
