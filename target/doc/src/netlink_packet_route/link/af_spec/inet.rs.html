12const IFLA_INET_CONF: u16 = 1;
13// This number might change when kernel add more IPV4_DEV_CONF
14const __IPV4_DEVCONF_MAX: usize = 34;
15const IPV4_DEVCONF_MAX: usize = __IPV4_DEVCONF_MAX - 1;
16const DEV_CONF_LEN: usize = IPV4_DEVCONF_MAX * 4;
17
18#[derive(Clone, Eq, PartialEq, Debug)]
19#[non_exhaustive]
20pub enum AfSpecInet {
21    DevConf(InetDevConf),
22    Other(DefaultNla),
23}
24
25pub(crate) struct VecAfSpecInet(pub(crate) Vec<AfSpecInet>);
26
27impl<'a, T: AsRef<[u8]> + ?Sized> Parseable<NlaBuffer<&'a T>>
28    for VecAfSpecInet
29{
30    fn parse(buf: &NlaBuffer<&'a T>) -> Result<Self, DecodeError> {
31        let mut nlas = vec![];
32        let err = "Invalid AF_INET NLA for IFLA_AF_SPEC(AF_UNSPEC)";
33        for nla in NlasIterator::new(buf.into_inner()) {
34            let nla = nla.context(err)?;
35            nlas.push(AfSpecInet::parse(&nla)?);
36        }
37        Ok(Self(nlas))
38    }
39}

41impl nla::Nla for AfSpecInet {
42    fn value_len(&self) -> usize {
43        use self::AfSpecInet::*;
44        match *self {
45            DevConf(ref c) => c.buffer_len(),
46            Other(ref nla) => nla.value_len(),
47        }
48    }
49
50    fn emit_value(&self, buffer: &mut [u8]) {
51        use self::AfSpecInet::*;
52        match *self {
53            DevConf(ref c) => c.emit(buffer),
54            Other(ref nla) => nla.emit_value(buffer),
55        }
56    }
57
58    fn kind(&self) -> u16 {
59        use self::AfSpecInet::*;
60        match *self {
61            DevConf(_) => IFLA_INET_CONF,
62            Other(ref nla) => nla.kind(),
63        }
64    }
65}

67impl<'a, T: AsRef<[u8]> + ?Sized> Parseable<NlaBuffer<&'a T>> for AfSpecInet {
68    fn parse(buf: &NlaBuffer<&'a T>) -> Result<Self, DecodeError> {
69        use self::AfSpecInet::*;
70
71        let payload = buf.value();
72        Ok(match buf.kind() {
73            IFLA_INET_CONF => {
74                DevConf(InetDevConf::parse(&InetDevConfBuffer::new(
75                    expand_buffer_if_small(
76                        payload,
77                        DEV_CONF_LEN,
78                        "IFLA_INET_CONF",
79                    )
80                    .as_slice(),
81                ))?)
82            }
83            kind => Other(DefaultNla::parse(buf).context(format!(
84                "Unknown NLA type {kind} for IFLA_AF_SPEC(inet)"
85            ))?),
86        })
87    }
88}
