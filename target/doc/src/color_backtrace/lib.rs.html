1//! Colorful and clean backtraces on panic.
2//!
3//! This library aims to make panics a little less painful by nicely colorizing
4//! them, skipping over frames of functions called after the panic was already
5//! initiated and printing relevant source snippets. Frames of functions in your
6//! application are colored in a different color (red) than those of
7//! dependencies (green).
8//!
9//! ### Screenshot
10//!
11//! ![Screenshot](https://i.imgur.com/yzp0KH6.png)
12//!
13//! ### Features
14//!
15//! - Colorize backtraces to be easier on the eyes
16//! - Show source snippets if source files are found on disk
17//! - Print frames of application code vs dependencies in different color
18//! - Hide all the frames after the panic was already initiated
19//! - Hide language runtime initialization frames
20//!
21//! ### Installing the panic handler
22//!
23//! In your main function, just insert the following snippet. That's it!
24//! ```rust
25//! color_backtrace::install();
26//! ```
27//!
28//! If you want to customize some settings, you can instead do:
29//! ```rust
30//! use color_backtrace::{default_output_stream, BacktracePrinter};
31//! BacktracePrinter::new().message("Custom message!").install(default_output_stream());
32//! ```
33//!
34//! ### Controlling verbosity
35//!
36//! The default verbosity is configured via the `RUST_BACKTRACE` environment
37//! variable. An unset `RUST_BACKTRACE` corresponds to
38//! [minimal](Verbosity::Minimal), `RUST_BACKTRACE=1` to
39//! [medium](Verbosity::Medium) and `RUST_BACKTRACE=full` to
40//! [full](Verbosity::Full) verbosity levels.
41

67/// Defines how verbose the backtrace is supposed to be.
68#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
69pub enum Verbosity {
70    /// Print a small message including the panic payload and the panic location.
71    Minimal,
72    /// Everything in `Minimal` and additionally print a backtrace.
73    Medium,
74    /// Everything in `Medium` plus source snippets for all backtrace locations.
75    Full,
76}
77
78impl Verbosity {
79    /// Get the verbosity level from `RUST_BACKTRACE` env variable.
80    pub fn from_env() -> Self {
81        Self::convert_env(env::var("RUST_BACKTRACE").ok())
82    }
83
84    /// Get the verbosity level from `RUST_LIB_BACKTRACE` env variable,
85    /// falling back to the `RUST_BACKTRACE`.
86    pub fn lib_from_env() -> Self {
87        Self::convert_env(
88            env::var("RUST_LIB_BACKTRACE")
89                .or_else(|_| env::var("RUST_BACKTRACE"))
90                .ok(),
91        )
92    }
93
94    fn convert_env(env: Option<String>) -> Self {
95        match env {
96            Some(ref x) if x == "full" => Verbosity::Full,
97            Some(_) => Verbosity::Medium,
98            None => Verbosity::Minimal,
99        }
100    }
101}

119/// Create the default output stream.
120///
121/// If stderr is attached to a tty, this is a colorized stderr, else it's
122/// a plain (colorless) stderr.
123pub fn default_output_stream() -> Box<StandardStream> {
124    Box::new(StandardStream::stderr(if std::io::stderr().is_terminal() {
125        ColorChoice::Always
126    } else {
127        ColorChoice::Never
128    }))
129}

131#[doc(hidden)]
132#[deprecated(
133    since = "0.4.0",
134    note = "Use `BacktracePrinter::into_panic_handler()` instead."
135)]
136pub fn create_panic_handler(
137    printer: BacktracePrinter,
138) -> Box<dyn Fn(&PanicInfo<'_>) + 'static + Sync + Send> {
139    let out_stream_mutex = Mutex::new(default_output_stream());
140    Box::new(move |pi| {
141        let mut lock = out_stream_mutex.lock().unwrap();
142        if let Err(e) = printer.print_panic_info(pi, &mut *lock) {
143            // Panicking while handling a panic would send us into a deadlock,
144            // so we just print the error to stderr instead.
145            eprintln!("Error while printing panic: {:?}", e);
146        }
147    })
148}

152pub fn install_with_settings(printer: BacktracePrinter) {
153    std::panic::set_hook(printer.into_panic_handler(default_output_stream()))
154}

160/// Abstraction over backtrace library implementations.
161pub trait Backtrace {
162    fn frames(&self) -> Vec<Frame>;
163}

165#[cfg(feature = "use-backtrace-crate")]
166impl Backtrace for backtrace::Backtrace {
167    fn frames(&self) -> Vec<Frame> {
168        backtrace::Backtrace::frames(self)
169            .iter()
170            .flat_map(|frame| frame.symbols().iter().map(move |sym| (frame.ip(), sym)))
171            .zip(1usize..)
172            .map(|((ip, sym), n)| Frame {
173                name: sym.name().map(|x| x.to_string()),
174                lineno: sym.lineno(),
175                filename: sym.filename().map(|x| x.into()),
176                n,
177                ip: Some(ip as usize),
178                _private_ctor: (),
179            })
180            .collect()
181    }
182}
183
184#[cfg(feature = "use-btparse-crate")]
185impl Backtrace for btparse::Backtrace {
186    fn frames(&self) -> Vec<Frame> {
187        self.frames
188            .iter()
189            .zip(1usize..)
190            .map(|(frame, n)| Frame {
191                n,
192                name: Some(frame.function.clone()),
193                lineno: frame.line.map(|x| x as u32),
194                filename: frame.file.as_ref().map(|x| x.clone().into()),
195                ip: None,
196                _private_ctor: (),
197            })
198            .collect()
199    }
200}

202// Capture a backtrace with the most reliable available backtrace implementation.
203fn capture_backtrace() -> Result<Box<dyn Backtrace>, Box<dyn std::error::Error>> {
204    #[cfg(all(feature = "use-backtrace-crate", feature = "use-btparse-crate"))]
205    return Ok(Box::new(backtrace::Backtrace::new()));
206
207    #[cfg(all(feature = "use-backtrace-crate", not(feature = "use-btparse-crate")))]
208    return Ok(Box::new(backtrace::Backtrace::new()));
209
210    #[cfg(all(not(feature = "use-backtrace-crate"), feature = "use-btparse-crate"))]
211    {
212        let bt = std::backtrace::Backtrace::force_capture();
213        return Ok(Box::new(btparse::deserialize(&bt).map_err(Box::new)?));
214    }
215
216    #[cfg(all(
217        not(feature = "use-backtrace-crate"),
218        not(feature = "use-btparse-crate")
219    ))]
220    {
221        return Err(Box::new(std::io::Error::new(
222            std::io::ErrorKind::Other,
223            "need to enable at least one backtrace crate selector feature",
224        )));
225    }
226}

234#[derive(Debug)]
235pub struct Frame {
236    pub n: usize,
237    pub name: Option<String>,
238    pub lineno: Option<u32>,
239    pub filename: Option<PathBuf>,
240    pub ip: Option<usize>,
241    _private_ctor: (),
242}

244impl Frame {
245    /// Heuristically determine whether the frame is likely to be part of a
246    /// dependency.
247    ///
248    /// If it fails to detect some patterns in your code base, feel free to drop
249    /// an issue / a pull request!
250    pub fn is_dependency_code(&self) -> bool {
251        const SYM_PREFIXES: &[&str] = &[
252            "std::",
253            "core::",
254            "backtrace::backtrace::",
255            "_rust_begin_unwind",
256            "color_traceback::",
257            "__rust_",
258            "___rust_",
259            "__pthread",
260            "_main",
261            "main",
262            "__scrt_common_main_seh",
263            "BaseThreadInitThunk",
264            "_start",
265            "__libc_start_main",
266            "start_thread",
267        ];
268
269        // Inspect name.
270        if let Some(ref name) = self.name {
271            if SYM_PREFIXES.iter().any(|x| name.starts_with(x)) {
272                return true;
273            }
274        }
275
276        const FILE_PREFIXES: &[&str] = &[
277            "/rustc/",
278            "src/libstd/",
279            "src/libpanic_unwind/",
280            "src/libtest/",
281        ];
282
283        // Inspect filename.
284        if let Some(ref filename) = self.filename {
285            let filename = filename.to_string_lossy();
286            if FILE_PREFIXES.iter().any(|x| filename.starts_with(x))
287                || filename.contains("/.cargo/registry/src/")
288            {
289                return true;
290            }
291        }
292
293        false
294    }
295
296    /// Heuristically determine whether a frame is likely to be a post panic
297    /// frame.
298    ///
299    /// Post panic frames are frames of a functions called after the actual panic
300    /// is already in progress and don't contain any useful information for a
301    /// reader of the backtrace.
302    pub fn is_post_panic_code(&self) -> bool {
303        const SYM_PREFIXES: &[&str] = &[
304            "_rust_begin_unwind",
305            "rust_begin_unwind",
306            "core::result::unwrap_failed",
307            "core::option::expect_none_failed",
308            "core::panicking::panic_fmt",
309            "color_backtrace::create_panic_handler",
310            "std::panicking::begin_panic",
311            "begin_panic_fmt",
312            "backtrace::capture",
313        ];
314
315        match self.name.as_ref() {
316            Some(name) => SYM_PREFIXES.iter().any(|x| name.starts_with(x)),
317            None => false,
318        }
319    }
320
321    /// Heuristically determine whether a frame is likely to be part of language
322    /// runtime.
323    pub fn is_runtime_init_code(&self) -> bool {
324        const SYM_PREFIXES: &[&str] = &[
325            "std::rt::lang_start::",
326            "test::run_test::run_test_inner::",
327            "std::sys_common::backtrace::__rust_begin_short_backtrace",
328        ];
329
330        let (name, file) = match (self.name.as_ref(), self.filename.as_ref()) {
331            (Some(name), Some(filename)) => (name, filename.to_string_lossy()),
332            _ => return false,
333        };
334
335        if SYM_PREFIXES.iter().any(|x| name.starts_with(x)) {
336            return true;
337        }
338
339        // For Linux, this is the best rule for skipping test init I found.
340        if name == "{{closure}}" && file == "src/libtest/lib.rs" {
341            return true;
342        }
343
344        false
345    }
346
347    fn print_source_if_avail(&self, mut out: impl WriteColor, s: &BacktracePrinter) -> IOResult {
348        let (lineno, filename) = match (self.lineno, self.filename.as_ref()) {
349            (Some(a), Some(b)) => (a, b),
350            // Without a line number and file name, we can't sensibly proceed.
351            _ => return Ok(()),
352        };
353
354        let file = match File::open(filename) {
355            Ok(file) => file,
356            Err(ref e) if e.kind() == ErrorKind::NotFound => return Ok(()),
357            e @ Err(_) => e?,
358        };
359
360        // Extract relevant lines.
361        let reader = BufReader::new(file);
362        let start_line = lineno - 2.min(lineno - 1);
363        let surrounding_src = reader.lines().skip(start_line as usize - 1).take(5);
364        for (line, cur_line_no) in surrounding_src.zip(start_line..) {
365            if cur_line_no == lineno {
366                // Print actual source line with brighter color.
367                out.set_color(&s.colors.selected_src_ln)?;
368                writeln!(out, "{:>8} > {}", cur_line_no, line?)?;
369                out.reset()?;
370            } else {
371                writeln!(out, "{:>8} │ {}", cur_line_no, line?)?;
372            }
373        }
374
375        Ok(())
376    }

378    /// Get the module's name by walking /proc/self/maps
379    #[cfg(all(
380        feature = "resolve-modules",
381        unix,
382        not(any(target_os = "macos", target_os = "ios"))
383    ))]
384    fn module_info(&self) -> Option<(String, usize)> {
385        use regex::Regex;
386        use std::path::Path;
387
388        let ip = match self.ip {
389            Some(x) => x,
390            None => return None,
391        };
392
393        let re = Regex::new(
394            r"(?x)
395                ^
396                (?P<start>[0-9a-f]{8,16})
397                -
398                (?P<end>[0-9a-f]{8,16})
399                \s
400                (?P<perm>[-rwxp]{4})
401                \s
402                (?P<offset>[0-9a-f]{8})
403                \s
404                [0-9a-f]+:[0-9a-f]+
405                \s
406                [0-9]+
407                \s+
408                (?P<path>.*)
409                $
410            ",
411        )
412        .unwrap();
413
414        let mapsfile = File::open("/proc/self/maps").expect("Unable to open /proc/self/maps");
415
416        for line in BufReader::new(mapsfile).lines() {
417            let line = line.unwrap();
418            if let Some(caps) = re.captures(&line) {
419                let (start, end, path) = (
420                    usize::from_str_radix(caps.name("start").unwrap().as_str(), 16).unwrap(),
421                    usize::from_str_radix(caps.name("end").unwrap().as_str(), 16).unwrap(),
422                    caps.name("path").unwrap().as_str().to_string(),
423                );
424                if ip >= start && ip < end {
425                    return if let Some(filename) = Path::new(&path).file_name() {
426                        Some((filename.to_str().unwrap().to_string(), start))
427                    } else {
428                        None
429                    };
430                }
431            }
432        }
433
434        None
435    }
436
437    #[cfg(not(all(
438        feature = "resolve-modules",
439        unix,
440        not(any(target_os = "macos", target_os = "ios"))
441    )))]
442    fn module_info(&self) -> Option<(String, usize)> {
443        None
444    }
445
446    fn print(&self, i: usize, out: &mut impl WriteColor, s: &BacktracePrinter) -> IOResult {
447        let is_dependency_code = self.is_dependency_code();
448
449        // Print frame index.
450        write!(out, "{:>2}: ", i)?;
451
452        if let Some(ip) = self.ip {
453            if s.should_print_addresses() {
454                if let Some((module_name, module_base)) = self.module_info() {
455                    write!(out, "{}:0x{:08x} - ", module_name, ip - module_base)?;
456                } else {
457                    write!(out, "0x{:016x} - ", ip)?;
458                }
459            }
460        }
461
462        // Does the function have a hash suffix?
463        // (dodging a dep on the regex crate here)
464        let name = self
465            .name
466            .as_ref()
467            .map(|s| s.as_str())
468            .unwrap_or("<unknown>");
469        let has_hash_suffix = name.len() > 19
470            && &name[name.len() - 19..name.len() - 16] == "::h"
471            && name[name.len() - 16..].chars().all(|x| x.is_digit(16));
472
473        // Print function name.
474        out.set_color(if is_dependency_code {
475            &s.colors.dependency_code
476        } else {
477            &s.colors.crate_code
478        })?;
479
480        if has_hash_suffix {
481            write!(out, "{}", &name[..name.len() - 19])?;
482            if s.strip_function_hash {
483                writeln!(out)?;
484            } else {
485                out.set_color(if is_dependency_code {
486                    &s.colors.dependency_code_hash
487                } else {
488                    &s.colors.crate_code_hash
489                })?;
490                writeln!(out, "{}", &name[name.len() - 19..])?;
491            }
492        } else {
493            writeln!(out, "{}", name)?;
494        }
495
496        out.reset()?;
497
498        // Print source location, if known.
499        if let Some(ref file) = self.filename {
500            let filestr = file.to_str().unwrap_or("<bad utf8>");
501            let lineno = self
502                .lineno
503                .map_or("<unknown line>".to_owned(), |x| x.to_string());
504            writeln!(out, "    at {}:{}", filestr, lineno)?;
505        } else {
506            writeln!(out, "    at <unknown source file>")?;
507        }
508
509        // Maybe print source.
510        if s.current_verbosity() >= Verbosity::Full {
511            self.print_source_if_avail(out, s)?;
512        }
513
514        Ok(())
515    }
516}

518/// The default frame filter. Heuristically determines whether a frame is likely to be an
519/// uninteresting frame. This filters out post panic frames and runtime init frames and dependency
520/// code.
521pub fn default_frame_filter(frames: &mut Vec<&Frame>) {
522    let top_cutoff = frames
523        .iter()
524        .rposition(|x| x.is_post_panic_code())
525        .map(|x| x + 2) // indices are 1 based
526        .unwrap_or(0);
527
528    let bottom_cutoff = frames
529        .iter()
530        .position(|x| x.is_runtime_init_code())
531        .unwrap_or_else(|| frames.len());
532
533    let rng = top_cutoff..=bottom_cutoff;
534    frames.retain(|x| rng.contains(&x.n))
535}

595/// Pretty-printer for backtraces and [`PanicInfo`] structs.

597pub struct BacktracePrinter {
598    message: String,
599    verbosity: Verbosity,
600    lib_verbosity: Verbosity,
601    strip_function_hash: bool,
602    is_panic_handler: bool,
603    colors: ColorScheme,
604    filters: Vec<Arc<FilterCallback>>,
605    should_print_addresses: bool,
606}
607
608impl Default for BacktracePrinter {
609    fn default() -> Self {
610        Self {
611            verbosity: Verbosity::from_env(),
612            lib_verbosity: Verbosity::lib_from_env(),
613            message: "The application panicked (crashed).".to_owned(),
614            strip_function_hash: false,
615            colors: ColorScheme::classic(),
616            is_panic_handler: false,
617            filters: vec![Arc::new(default_frame_filter)],
618            should_print_addresses: false,
619        }
620    }
621}

631            .field("print_addresses", &self.should_print_addresses)
632            .field("colors", &self.colors)

634    }

697    ///
698    /// ```rust
699    /// use color_backtrace::{default_output_stream, BacktracePrinter};
700    ///
701    /// BacktracePrinter::new()
702    ///     .add_frame_filter(Box::new(|frames| {
703    ///         frames.retain(|x| matches!(&x.name, Some(n) if !n.starts_with("blabla")))
704    ///     }))
705    ///     .install(default_output_stream());
706    /// ```
707    pub fn add_frame_filter(mut self, filter: Box<FilterCallback>) -> Self {
708        self.filters.push(filter.into());
709        self
710    }
711
712    /// Clears all filters associated with this printer, including the default filter
713    pub fn clear_frame_filters(mut self) -> Self {
714        self.filters.clear();
715        self
716    }
717}

719/// Routines for putting the panic printer to use.
720impl BacktracePrinter {
721    /// Install the `color_backtrace` handler with default settings.
722    ///
723    /// Output streams can be created via `default_output_stream()` or
724    /// using any other stream that implements [`termcolor::WriteColor`].
725    pub fn install(self, out: impl WriteColor + Sync + Send + 'static) {
726        std::panic::set_hook(self.into_panic_handler(out))
727    }
728
729    /// Create a `color_backtrace` panic handler from this panic printer.
730    ///
731    /// This can be used if you want to combine the handler with other handlers.
732    pub fn into_panic_handler(
733        mut self,
734        out: impl WriteColor + Sync + Send + 'static,
735    ) -> Box<dyn Fn(&PanicInfo<'_>) + 'static + Sync + Send> {
736        self.is_panic_handler = true;
737        let out_stream_mutex = Mutex::new(out);
738        Box::new(move |pi| {
739            let mut lock = out_stream_mutex.lock().unwrap();
740            if let Err(e) = self.print_panic_info(pi, &mut *lock) {
741                // Panicking while handling a panic would send us into a deadlock,
742                // so we just print the error to stderr instead.
743                eprintln!("Error while printing panic: {:?}", e);
744            }
745        })
746    }
747
748    /// Pretty-prints a [`backtrace::Backtrace`] to an output stream.
749    pub fn print_trace(&self, trace: &dyn Backtrace, out: &mut impl WriteColor) -> IOResult {
750        writeln!(out, "{:━^80}", " BACKTRACE ")?;
751
752        // Collect frame info.
753        let frames = trace.frames();
754
755        let mut filtered_frames = frames.iter().collect();
756        match env::var("COLORBT_SHOW_HIDDEN").ok().as_deref() {
757            Some("1") | Some("on") | Some("y") => (),
758            _ => {
759                for filter in &self.filters {
760                    filter(&mut filtered_frames);
761                }
762            }
763        }
764
765        if filtered_frames.is_empty() {
766            // TODO: Would probably look better centered.
767            return writeln!(out, "<empty backtrace>");
768        }
769
770        // Don't let filters mess with the order.
771        filtered_frames.sort_by_key(|x| x.n);
772
773        macro_rules! print_hidden {
774            ($n:expr) => {
775                out.set_color(&self.colors.frames_omitted_msg)?;
776                let n = $n;
777                let text = format!(
778                    "{decorator} {n} frame{plural} hidden {decorator}",
779                    n = n,
780                    plural = if n == 1 { "" } else { "s" },
781                    decorator = "⋮",
782                );
783                writeln!(out, "{:^80}", text)?;
784                out.reset()?;
785            };
786        }
787
788        let mut last_n = 0;
789        for frame in &filtered_frames {
790            let frame_delta = frame.n - last_n - 1;
791            if frame_delta != 0 {
792                print_hidden!(frame_delta);
793            }
794            frame.print(frame.n, out, self)?;
795            last_n = frame.n;
796        }
797
798        let last_filtered_n = filtered_frames.last().unwrap().n;
799        let last_unfiltered_n = frames.last().unwrap().n;
800        if last_filtered_n < last_unfiltered_n {
801            print_hidden!(last_unfiltered_n - last_filtered_n);
802        }
803
804        Ok(())
805    }
806
807    /// Pretty-print a backtrace to a `String`, using VT100 color codes.
808    pub fn format_trace_to_string(&self, trace: &dyn Backtrace) -> IOResult<String> {
809        // TODO: should we implicitly enable VT100 support on Windows here?
810        let mut ansi = Ansi::new(vec![]);
811        self.print_trace(trace, &mut ansi)?;
812        Ok(String::from_utf8(ansi.into_inner()).unwrap())
813    }
814
815    /// Pretty-prints a [`PanicInfo`] struct to an output stream.
816    pub fn print_panic_info(&self, pi: &PanicInfo, out: &mut impl WriteColor) -> IOResult {
817        out.set_color(&self.colors.header)?;
818        writeln!(out, "{}", self.message)?;
819        out.reset()?;
820
821        // Print panic message.
822        let payload = pi
823            .payload()
824            .downcast_ref::<String>()
825            .map(String::as_str)
826            .or_else(|| pi.payload().downcast_ref::<&str>().cloned())
827            .unwrap_or("<non string panic payload>");
828
829        write!(out, "Message:  ")?;
830        out.set_color(&self.colors.msg_loc_prefix)?;
831        writeln!(out, "{}", payload)?;
832        out.reset()?;
833
834        // If known, print panic location.
835        write!(out, "Location: ")?;
836        if let Some(loc) = pi.location() {
837            out.set_color(&self.colors.src_loc)?;
838            write!(out, "{}", loc.file())?;
839            out.set_color(&self.colors.src_loc_separator)?;
840            write!(out, ":")?;
841            out.set_color(&self.colors.src_loc)?;
842            writeln!(out, "{}", loc.line())?;
843            out.reset()?;
844        } else {
845            writeln!(out, "<unknown>")?;
846        }
847
848        // Print some info on how to increase verbosity.
849        if self.current_verbosity() == Verbosity::Minimal {
850            write!(out, "\nBacktrace omitted.\n\nRun with ")?;
851            out.set_color(&self.colors.env_var)?;
852            write!(out, "RUST_BACKTRACE=1")?;
853            out.reset()?;
854            writeln!(out, " environment variable to display it.")?;
855        } else {
856            // This text only makes sense if frames are displayed.
857            write!(out, "\nRun with ")?;
858            out.set_color(&self.colors.env_var)?;
859            write!(out, "COLORBT_SHOW_HIDDEN=1")?;
860            out.reset()?;
861            writeln!(out, " environment variable to disable frame filtering.")?;
862        }
863        if self.current_verbosity() <= Verbosity::Medium {
864            write!(out, "Run with ")?;
865            out.set_color(&self.colors.env_var)?;
866            write!(out, "RUST_BACKTRACE=full")?;
867            out.reset()?;
868            writeln!(out, " to include source snippets.")?;
869        }
870
871        if self.current_verbosity() >= Verbosity::Medium {
872            match capture_backtrace() {
873                Ok(trace) => self.print_trace(&*trace, out)?,
874                Err(e) => {
875                    out.set_color(&self.colors.header)?;
876                    writeln!(out, "\nFailed to capture backtrace: {e}")?;
877                    out.reset()?;
878                }
879            }
880        }
881
882        Ok(())
883    }
884
885    fn current_verbosity(&self) -> Verbosity {
886        if self.is_panic_handler {
887            self.verbosity
888        } else {
889            self.lib_verbosity
890        }
891    }
892
893    fn should_print_addresses(&self) -> bool {
894        self.should_print_addresses
895    }
896}
