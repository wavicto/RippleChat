22/// Represents a UDP Packet.
23#[packet]
24pub struct Udp {
25    pub source: u16be,
26    pub destination: u16be,
27    pub length: u16be,
28    pub checksum: u16be,
29    #[payload]
30    pub payload: Vec<u8>,
31}
32
33/// Calculate a checksum for a packet built on IPv4.
34pub fn ipv4_checksum(packet: &UdpPacket, source: &Ipv4Addr, destination: &Ipv4Addr) -> u16be {
35    ipv4_checksum_adv(packet, &[], source, destination)
36}
37
38/// Calculate a checksum for a packet built on IPv4. Advanced version which
39/// accepts an extra slice of data that will be included in the checksum
40/// as being part of the data portion of the packet.
41///
42/// If `packet` contains an odd number of bytes the last byte will not be
43/// counted as the first byte of a word together with the first byte of
44/// `extra_data`.
45pub fn ipv4_checksum_adv(packet: &UdpPacket,
46                         extra_data: &[u8],
47                         source: &Ipv4Addr,
48                         destination: &Ipv4Addr)
49    -> u16be {
50    util::ipv4_checksum(packet.packet(),
51                        3,
52                        extra_data,
53                        source,
54                        destination,
55                        IpNextHeaderProtocols::Udp)
56}

58#[test]
59fn udp_header_ipv4_test() {
60    use crate::ip::IpNextHeaderProtocols;
61    use crate::ipv4::MutableIpv4Packet;
62
63    let mut packet = [0u8; 20 + 8 + 4];
64    let ipv4_source = Ipv4Addr::new(192, 168, 0, 1);
65    let ipv4_destination = Ipv4Addr::new(192, 168, 0, 199);
66    {
67        let mut ip_header = MutableIpv4Packet::new(&mut packet[..]).unwrap();
68        ip_header.set_next_level_protocol(IpNextHeaderProtocols::Udp);
69        ip_header.set_source(ipv4_source);
70        ip_header.set_destination(ipv4_destination);
71    }
72
73    // Set data
74    packet[20 + 8] = 't' as u8;
75    packet[20 + 8 + 1] = 'e' as u8;
76    packet[20 + 8 + 2] = 's' as u8;
77    packet[20 + 8 + 3] = 't' as u8;
78
79    {
80        let mut udp_header = MutableUdpPacket::new(&mut packet[20..]).unwrap();
81        udp_header.set_source(12345);
82        assert_eq!(udp_header.get_source(), 12345);
83
84        udp_header.set_destination(54321);
85        assert_eq!(udp_header.get_destination(), 54321);
86
87        udp_header.set_length(8 + 4);
88        assert_eq!(udp_header.get_length(), 8 + 4);
89
90        let checksum = ipv4_checksum(&udp_header.to_immutable(), &ipv4_source, &ipv4_destination);
91        udp_header.set_checksum(checksum);
92        assert_eq!(udp_header.get_checksum(), 0x9178);
93    }
94
95    let ref_packet = [0x30, 0x39, /* source */
96                      0xd4, 0x31, /* destination */
97                      0x00, 0x0c, /* length */
98                      0x91, 0x78  /* checksum */];
99    assert_eq!(&ref_packet[..], &packet[20..28]);
100}
