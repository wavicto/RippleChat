19/// Represents an Ethernet packet.
20#[packet]
21pub struct Ethernet {
22    #[construct_with(u8, u8, u8, u8, u8, u8)]
23    pub destination: MacAddr,
24    #[construct_with(u8, u8, u8, u8, u8, u8)]
25    pub source: MacAddr,
26    #[construct_with(u16)]
27    pub ethertype: EtherType,
28    #[payload]
29    pub payload: Vec<u8>,
30}

32#[test]
33fn ethernet_header_test() {
34    let mut packet = [0u8; 14];
35    {
36        let mut ethernet_header = MutableEthernetPacket::new(&mut packet[..]).unwrap();
37
38        let source = MacAddr(0x12, 0x34, 0x56, 0x78, 0x9a, 0xbc);
39        ethernet_header.set_source(source);
40        assert_eq!(ethernet_header.get_source(), source);
41
42        let dest = MacAddr(0xde, 0xf0, 0x12, 0x34, 0x45, 0x67);
43        ethernet_header.set_destination(dest);
44        assert_eq!(ethernet_header.get_destination(), dest);
45
46        ethernet_header.set_ethertype(EtherTypes::Ipv6);
47        assert_eq!(ethernet_header.get_ethertype(), EtherTypes::Ipv6);
48    }
49
50    let ref_packet = [0xde, 0xf0, 0x12, 0x34, 0x45, 0x67, /* destination */
51                      0x12, 0x34, 0x56, 0x78, 0x9a, 0xbc, /* source */
52                      0x86, 0xdd /* ethertype */];
53    assert_eq!(&ref_packet[..], &packet[..]);
54}

56/// `EtherTypes` are defined at:
57/// <http://www.iana.org/assignments/ieee-802-numbers/ieee-802-numbers.xhtml>.
58/// These values should be used in the `Ethernet` `EtherType` field.
59///
60/// FIXME Should include all
61/// A handful of these have been selected since most are archaic and unused.
62#[allow(non_snake_case)]
63#[allow(non_upper_case_globals)]
64pub mod EtherTypes {
65    use crate::ethernet::EtherType;
66
67    /// Internet Protocol version 4 (IPv4) \[RFC7042\].
68    pub const Ipv4: EtherType = EtherType(0x0800);
69    /// Address Resolution Protocol (ARP) \[RFC7042\].
70    pub const Arp: EtherType = EtherType(0x0806);
71    /// Wake on Lan.
72    pub const WakeOnLan: EtherType = EtherType(0x0842);
73    /// IETF TRILL Protocol \[IEEE\].
74    pub const Trill: EtherType = EtherType(0x22F3);
75    /// DECnet Phase IV.
76    pub const DECnet: EtherType = EtherType(0x6003);
77    /// Reverse Address Resolution Protocol (RARP) \[RFC903\].
78    pub const Rarp: EtherType = EtherType(0x8035); 
79    /// AppleTalk - EtherTalk \[Apple\].
80    pub const AppleTalk: EtherType = EtherType(0x809B);
81    /// AppleTalk Address Resolution Protocol (AARP) \[Apple\].
82    pub const Aarp: EtherType = EtherType(0x80F3);
83    /// IPX \[Xerox\].
84    pub const Ipx: EtherType = EtherType(0x8137);
85    /// QNX Qnet \[QNX Software Systems\].
86    pub const Qnx: EtherType = EtherType(0x8204);
87    /// Internet Protocol version 6 (IPv6) \[RFC7042\].
88    pub const Ipv6: EtherType = EtherType(0x86DD);
89    /// Ethernet Flow Control \[IEEE 802.3x\].
90    pub const FlowControl: EtherType = EtherType(0x8808);
91    /// CobraNet \[CobraNet\].
92    pub const CobraNet: EtherType = EtherType(0x8819);
93    /// MPLS Unicast \[RFC 3032\].
94    pub const Mpls: EtherType = EtherType(0x8847);
95    /// MPLS Multicast \[RFC 5332\].
96    pub const MplsMcast: EtherType = EtherType(0x8848);
97    /// PPPOE Discovery Stage \[RFC 2516\].
98    pub const PppoeDiscovery: EtherType = EtherType(0x8863);
99    /// PPPoE Session Stage \[RFC 2516\].
100    pub const PppoeSession: EtherType = EtherType(0x8864);
101    /// VLAN-tagged frame (IEEE 802.1Q).
102    pub const Vlan: EtherType = EtherType(0x8100);
103    /// Provider Bridging \[IEEE 802.1ad / IEEE 802.1aq\].
104    pub const PBridge: EtherType = EtherType(0x88a8);
105    /// Link Layer Discovery Protocol (LLDP) \[IEEE 802.1AB\].
106    pub const Lldp: EtherType = EtherType(0x88cc);
107    /// Precision Time Protocol (PTP) over Ethernet \[IEEE 1588\].
108    pub const Ptp: EtherType = EtherType(0x88f7);
109    /// CFM / Y.1731 \[IEEE 802.1ag\].
110    pub const Cfm: EtherType = EtherType(0x8902);
111    /// Q-in-Q Vlan Tagging \[IEEE 802.1Q\].
112    pub const QinQ: EtherType = EtherType(0x9100);
113}

115/// Represents the `Ethernet::ethertype` field.
116#[derive(Copy, Clone, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
117pub struct EtherType(pub u16);
118
119impl EtherType {
120    /// Construct a new `EtherType` instance.
121    pub fn new(val: u16) -> EtherType {
122        EtherType(val)
123    }
124}
125
126impl PrimitiveValues for EtherType {
127    type T = (u16,);
128    fn to_primitive_values(&self) -> (u16,) {
129        (self.0,)
130    }
131}
132
133impl fmt::Display for EtherType {
134    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
135        write!(f,
136               "{}",
137               match self {
138                   &EtherTypes::Ipv4 => "Ipv4", //(0x0800)
139                   &EtherTypes::Arp => "Arp", //(0x0806)
140                   &EtherTypes::WakeOnLan => "WakeOnLan", //(0x0842)
141                   &EtherTypes::Trill => "Trill", //(0x22F3)
142                   &EtherTypes::DECnet => "DECnet", //(0x6003)
143                   &EtherTypes::Rarp => "Rarp", //(0x8035)
144                   &EtherTypes::AppleTalk => "AppleTalk", //(0x809B)
145                   &EtherTypes::Aarp => "Aarp", //(0x80F3)
146                   &EtherTypes::Ipx => "Ipx", //(0x8137)
147                   &EtherTypes::Qnx => "Qnx", //(0x8204)
148                   &EtherTypes::Ipv6 => "Ipv6", //(0x86DD)
149                   &EtherTypes::FlowControl => "FlowControl", //(0x8808)
150                   &EtherTypes::CobraNet => "CobraNet", //(0x8819)
151                   &EtherTypes::Mpls => "Mpls", //(0x8847)
152                   &EtherTypes::MplsMcast => "MplsMcast", //(0x8848)
153                   &EtherTypes::PppoeDiscovery => "PppoeDiscovery", //(0x8863)
154                   &EtherTypes::PppoeSession => "PppoeSession", //(0x8864)
155                   &EtherTypes::Vlan => "Vlan", //(0x8100)
156                   &EtherTypes::PBridge => "PBridge", //(0x88a8)
157                   &EtherTypes::Lldp => "Lldp", //(0x88cc)
158                   &EtherTypes::Ptp => "Ptp", //(0x88f7)
159                   &EtherTypes::Cfm => "Cfm", //(0x8902)
160                   &EtherTypes::QinQ => "QinQ", //(0x9100)
161                   _ => "unknown",
162               })
163    }
164}

166#[cfg(feature = "std")]
167#[test]
168fn ether_type_to_str() {
169    use std::format;
170    let ipv4 = EtherType(0x0800);
171    assert_eq!(format!("{}", ipv4), "Ipv4");
172    let arp = EtherType(0x0806);
173    assert_eq!(format!("{}", arp), "Arp");
174    let unknown = EtherType(0x0666);
175    assert_eq!(format!("{}", unknown), "unknown");
176}
