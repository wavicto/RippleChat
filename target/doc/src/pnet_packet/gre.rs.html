20/// GRE (Generic Routing Encapsulation) Packet.
21///
22/// See RFCs 1701, 2784, 2890, 7676, 2637
23///
24/// Current status of implementation:
25///
26/// - [RFC 1701](https://tools.ietf.org/html/rfc1701) except for source routing and checksums.
27///   Processing a source routed packet will panic. Checksums are able to be inspected, but not
28///   calculated or verified.
29///
30/// - [RFC 2784](https://tools.ietf.org/html/rfc2784) except for checksums (same as 1701 status).
31///   Note that it is possible to generate noncompliant packets by setting any of the reserved bits
32///   (but see 2890).
33///
34/// - [RFC 2890](https://tools.ietf.org/html/rfc2890) implemented.
35///
36/// - [RFC 7676](https://tools.ietf.org/html/rfc7676) has no packet changes - compliance is up to
37///   the user.
38///
39/// - [RFC 2637](https://tools.ietf.org/html/rfc2637) not implemented.
40///
41/// Note that routing information from RFC 1701 is not implemented, packets
42/// with `routing_present` true will currently cause a panic.
43#[packet]
44pub struct Gre {
45    pub checksum_present: u1,
46    pub routing_present: u1,
47    pub key_present: u1,
48    pub sequence_present: u1,
49    pub strict_source_route: u1,
50    pub recursion_control: u3,
51    pub zero_flags: u5,
52    pub version: u3,
53    pub protocol_type: u16be, // 0x800 for ipv4 [basically an ethertype
54    #[length_fn = "gre_checksum_length"]
55    pub checksum: Vec<U16BE>,
56    #[length_fn = "gre_offset_length"]
57    pub offset: Vec<U16BE>,
58    #[length_fn = "gre_key_length"]
59    pub key: Vec<U32BE>,
60    #[length_fn = "gre_sequence_length"]
61    pub sequence: Vec<U32BE>,
62    #[length_fn = "gre_routing_length"]
63    pub routing: Vec<u8>,
64    #[payload]
65    pub payload: Vec<u8>,
66}
67
68fn gre_checksum_length(gre: &GrePacket) -> usize {
69    (gre.get_checksum_present() | gre.get_routing_present()) as usize * 2
70}
71
72fn gre_offset_length(gre: &GrePacket) -> usize {
73    (gre.get_checksum_present() | gre.get_routing_present()) as usize * 2
74}
75
76fn gre_key_length(gre: &GrePacket) -> usize {
77    gre.get_key_present() as usize * 4
78}
79
80fn gre_sequence_length(gre: &GrePacket) -> usize {
81    gre.get_sequence_present() as usize * 4
82}
83
84fn gre_routing_length(gre: &GrePacket) -> usize {
85    if 0 == gre.get_routing_present() {
86        0
87    } else {
88        panic!("Source routed GRE packets not supported")
89    }
90}

111#[test]
112fn gre_packet_test() {
113    let mut packet = [0u8; 4];
114    {
115        let mut gre_packet = MutableGrePacket::new(&mut packet[..]).unwrap();
116        gre_packet.set_protocol_type(0x0800);
117        assert_eq!(gre_packet.payload().len(), 0);
118    }
119
120    let ref_packet = [0x00 /* no flags */,
121                      0x00 /* no flags, version 0 */,
122                      0x08 /* protocol 0x0800 */,
123                      0x00];
124
125    assert_eq!(&ref_packet[..], &packet[..]);
126}

128#[test]
129fn gre_checksum_test() {
130    let mut packet = [0u8; 8];
131    {
132        let mut gre_packet = MutableGrePacket::new(&mut packet[..]).unwrap();
133        gre_packet.set_checksum_present(1);
134        assert_eq!(gre_packet.payload().len(), 0);
135        assert_eq!(gre_packet.get_checksum().len(), 1);
136        assert_eq!(gre_packet.get_offset().len(), 1);
137    }
138
139    let ref_packet = [0x80 /* checksum on */,
140                      0x00 /* no flags, version 0 */,
141                      0x00 /* protocol 0x0000 */,
142                      0x00,
143                      0x00 /* 16 bits of checksum */,
144                      0x00,
145                      0x00 /* 16 bits of offset */,
146                      0x00];
147
148    assert_eq!(&ref_packet[..], &packet[..]);
149}
