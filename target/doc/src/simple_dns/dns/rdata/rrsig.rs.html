6/// An RRSIG record see [rfc4034](https://www.rfc-editor.org/rfc/rfc4034#section-3)
7#[derive(Debug, PartialEq, Eq, Hash, Clone)]
8pub struct RRSIG<'a> {
9    /// The type of RR that is covered by this RRSIG
10    pub type_covered: u16,
11    /// The cryptographic algorithm used for the signature
12    pub algorithm: u8,
13    /// The number of labels in the original RRSIG RR owner name
14    pub labels: u8,
15    /// The original TTL value of the covered record
16    pub original_ttl: u32,
17    /// When the signature expires (seconds since Jan 1 1970)
18    pub signature_expiration: u32,
19    /// When the signature was created (seconds since Jan 1 1970)
20    pub signature_inception: u32,
21    /// Key tag value of the DNSKEY RR that validates this signature
22    pub key_tag: u16,
23    /// The domain name of the zone that contains the signed RRset
24    pub signer_name: Name<'a>,
25    /// The cryptographic signature that covers the RRSIG RDATA
26    pub signature: Cow<'a, [u8]>,
27}
28
29impl RR for RRSIG<'_> {
30    const TYPE_CODE: u16 = 46;
31}
32
33impl<'a> WireFormat<'a> for RRSIG<'a> {
34    const MINIMUM_LEN: usize = 18;
35
36    fn parse_after_check(data: &'a [u8], position: &mut usize) -> crate::Result<Self>
37    where
38        Self: Sized,
39    {
40        let type_covered = u16::from_be_bytes(data[*position..*position + 2].try_into()?);
41        *position += 2;
42
43        let algorithm = data[*position];
44        *position += 1;
45
46        let labels = data[*position];
47        *position += 1;
48
49        let original_ttl = u32::from_be_bytes(data[*position..*position + 4].try_into()?);
50        *position += 4;
51
52        let signature_expiration = u32::from_be_bytes(data[*position..*position + 4].try_into()?);
53        *position += 4;
54
55        let signature_inception = u32::from_be_bytes(data[*position..*position + 4].try_into()?);
56        *position += 4;
57
58        let key_tag = u16::from_be_bytes(data[*position..*position + 2].try_into()?);
59        *position += 2;
60
61        let signer_name = Name::parse(data, position)?;
62        let signature = Cow::Borrowed(&data[*position..]);
63        *position += signature.len();
64
65        Ok(Self {
66            type_covered,
67            algorithm,
68            labels,
69            original_ttl,
70            signature_expiration,
71            signature_inception,
72            key_tag,
73            signer_name,
74            signature,
75        })
76    }
77
78    fn write_to<T: std::io::Write>(&self, out: &mut T) -> crate::Result<()> {
79        out.write_all(&self.type_covered.to_be_bytes())?;
80        out.write_all(&[self.algorithm])?;
81        out.write_all(&[self.labels])?;
82        out.write_all(&self.original_ttl.to_be_bytes())?;
83        out.write_all(&self.signature_expiration.to_be_bytes())?;
84        out.write_all(&self.signature_inception.to_be_bytes())?;
85        out.write_all(&self.key_tag.to_be_bytes())?;
86        self.signer_name.write_to(out)?;
87        out.write_all(&self.signature)?;
88
89        Ok(())
90    }
91
92    fn len(&self) -> usize {
93        self.signer_name.len() + self.signature.len() + Self::MINIMUM_LEN
94    }
95}
96
97impl RRSIG<'_> {
98    /// Transforms the inner data into its owned type
99    pub fn into_owned<'b>(self) -> RRSIG<'b> {
100        RRSIG {
101            type_covered: self.type_covered,
102            algorithm: self.algorithm,
103            labels: self.labels,
104            original_ttl: self.original_ttl,
105            signature_expiration: self.signature_expiration,
106            signature_inception: self.signature_inception,
107            key_tag: self.key_tag,
108            signer_name: self.signer_name.into_owned(),
109            signature: Cow::Owned(self.signature.into_owned()),
110        }
111    }
112}
