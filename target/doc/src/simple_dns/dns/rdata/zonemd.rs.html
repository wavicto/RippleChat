6/// A ZoneMD record see [rfc8976](https://www.rfc-editor.org/rfc/rfc8976.html)
7#[derive(Debug, PartialEq, Eq, Hash, Clone)]
8pub struct ZONEMD<'a> {
9    /// The serial number of the zone's SOA record
10    pub serial: u32,
11    /// The scheme in which data is hashed
12    pub scheme: u8,
13    /// The hashing algorithm ID to use (see [rfc8976](https://www.rfc-editor.org/rfc/rfc8976.html#name-the-hash-algorithm-field))
14    pub algorithm: u8,
15    /// The output data of the hash algorithm.
16    pub digest: Cow<'a, [u8]>,
17}
18
19impl RR for ZONEMD<'_> {
20    const TYPE_CODE: u16 = 63;
21}
22
23impl<'a> WireFormat<'a> for ZONEMD<'a> {
24    const MINIMUM_LEN: usize = 6;
25
26    fn parse_after_check(data: &'a [u8], position: &mut usize) -> crate::Result<Self>
27    where
28        Self: Sized,
29    {
30        let serial = u32::from_be_bytes(data[*position..*position + 4].try_into()?);
31        *position += 4;
32        let scheme = data[*position];
33        *position += 1;
34        let algorithm = data[*position];
35        *position += 1;
36        let digest = Cow::Borrowed(&data[*position..]);
37        *position += digest.len();
38
39        Ok(Self {
40            serial,
41            scheme,
42            algorithm,
43            digest,
44        })
45    }
46
47    fn write_to<T: std::io::Write>(&self, out: &mut T) -> crate::Result<()> {
48        out.write_all(&self.serial.to_be_bytes())?;
49        out.write_all(&[self.scheme])?;
50        out.write_all(&[self.algorithm])?;
51        out.write_all(&self.digest)?;
52
53        Ok(())
54    }
55
56    fn len(&self) -> usize {
57        self.digest.len() + Self::MINIMUM_LEN
58    }
59}
60
61impl ZONEMD<'_> {
62    /// Transforms the inner data into its owned type
63    pub fn into_owned<'b>(self) -> ZONEMD<'b> {
64        ZONEMD {
65            scheme: self.scheme,
66            serial: self.serial,
67            algorithm: self.algorithm,
68            digest: self.digest.into_owned().into(),
69        }
70    }
71}

82    #[test]
83    fn parse_and_write_srv() {
84        let zonemd = ZONEMD {
85            serial: 1,
86            scheme: 2,
87            algorithm: 3,
88            digest: Cow::Borrowed(&[4, 5, 6]),
89        };
90
91        let mut bytes = Vec::new();
92        assert!(zonemd.write_to(&mut bytes).is_ok());
93
94        let zonemd = ZONEMD::parse(&bytes, &mut 0);
95        assert!(zonemd.is_ok());
96        let zonemd = zonemd.unwrap();
97
98        assert_eq!(zonemd.serial, 1);
99        assert_eq!(zonemd.scheme, 2);
100        assert_eq!(zonemd.algorithm, 3);
101        assert_eq!(*zonemd.digest, *b"\x04\x05\x06");
102    }
103
104    #[test]
105    fn parse_sample() -> Result<(), Box<dyn std::error::Error>> {
