5/// RFC 3403: Used to map a domain name to a set of services. The fields determine
6///           the order of processing, specify the protocol and service to be used,
7///           and transform the original domain name into a new domain name or URI.
8
9#[derive(Debug, PartialEq, Eq, Hash, Clone)]
10pub struct NAPTR<'a> {
11    /// Order in which NAPTR records must be processed
12    pub order: u16,
13    /// Order in which NAPTR records with equal Order values should be processed
14    pub preference: u16,
15    /// Control rewriting and interpretation of the fields in the record
16    pub flags: CharacterString<'a>,
17    /// Service Parameters applicable to this this delegation path
18    pub services: CharacterString<'a>,
19    /// Regular expression applied to original string from client
20    pub regexp: CharacterString<'a>,
21    /// Next domain-name to query for
22    pub replacement: Name<'a>,
23}
24
25impl RR for NAPTR<'_> {
26    const TYPE_CODE: u16 = 35;
27}
28
29impl NAPTR<'_> {
30    /// Transforms the inner data into it owned type
31    pub fn into_owned<'b>(self) -> NAPTR<'b> {
32        NAPTR {
33            order: self.order,
34            preference: self.preference,
35            flags: self.flags.into_owned(),
36            services: self.services.into_owned(),
37            regexp: self.regexp.into_owned(),
38            replacement: self.replacement.into_owned(),
39        }
40    }
41}
42
43impl<'a> WireFormat<'a> for NAPTR<'a> {
44    const MINIMUM_LEN: usize = 4;
45
46    fn parse_after_check(data: &'a [u8], position: &mut usize) -> crate::Result<Self>
47    where
48        Self: Sized,
49    {
50        let order = u16::from_be_bytes(data[*position..*position + 2].try_into()?);
51        *position += 2;
52        let preference = u16::from_be_bytes(data[*position..*position + 2].try_into()?);
53        *position += 2;
54        let flags = CharacterString::parse(data, position)?;
55        let services = CharacterString::parse(data, position)?;
56        let regexp = CharacterString::parse(data, position)?;
57        let replacement = Name::parse(data, position)?;
58
59        Ok(Self {
60            order,
61            preference,
62            flags,
63            services,
64            regexp,
65            replacement,
66        })
67    }
68
69    fn write_to<T: std::io::Write>(&self, out: &mut T) -> crate::Result<()> {
70        out.write_all(&self.order.to_be_bytes())?;
71        out.write_all(&self.preference.to_be_bytes())?;
72        self.flags.write_to(out)?;
73        self.services.write_to(out)?;
74        self.regexp.write_to(out)?;
75        self.replacement.write_to(out)
76    }
77
78    fn len(&self) -> usize {
79        self.flags.len()
80            + self.services.len()
81            + self.regexp.len()
82            + self.replacement.len()
83            + Self::MINIMUM_LEN
84    }
85}

87#[cfg(test)]
88mod tests {
89    use super::*;
90
91    #[test]
92    fn parse_and_write_naptr() {
93        let naptr = NAPTR {
94            order: 0,
95            preference: 1,
96            flags: CharacterString::new(b"123abc").unwrap(),
97            services: CharacterString::new(b"test").unwrap(),
98            regexp: CharacterString::new(b"@\\w+\\.\\w{2,3}(\\.\\w{2,3})?").unwrap(),
99            replacement: Name::new("e.exchange.com").unwrap(),
100        };
101
102        let mut data = Vec::new();
103        assert!(naptr.write_to(&mut data).is_ok());
104
105        let naptr = NAPTR::parse(&data, &mut 0);
106        assert!(naptr.is_ok());
107        let naptr = naptr.unwrap();
108
109        assert_eq!(data.len(), naptr.len());
110        assert_eq!(0, naptr.order);
111        assert_eq!(1, naptr.preference);
112        assert_eq!("123abc", naptr.flags.to_string());
113        assert_eq!("test", naptr.services.to_string());
114        assert_eq!("@\\w+\\.\\w{2,3}(\\.\\w{2,3})?", naptr.regexp.to_string());
115        assert_eq!("e.exchange.com", naptr.replacement.to_string());
116    }
