20/// Input event to the topic state handler.
21#[derive(Clone, Debug)]
22pub enum InEvent<PI> {
23    /// Message received from the network.
24    RecvMessage(PI, Message<PI>),
25    /// Execute a command from the application.
26    Command(Command<PI>),
27    /// Trigger a previously scheduled timer.
28    TimerExpired(Timer<PI>),
29    /// Peer disconnected on the network level.
30    PeerDisconnected(PI),
31    /// Update the opaque peer data about yourself.
32    UpdatePeerData(PeerData),
33}
34
35/// An output event from the state handler.
36#[derive(Debug, PartialEq, Eq)]
37pub enum OutEvent<PI> {
38    /// Send a message on the network
39    SendMessage(PI, Message<PI>),
40    /// Emit an event to the application.
41    EmitEvent(Event<PI>),
42    /// Schedule a timer. The runtime is responsible for sending an [InEvent::TimerExpired]
43    /// after the duration.
44    ScheduleTimer(Duration, Timer<PI>),
45    /// Close the connection to a peer on the network level.
46    DisconnectPeer(PI),
47    /// Emitted when new [`PeerData`] was received for a peer.
48    PeerData(PI, PeerData),
49}
50
51impl<PI> From<hyparview::OutEvent<PI>> for OutEvent<PI> {
52    fn from(event: hyparview::OutEvent<PI>) -> Self {
53        use hyparview::OutEvent::*;
54        match event {
55            SendMessage(to, message) => Self::SendMessage(to, message.into()),
56            ScheduleTimer(delay, timer) => Self::ScheduleTimer(delay, timer.into()),
57            DisconnectPeer(peer) => Self::DisconnectPeer(peer),
58            EmitEvent(event) => Self::EmitEvent(event.into()),
59            PeerData(peer, data) => Self::PeerData(peer, data),
60        }
61    }
62}
63
64impl<PI> From<plumtree::OutEvent<PI>> for OutEvent<PI> {
65    fn from(event: plumtree::OutEvent<PI>) -> Self {
66        use plumtree::OutEvent::*;
67        match event {
68            SendMessage(to, message) => Self::SendMessage(to, message.into()),
69            ScheduleTimer(delay, timer) => Self::ScheduleTimer(delay, timer.into()),
70            EmitEvent(event) => Self::EmitEvent(event.into()),
71        }
72    }
73}

75/// A trait for a concrete type to push `OutEvent`s to.
76///
77/// The implementation is generic over this trait, which allows the upper layer to supply a
78/// container of their choice for `OutEvent`s emitted from the protocol state.
79pub trait IO<PI: Clone> {
80    /// Push an event in the IO container
81    fn push(&mut self, event: impl Into<OutEvent<PI>>);
82
83    /// Push all events from an iterator into the IO container
84    fn push_from_iter(&mut self, iter: impl IntoIterator<Item = impl Into<OutEvent<PI>>>) {
85        for event in iter.into_iter() {
86            self.push(event);
87        }
88    }
89}

91/// A protocol message for a particular topic
92#[derive(From, Debug, Serialize, Deserialize, Clone, PartialEq, Eq)]
93pub enum Message<PI> {
94    /// A message of the swarm membership layer
95    Swarm(hyparview::Message<PI>),
96    /// A message of the gossip broadcast layer
97    Gossip(plumtree::Message),
98}
99
100impl<PI> Message<PI> {
101    /// Get the kind of this message
102    pub fn kind(&self) -> MessageKind {
103        match self {
104            Message::Swarm(_) => MessageKind::Control,
105            Message::Gossip(message) => match message {
106                plumtree::Message::Gossip(_) => MessageKind::Data,
107                _ => MessageKind::Control,
108            },
109        }
110    }
111
112    /// Returns `true` if this is a disconnect message (which is the last message sent to a peer per topic).
113    pub fn is_disconnect(&self) -> bool {
114        matches!(self, Message::Swarm(hyparview::Message::Disconnect(_)))
115    }
116}

118/// An event to be emitted to the application for a particular topic.
119#[derive(Clone, PartialEq, Eq, PartialOrd, Ord, Debug, Serialize, Deserialize)]
120pub enum Event<PI> {
121    /// We have a new, direct neighbor in the swarm membership layer for this topic
122    NeighborUp(PI),
123    /// We dropped direct neighbor in the swarm membership layer for this topic
124    NeighborDown(PI),
125    /// A gossip message was received for this topic
126    Received(GossipEvent<PI>),
127}
128
129impl<PI> From<hyparview::Event<PI>> for Event<PI> {
130    fn from(value: hyparview::Event<PI>) -> Self {
131        match value {
132            hyparview::Event::NeighborUp(peer) => Self::NeighborUp(peer),
133            hyparview::Event::NeighborDown(peer) => Self::NeighborDown(peer),
134        }
135    }
136}
137
138impl<PI> From<plumtree::Event<PI>> for Event<PI> {
139    fn from(value: plumtree::Event<PI>) -> Self {
140        match value {
141            plumtree::Event::Received(event) => Self::Received(event),
142        }
143    }
144}

158/// A command to the protocol state for a particular topic.
159#[derive(Clone, derive_more::Debug)]
160pub enum Command<PI> {
161    /// Join this topic and connect to peers.
162    ///
163    /// If the list of peers is empty, will prepare the state and accept incoming join requests,
164    /// but only become operational after the first join request by another peer.
165    Join(Vec<PI>),
166    /// Broadcast a message for this topic.
167    Broadcast(#[debug("<{}b>", _0.len())] Bytes, Scope),
168    /// Leave this topic and drop all state.
169    Quit,
170}

174        self.push_back(event.into())
175    }

178/// Protocol configuration
179#[derive(Clone, Debug, Serialize, Deserialize)]
180#[serde(default)]
181pub struct Config {
182    /// Configuration for the swarm membership layer
183    pub membership: hyparview::Config,
184    /// Configuration for the gossip broadcast layer
185    pub broadcast: plumtree::Config,
186    /// Max message size in bytes.
187    ///
188    /// This size should be the same across a network to ensure all nodes can transmit and read large messages.
189    ///
190    /// At minimum, this size should be large enough to send gossip control messages. This can vary, depending on the size of the [`PeerIdentity`] you use and the size of the [`PeerData`] you transmit in your messages.
191    ///
192    /// The default is [`DEFAULT_MAX_MESSAGE_SIZE`].
193    pub max_message_size: usize,
194}
195
196impl Default for Config {
197    fn default() -> Self {
198        Self {
199            membership: Default::default(),
200            broadcast: Default::default(),
201            max_message_size: DEFAULT_MAX_MESSAGE_SIZE,
202        }
203    }
204}

206/// The topic state maintains the swarm membership and broadcast tree for a particular topic.
207#[derive(Debug)]
208pub struct State<PI, R> {
209    me: PI,
210    pub(crate) swarm: hyparview::State<PI, R>,
211    pub(crate) gossip: plumtree::State<PI>,
212    outbox: VecDeque<OutEvent<PI>>,
213    stats: Stats,
214}

216impl<PI: PeerIdentity> State<PI, rand::rngs::StdRng> {
217    /// Initialize the local state with the default random number generator.
218    ///
219    /// ## Panics
220    ///
221    /// Panics if [`Config::max_message_size`] is below [`MIN_MAX_MESSAGE_SIZE`].
222    pub fn new(me: PI, me_data: Option<PeerData>, config: Config) -> Self {
223        Self::with_rng(me, me_data, config, rand::rngs::StdRng::from_entropy())
224    }
225}

234impl<PI: PeerIdentity, R: Rng> State<PI, R> {
235    /// Initialize the local state with a custom random number generator.
236    ///
237    /// ## Panics
238    ///
239    /// Panics if [`Config::max_message_size`] is below [`MIN_MAX_MESSAGE_SIZE`].
240    pub fn with_rng(me: PI, me_data: Option<PeerData>, config: Config, rng: R) -> Self {
241        assert!(
242            config.max_message_size >= MIN_MAX_MESSAGE_SIZE,
243            "max_message_size must be at least {MIN_MAX_MESSAGE_SIZE}"
244        );
245        let max_payload_size =
246            config.max_message_size - super::Message::<PI>::postcard_header_size();
247        Self {
248            swarm: hyparview::State::new(me, me_data, config.membership, rng),
249            gossip: plumtree::State::new(me, config.broadcast, max_payload_size),
250            me,
251            outbox: VecDeque::new(),
252            stats: Stats::default(),
253        }
254    }
255
256    /// Handle an incoming event.
257    ///
258    /// Returns an iterator of outgoing events that must be processed by the application.
259    pub fn handle(
260        &mut self,
261        event: InEvent<PI>,
262        now: Instant,
263    ) -> impl Iterator<Item = OutEvent<PI>> + '_ {
264        let io = &mut self.outbox;
265        // Process the event, store out events in outbox.
266        match event {
267            InEvent::Command(command) => match command {
268                Command::Join(peers) => {
269                    for peer in peers {
270                        self.swarm.handle(SwarmIn::RequestJoin(peer), io);
271                    }
272                }
273                Command::Broadcast(data, scope) => {
274                    self.gossip
275                        .handle(GossipIn::Broadcast(data, scope), now, io)
276                }
277                Command::Quit => self.swarm.handle(SwarmIn::Quit, io),
278            },
279            InEvent::RecvMessage(from, message) => {
280                self.stats.messages_received += 1;
281                match message {
282                    Message::Swarm(message) => {
283                        self.swarm.handle(SwarmIn::RecvMessage(from, message), io)
284                    }
285                    Message::Gossip(message) => {
286                        self.gossip
287                            .handle(GossipIn::RecvMessage(from, message), now, io)
288                    }
289                }
290            }
291            InEvent::TimerExpired(timer) => match timer {
292                Timer::Swarm(timer) => self.swarm.handle(SwarmIn::TimerExpired(timer), io),
293                Timer::Gossip(timer) => self.gossip.handle(GossipIn::TimerExpired(timer), now, io),
294            },
295            InEvent::PeerDisconnected(peer) => {
296                self.swarm.handle(SwarmIn::PeerDisconnected(peer), io);
297                self.gossip.handle(GossipIn::NeighborDown(peer), now, io);
298            }
299            InEvent::UpdatePeerData(data) => self.swarm.handle(SwarmIn::UpdatePeerData(data), io),
300        }
301
302        // Forward NeighborUp and NeighborDown events from hyparview to plumtree
303        let mut io = VecDeque::new();
304        for event in self.outbox.iter() {
305            match event {
306                OutEvent::EmitEvent(Event::NeighborUp(peer)) => {
307                    self.gossip
308                        .handle(GossipIn::NeighborUp(*peer), now, &mut io)
309                }
310                OutEvent::EmitEvent(Event::NeighborDown(peer)) => {
311                    self.gossip
312                        .handle(GossipIn::NeighborDown(*peer), now, &mut io)
313                }
314                _ => {}
315            }
316        }
317        // Note that this is a no-op because plumtree::handle(NeighborUp | NeighborDown)
318        // above does not emit any OutEvents.
319        self.outbox.extend(io.drain(..));
320
321        // Update sent message counter
322        self.stats.messages_sent += self
323            .outbox
324            .iter()
325            .filter(|event| matches!(event, OutEvent::SendMessage(_, _)))
326            .count();
327
328        self.outbox.drain(..)
329    }
