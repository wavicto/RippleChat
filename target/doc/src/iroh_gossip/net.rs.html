67/// Publish and subscribe on gossiping topics.
68///
69/// Each topic is a separate broadcast tree with separate memberships.
70///
71/// A topic has to be joined before you can publish or subscribe on the topic.
72/// To join the swarm for a topic, you have to know the [`PublicKey`] of at least one peer that also joined the topic.
73///
74/// Messages published on the swarm will be delivered to all peers that joined the swarm for that
75/// topic. You will also be relaying (gossiping) messages published by other peers.
76///
77/// With the default settings, the protocol will maintain up to 5 peer connections per topic.
78///
79/// Even though the [`Gossip`] is created from a [`Endpoint`], it does not accept connections
80/// itself. You should run an accept loop on the [`Endpoint`] yourself, check the ALPN protocol of incoming
81/// connections, and if the ALPN protocol equals [`GOSSIP_ALPN`], forward the connection to the
82/// gossip actor through [Self::handle_connection].
83///
84/// The gossip actor will, however, initiate new connections to other peers by itself.
85#[derive(Debug, Clone)]
86pub struct Gossip {
87    pub(crate) inner: Arc<Inner>,
88}
89
90impl std::ops::Deref for Gossip {
91    type Target = GossipApi;
92    fn deref(&self) -> &Self::Target {
93        &self.inner.api
94    }
95}

176    /// Spawn a gossip actor and get a handle for it
177    pub fn spawn(self, endpoint: Endpoint) -> Gossip {
178        let metrics = Arc::new(Metrics::default());
179        let (actor, rpc_tx, local_tx) = Actor::new(endpoint, self.config, metrics.clone());
180        let me = actor.endpoint.node_id().fmt_short();
181        let max_message_size = actor.state.max_message_size();
182
183        let actor_handle = task::spawn(actor.run().instrument(error_span!("gossip", %me)));
184
185        let api = GossipApi::local(rpc_tx);
186
187        Gossip {
188            inner: Inner {
189                api,
190                local_tx,
191                _actor_handle: AbortOnDropHandle::new(actor_handle),
192                max_message_size,
193                metrics,
194            }
195            .into(),
196        }
197    }

250/// Actor that sends and handles messages between the connection and main state loops
251struct Actor {
252    /// Protocol state
253    state: proto::State<PublicKey, StdRng>,
254    /// The endpoint through which we dial peers
255    endpoint: Endpoint,
256    /// Dial machine to connect to peers
257    dialer: Dialer,
258    /// Input messages to the actor
259    rpc_rx: mpsc::Receiver<RpcMessage>,
260    local_rx: mpsc::Receiver<LocalActorMessage>,
261    /// Sender for the state input (cloned into the connection loops)
262    in_event_tx: mpsc::Sender<InEvent>,
263    /// Input events to the state (emitted from the connection loops)
264    in_event_rx: mpsc::Receiver<InEvent>,
265    /// Queued timers
266    timers: Timers<Timer>,
267    /// Map of topics to their state.
268    topics: HashMap<TopicId, TopicState>,
269    /// Map of peers to their state.
270    peers: HashMap<NodeId, PeerState>,
271    /// Stream of commands from topic handles.
272    command_rx: stream_group::Keyed<TopicCommandStream>,
273    /// Internal queue of topic to close because all handles were dropped.
274    quit_queue: VecDeque<TopicId>,
275    /// Tasks for the connection loops, to keep track of panics.
276    connection_tasks: JoinSet<(NodeId, Connection, Result<(), ConnectionLoopError>)>,
277    metrics: Arc<Metrics>,
278    topic_event_forwarders: JoinSet<TopicId>,
279}

318            topic_event_forwarders: Default::default(),

321        (actor, rpc_tx, local_tx)

324    pub async fn run(mut self) {
325        let mut addr_update_stream = self.setup().await;
326
327        let mut i = 0;
328        while self.event_loop(&mut addr_update_stream, i).await {
329            i += 1;
330        }
331    }

333    /// Performs the initial actor setup to run the [`Actor::event_loop`].
334    ///
335    /// This updates our current address and return it. It also returns the home relay stream and
336    /// direct addr stream.
337    async fn setup(&mut self) -> impl Stream<Item = NodeAddr> + Send + Unpin {
338        let addr_update_stream = self.endpoint.node_addr().stream().filter_map(|x| x);
339        // TODO(Frando): Fail if endpoint disconnected?
340        if let Ok(initial_addr) = self.endpoint.node_addr().initialized().await {
341            self.handle_addr_update(initial_addr).await;
342        }
343        addr_update_stream
344    }

346    /// One event loop processing step.
347    ///
348    /// None is returned when no further processing should be performed.
349    async fn event_loop(
350        &mut self,
351        addr_updates: &mut (impl Stream<Item = NodeAddr> + Send + Unpin),
352        i: usize,
353    ) -> bool {
354        self.metrics.actor_tick_main.inc();
355        tokio::select! {
356            biased;
357            conn = self.local_rx.recv() => {
358                match conn {
359                    Some(LocalActorMessage::Shutdown { reply }) => {
360                        debug!("received shutdown message, quit all topics");
361                        self.quit_queue.extend(self.topics.keys().copied());
362                        self.process_quit_queue().await;
363                        debug!("all topics quit, stop gossip actor");
364                        reply.send(()).ok();
365                        return false;
366                    },
367                    Some(LocalActorMessage::HandleConnection(conn)) => {
368                        if let Ok(remote_node_id) = conn.remote_node_id() {
369                            self.handle_connection(remote_node_id, ConnOrigin::Accept, conn);
370                        }
371                    }
372                    None => {
373                        debug!("all gossip handles dropped, stop gossip actor");
374                        return false;
375                    }
376                }
377            }
378            msg = self.rpc_rx.recv() => {
379                trace!(?i, "tick: to_actor_rx");
380                self.metrics.actor_tick_rx.inc();
381                match msg {
382                    Some(msg) => {
383                        self.handle_rpc_msg(msg, Instant::now()).await;
384                    }
385                    None => {
386                        debug!("all gossip handles dropped, stop gossip actor");
387                        return false;
388                    }
389                }
390            },
391            Some((key, (topic, command))) = self.command_rx.next(), if !self.command_rx.is_empty() => {
392                trace!(?i, "tick: command_rx");
393                self.handle_command(topic, key, command).await;
394            },
395            Some(new_address) = addr_updates.next() => {
396                trace!(?i, "tick: new_address");
397                self.metrics.actor_tick_endpoint.inc();
398                self.handle_addr_update(new_address).await;
399            }
400            (peer_id, res) = self.dialer.next_conn() => {
401                trace!(?i, "tick: dialer");
402                self.metrics.actor_tick_dialer.inc();
403                match res {
404                    Some(Ok(conn)) => {
405                        debug!(peer = %peer_id.fmt_short(), "dial successful");
406                        self.metrics.actor_tick_dialer_success.inc();
407                        self.handle_connection(peer_id, ConnOrigin::Dial, conn);
408                    }
409                    Some(Err(err)) => {
410                        warn!(peer = %peer_id.fmt_short(), "dial failed: {err}");
411                        self.metrics.actor_tick_dialer_failure.inc();
412                        let peer_state = self.peers.get(&peer_id);
413                        let is_active = matches!(peer_state, Some(PeerState::Active { .. }));
414                        if !is_active {
415                            self.handle_in_event(InEvent::PeerDisconnected(peer_id), Instant::now())
416                                .await;
417                        }
418                    }
419                    None => {
420                        warn!(peer = %peer_id.fmt_short(), "dial disconnected");
421                        self.metrics.actor_tick_dialer_failure.inc();
422                    }
423                }
424            }
425            event = self.in_event_rx.recv() => {
426                trace!(?i, "tick: in_event_rx");
427                self.metrics.actor_tick_in_event_rx.inc();
428                let event = event.expect("unreachable: in_event_tx is never dropped before receiver");
429                self.handle_in_event(event, Instant::now()).await;
430            }
431            _ = self.timers.wait_next() => {
432                trace!(?i, "tick: timers");
433                self.metrics.actor_tick_timers.inc();
434                let now = Instant::now();
435                while let Some((_instant, timer)) = self.timers.pop_before(now) {
436                    self.handle_in_event(InEvent::TimerExpired(timer), now).await;
437                }
438            }
439            Some(res) = self.connection_tasks.join_next(), if !self.connection_tasks.is_empty() => {
440                trace!(?i, "tick: connection_tasks");
441                let (peer_id, conn, result) = res.expect("connection task panicked");
442                self.handle_connection_task_finished(peer_id, conn, result).await;
443            }
444            Some(res) = self.topic_event_forwarders.join_next(), if !self.topic_event_forwarders.is_empty() => {
445                let topic_id = res.expect("topic event forwarder panicked");
446                if let Some(state) = self.topics.get_mut(&topic_id) {
447                    if !state.still_needed() {
448                        self.quit_queue.push_back(topic_id);
449                        self.process_quit_queue().await;
450                    }
451                }
452            }
453        }
454
455        true
456    }

465    async fn handle_command(
466        &mut self,
467        topic: TopicId,
468        key: stream_group::Key,
469        command: Option<Command>,
470    ) {
471        debug!(?topic, ?key, ?command, "handle command");
472        let Some(state) = self.topics.get_mut(&topic) else {
473            // TODO: unreachable?
474            warn!("received command for unknown topic");
475            return;
476        };
477        match command {
478            Some(command) => {
479                let command = match command {
480                    Command::Broadcast(message) => ProtoCommand::Broadcast(message, Scope::Swarm),
481                    Command::BroadcastNeighbors(message) => {
482                        ProtoCommand::Broadcast(message, Scope::Neighbors)
483                    }
484                    Command::JoinPeers(peers) => ProtoCommand::Join(peers),
485                };
486                self.handle_in_event(proto::InEvent::Command(topic, command), Instant::now())
487                    .await;
488            }
489            None => {
490                state.command_rx_keys.remove(&key);
491                if !state.still_needed() {
492                    self.quit_queue.push_back(topic);
493                    self.process_quit_queue().await;
494                }
495            }
496        }
497    }

499    fn handle_connection(&mut self, peer_id: NodeId, origin: ConnOrigin, conn: Connection) {
500        let (send_tx, send_rx) = mpsc::channel(SEND_QUEUE_CAP);
501        let conn_id = conn.stable_id();
502
503        let queue = match self.peers.entry(peer_id) {
504            Entry::Occupied(mut entry) => entry.get_mut().accept_conn(send_tx, conn_id),
505            Entry::Vacant(entry) => {
506                entry.insert(PeerState::Active {
507                    active_send_tx: send_tx,
508                    active_conn_id: conn_id,
509                    other_conns: Vec::new(),
510                });
511                Vec::new()
512            }
513        };
514
515        let max_message_size = self.state.max_message_size();
516        let in_event_tx = self.in_event_tx.clone();
517
518        // Spawn a task for this connection
519        self.connection_tasks.spawn(
520            async move {
521                let res = connection_loop(
522                    peer_id,
523                    conn.clone(),
524                    origin,
525                    send_rx,
526                    in_event_tx,
527                    max_message_size,
528                    queue,
529                )
530                .await;
531                (peer_id, conn, res)
532            }
533            .instrument(error_span!("conn", peer = %peer_id.fmt_short())),
534        );
535    }

537    #[tracing::instrument(name = "conn", skip_all, fields(peer = %peer_id.fmt_short()))]
538    async fn handle_connection_task_finished(
539        &mut self,
540        peer_id: NodeId,
541        conn: Connection,
542        task_result: Result<(), ConnectionLoopError>,
543    ) {
544        if conn.close_reason().is_none() {
545            conn.close(0u32.into(), b"close from disconnect");
546        }
547        let reason = conn.close_reason().expect("just closed");
548        let error = task_result.err();
549        debug!(%reason, ?error, "connection closed");
550        if let Some(PeerState::Active {
551            active_conn_id,
552            other_conns,
553            ..
554        }) = self.peers.get_mut(&peer_id)
555        {
556            if conn.stable_id() == *active_conn_id {
557                debug!("active send connection closed, mark peer as disconnected");
558                self.handle_in_event(InEvent::PeerDisconnected(peer_id), Instant::now())
559                    .await;
560            } else {
561                other_conns.retain(|x| *x != conn.stable_id());
562                debug!("remaining {} other connections", other_conns.len() + 1);
563            }
564        } else {
565            debug!("peer already marked as disconnected");
566        }
567    }

569    async fn handle_rpc_msg(&mut self, msg: RpcMessage, now: Instant) {
570        trace!("handle to_actor  {msg:?}");
571        match msg {
572            RpcMessage::Join(msg) => {
573                let WithChannels {
574                    inner,
575                    rx,
576                    mut tx,
577                    // TODO(frando): make use of span?
578                    span: _,
579                } = msg;
580                let api::JoinRequest {
581                    topic_id,
582                    bootstrap,
583                } = inner;
584                let TopicState {
585                    neighbors,
586                    event_sender,
587                    command_rx_keys,
588                } = self.topics.entry(topic_id).or_default();
589                let mut sender_dead = false;
590                if !neighbors.is_empty() {
591                    for neighbor in neighbors.iter() {
592                        if let Err(_err) = tx.try_send(Event::NeighborUp(*neighbor)).await {
593                            sender_dead = true;
594                            break;
595                        }
596                    }
597                }
598
599                if !sender_dead {
600                    let fut =
601                        topic_subscriber_loop(tx, event_sender.subscribe()).map(move |_| topic_id);
602                    self.topic_event_forwarders.spawn(fut);
603                }
604                let command_rx = TopicCommandStream::new(topic_id, Box::pin(rx.into_stream()));
605                let key = self.command_rx.insert(command_rx);
606                command_rx_keys.insert(key);
607
608                self.handle_in_event(
609                    InEvent::Command(
610                        topic_id,
611                        ProtoCommand::Join(bootstrap.into_iter().collect()),
612                    ),
613                    now,
614                )
615                .await;
616            }
617        }
618    }

620    async fn handle_in_event(&mut self, event: InEvent, now: Instant) {
621        self.handle_in_event_inner(event, now).await;
622        self.process_quit_queue().await;
623    }
624
625    async fn process_quit_queue(&mut self) {
626        while let Some(topic_id) = self.quit_queue.pop_front() {
627            self.handle_in_event_inner(
628                InEvent::Command(topic_id, ProtoCommand::Quit),
629                Instant::now(),
630            )
631            .await;
632            if self.topics.remove(&topic_id).is_some() {
633                tracing::debug!(%topic_id, "publishers and subscribers gone; unsubscribing");
634            }
635        }
636    }

638    async fn handle_in_event_inner(&mut self, event: InEvent, now: Instant) {
639        if matches!(event, InEvent::TimerExpired(_)) {
640            trace!(?event, "handle in_event");
641        } else {
642            debug!(?event, "handle in_event");
643        };
644        let out = self.state.handle(event, now, Some(&self.metrics));
645        for event in out {
646            if matches!(event, OutEvent::ScheduleTimer(_, _)) {
647                trace!(?event, "handle out_event");
648            } else {
649                debug!(?event, "handle out_event");
650            };
651            match event {
652                OutEvent::SendMessage(peer_id, message) => {
653                    let state = self.peers.entry(peer_id).or_default();
654                    match state {
655                        PeerState::Active { active_send_tx, .. } => {
656                            if let Err(_err) = active_send_tx.send(message).await {
657                                // Removing the peer is handled by the in_event PeerDisconnected sent
658                                // in [`Self::handle_connection_task_finished`].
659                                warn!(
660                                    peer = %peer_id.fmt_short(),
661                                    "failed to send: connection task send loop terminated",
662                                );
663                            }
664                        }
665                        PeerState::Pending { queue } => {
666                            if queue.is_empty() {
667                                debug!(peer = %peer_id.fmt_short(), "start to dial");
668                                self.dialer.queue_dial(peer_id, GOSSIP_ALPN);
669                            }
670                            queue.push(message);
671                        }
672                    }
673                }
674                OutEvent::EmitEvent(topic_id, event) => {
675                    let Some(state) = self.topics.get_mut(&topic_id) else {
676                        // TODO: unreachable?
677                        warn!(?topic_id, "gossip state emitted event for unknown topic");
678                        continue;
679                    };
680                    let TopicState {
681                        neighbors,
682                        event_sender,
683                        ..
684                    } = state;
685                    match &event {
686                        ProtoEvent::NeighborUp(neighbor) => {
687                            neighbors.insert(*neighbor);
688                        }
689                        ProtoEvent::NeighborDown(neighbor) => {
690                            neighbors.remove(neighbor);
691                        }
692                        _ => {}
693                    }
694                    event_sender.send(event).ok();
695                    if !state.still_needed() {
696                        self.quit_queue.push_back(topic_id);
697                    }
698                }
699                OutEvent::ScheduleTimer(delay, timer) => {
700                    self.timers.insert(now + delay, timer);
701                }
702                OutEvent::DisconnectPeer(peer_id) => {
703                    // signal disconnection by dropping the senders to the connection
704                    debug!(peer=%peer_id.fmt_short(), "gossip state indicates disconnect: drop peer");
705                    self.peers.remove(&peer_id);
706                }
707                OutEvent::PeerData(node_id, data) => match decode_peer_data(&data) {
708                    Err(err) => warn!("Failed to decode {data:?} from {node_id}: {err}"),
709                    Ok(info) => {
710                        debug!(peer = ?node_id, "add known addrs: {info:?}");
711                        let node_addr = NodeAddr {
712                            node_id,
713                            relay_url: info.relay_url,
714                            direct_addresses: info.direct_addresses,
715                        };
716                        if let Err(err) = self
717                            .endpoint
718                            .add_node_addr_with_source(node_addr, SOURCE_NAME)
719                        {
720                            debug!(peer = ?node_id, "add known failed: {err:?}");
721                        }
722                    }
723                },
724            }
725        }
726    }
727}

731#[derive(Debug)]
732enum PeerState {
733    Pending {
734        queue: Vec<ProtoMessage>,
735    },
736    Active {
737        active_send_tx: mpsc::Sender<ProtoMessage>,
738        active_conn_id: ConnId,
739        other_conns: Vec<ConnId>,
740    },
741}
742
743impl PeerState {
744    fn accept_conn(
745        &mut self,
746        send_tx: mpsc::Sender<ProtoMessage>,
747        conn_id: ConnId,
748    ) -> Vec<ProtoMessage> {
749        match self {
750            PeerState::Pending { queue } => {
751                let queue = std::mem::take(queue);
752                *self = PeerState::Active {
753                    active_send_tx: send_tx,
754                    active_conn_id: conn_id,
755                    other_conns: Vec::new(),
756                };
757                queue
758            }
759            PeerState::Active {
760                active_send_tx,
761                active_conn_id,
762                other_conns,
763            } => {
764                // We already have an active connection. We keep the old connection intact,
765                // but only use the new connection for sending from now on.
766                // By dropping the `send_tx` of the old connection, the send loop part of
767                // the `connection_loop` of the old connection will terminate, which will also
768                // notify the peer that the old connection may be dropped.
769                other_conns.push(*active_conn_id);
770                *active_send_tx = send_tx;
771                *active_conn_id = conn_id;
772                Vec::new()
773            }
774        }
775    }
776}
777
778impl Default for PeerState {
779    fn default() -> Self {
780        PeerState::Pending { queue: Vec::new() }
781    }

853async fn connection_loop(
854    from: PublicKey,
855    conn: Connection,
856    origin: ConnOrigin,
857    send_rx: mpsc::Receiver<ProtoMessage>,
858    in_event_tx: mpsc::Sender<InEvent>,
859    max_message_size: usize,
860    queue: Vec<ProtoMessage>,
861) -> Result<(), ConnectionLoopError> {
862    debug!(?origin, "connection established");
863
864    let mut send_loop = SendLoop::new(conn.clone(), send_rx, max_message_size);
865    let mut recv_loop = RecvLoop::new(from, conn, in_event_tx, max_message_size);
866
867    let send_fut = send_loop.run(queue).instrument(error_span!("send"));
868    let recv_fut = recv_loop.run().instrument(error_span!("recv"));
869
870    let (send_res, recv_res) = tokio::join!(send_fut, recv_fut);
871    send_res?;
872    recv_res?;
873    Ok(())
874}

911async fn topic_subscriber_loop(
912    sender: irpc::channel::mpsc::Sender<Event>,
913    mut topic_events: broadcast::Receiver<ProtoEvent>,
914) {
915    loop {
916        tokio::select! {
917           biased;
918           msg = topic_events.recv() => {
919               let event = match msg {
920                   Err(broadcast::error::RecvError::Closed) => break,
921                   Err(broadcast::error::RecvError::Lagged(_)) => Event::Lagged,
922                   Ok(event) => event.into(),
923               };
924               if sender.send(event).await.is_err() {
925                   break;
926               }
927           }
928           _ = sender.closed() => break,
929        }
930    }

954impl Stream for TopicCommandStream {
955    type Item = (TopicId, Option<Command>);
956    fn poll_next(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
957        if self.closed {
958            return Poll::Ready(None);
959        }
960        match Pin::new(&mut self.stream).poll_next(cx) {
961            Poll::Ready(Some(Ok(item))) => Poll::Ready(Some((self.topic_id, Some(item)))),
962            Poll::Ready(None) | Poll::Ready(Some(Err(_))) => {
963                self.closed = true;
964                Poll::Ready(Some((self.topic_id, None)))
965            }
966            Poll::Pending => Poll::Pending,
967        }
968    }

971#[derive(Debug)]
972struct Dialer {
973    endpoint: Endpoint,
974    pending: JoinSet<(
975        NodeId,
976        Option<Result<Connection, iroh::endpoint::ConnectError>>,
977    )>,
978    pending_dials: HashMap<NodeId, CancellationToken>,
979}
980
981impl Dialer {
982    /// Create a new dialer for a [`Endpoint`]
983    fn new(endpoint: Endpoint) -> Self {
984        Self {
985            endpoint,
986            pending: Default::default(),
987            pending_dials: Default::default(),
988        }
989    }
990
991    /// Starts to dial a node by [`NodeId`].
992    fn queue_dial(&mut self, node_id: NodeId, alpn: &'static [u8]) {
993        if self.is_pending(node_id) {
994            return;
995        }
996        let cancel = CancellationToken::new();
997        self.pending_dials.insert(node_id, cancel.clone());
998        let endpoint = self.endpoint.clone();
999        self.pending.spawn(async move {
1000            let res = tokio::select! {
1001                biased;
1002                _ = cancel.cancelled() => None,
1003                res = endpoint.connect(node_id, alpn) => Some(res),
1004            };
1005            (node_id, res)
1006        });
1007    }
1008
1009    /// Checks if a node is currently being dialed.
1010    fn is_pending(&self, node: NodeId) -> bool {
1011        self.pending_dials.contains_key(&node)
1012    }
1013
1014    /// Waits for the next dial operation to complete.
1015    /// `None` means disconnected
1016    async fn next_conn(
1017        &mut self,
1018    ) -> (
1019        NodeId,
1020        Option<Result<Connection, iroh::endpoint::ConnectError>>,
1021    ) {
1022        match self.pending_dials.is_empty() {
1023            false => {
1024                let (node_id, res) = loop {
1025                    match self.pending.join_next().await {
1026                        Some(Ok((node_id, res))) => {
1027                            self.pending_dials.remove(&node_id);
1028                            break (node_id, res);
1029                        }
1030                        Some(Err(e)) => {
1031                            error!("next conn error: {:?}", e);
1032                        }
1033                        None => {
1034                            error!("no more pending conns available");
1035                            std::future::pending().await
1036                        }
1037                    }
1038                };
1039
1040                (node_id, res)
1041            }
1042            true => std::future::pending().await,
1043        }
1044    }
1045}

1203        endpoint: Endpoint,
1204        gossip: Gossip,
1205        cancel: CancellationToken,
1206    ) -> Result<()> {
1207        loop {
1208            tokio::select! {
1209                biased;
1210                _ = cancel.cancelled() => break,
1211                incoming = endpoint.accept() => match incoming {
1212                    None => break,
1213                    Some(incoming) => {
1214                        let connecting = match incoming.accept() {
1215                            Ok(connecting) => connecting,
1216                            Err(err) => {
1217                                warn!("incoming connection failed: {err:#}");

1219                                // this can be caused by retransmitted datagrams
1220                                continue;
1221                            }
1222                        };
1223                        gossip.handle_connection(connecting.await.e()?).await?
1224                    }
1225                }
1226            }
1227        }
1228        Ok(())
1229    }

1568    #[tokio::test]
1569    #[traced_test]
1570    async fn can_die_and_reconnect() -> Result {
1571        /// Runs a future in a separate runtime on a separate thread, cancelling everything
1572        /// abruptly once `cancel` is invoked.
1573        fn run_in_thread<T: Send + 'static>(
1574            cancel: CancellationToken,
1575            fut: impl std::future::Future<Output = T> + Send + 'static,
1576        ) -> std::thread::JoinHandle<Option<T>> {
1577            std::thread::spawn(move || {
1578                let rt = tokio::runtime::Builder::new_current_thread()
1579                    .enable_all()
1580                    .build()
1581                    .unwrap();
1582                rt.block_on(async move { cancel.run_until_cancelled(fut).await })
1583            })
1584        }
1585
1586        /// Spawns a new endpoint and gossip instance.
1587        async fn spawn_gossip(
1588            secret_key: SecretKey,
1589            relay_map: RelayMap,
1590        ) -> Result<(Router, Gossip), BindError> {
1591            let ep = Endpoint::builder()
1592                .secret_key(secret_key)
1593                .relay_mode(RelayMode::Custom(relay_map))
1594                .insecure_skip_relay_cert_verify(true)
1595                .bind()
1596                .await?;
1597            let gossip = Gossip::builder().spawn(ep.clone());
1598            let router = Router::builder(ep)
1599                .accept(GOSSIP_ALPN, gossip.clone())
1600                .spawn();
1601            Ok((router, gossip))
1602        }
1603
1604        /// Spawns a gossip node, and broadcasts a single message, then sleep until cancelled externally.
1605        async fn broadcast_once(
1606            secret_key: SecretKey,
1607            relay_map: RelayMap,
1608            bootstrap_addr: NodeAddr,
1609            topic_id: TopicId,
1610            message: String,
1611        ) -> Result {
1612            let (router, gossip) = spawn_gossip(secret_key, relay_map).await?;
1613            info!(node_id = %router.endpoint().node_id().fmt_short(), "broadcast node spawned");
1614            let bootstrap = vec![bootstrap_addr.node_id];
1615            router.endpoint().add_node_addr(bootstrap_addr)?;
1616            let mut topic = gossip.subscribe_and_join(topic_id, bootstrap).await?;
1617            topic.broadcast(message.as_bytes().to_vec().into()).await?;
1618            std::future::pending::<()>().await;
1619            Ok(())
1620        }
1621
1622        let (relay_map, _relay_url, _guard) = iroh::test_utils::run_relay_server().await.unwrap();
1623        let mut rng = &mut rand_chacha::ChaCha12Rng::seed_from_u64(1);
1624        let topic_id = TopicId::from_bytes(rng.gen());
1625
1626        // spawn a gossip node, send the node's address on addr_tx,
1627        // then wait to receive `count` messages, and terminate.
1628        let (addr_tx, addr_rx) = tokio::sync::oneshot::channel();
1629        let (msgs_recv_tx, mut msgs_recv_rx) = tokio::sync::mpsc::channel(3);
1630        let recv_task = tokio::task::spawn({
1631            let relay_map = relay_map.clone();
1632            let secret_key = SecretKey::generate(&mut rng);
1633            async move {
1634                let (router, gossip) = spawn_gossip(secret_key, relay_map).await?;
1635                // wait for the relay to be set. iroh currently has issues when trying
1636                // to immediately reconnect with changed direct addresses, but when the
1637                // relay path is available it works.
1638                // See https://github.com/n0-computer/iroh/pull/3372
1639                router.endpoint().home_relay().initialized().await?;
1640                let addr = router.endpoint().node_addr().initialized().await?;
1641                info!(node_id = %addr.node_id.fmt_short(), "recv node spawned");
1642                addr_tx.send(addr).unwrap();
1643                let mut topic = gossip.subscribe_and_join(topic_id, vec![]).await?;
1644                while let Some(event) = topic.try_next().await.unwrap() {
1645                    if let Event::Received(message) = event {
1646                        let message = std::str::from_utf8(&message.content).e()?.to_string();
1647                        msgs_recv_tx.send(message).await.e()?;
1648                    }
1649                }
1650                Result::<_, n0_snafu::Error>::Ok(())
1651            }
1652        });
1653
1654        let node0_addr = addr_rx.await.e()?;
1655        let max_wait = Duration::from_secs(5);
1656
1657        // spawn a node, send a message, and then abruptly terminate the node ungracefully
1658        // after the message was received on our receiver node.
1659        let cancel = CancellationToken::new();
1660        let secret = SecretKey::generate(&mut rng);
1661        let join_handle_1 = run_in_thread(
1662            cancel.clone(),
1663            broadcast_once(
1664                secret.clone(),
1665                relay_map.clone(),
1666                node0_addr.clone(),
1667                topic_id,
1668                "msg1".to_string(),
1669            ),
1670        );
1671        // assert that we received the message on the receiver node.
1672        let msg = timeout(max_wait, msgs_recv_rx.recv()).await.e()?.unwrap();
1673        assert_eq!(&msg, "msg1");
1674        info!("kill broadcast node");
1675        cancel.cancel();
1676
1677        // spawns the node again with the same node id, and send another message
1678        let cancel = CancellationToken::new();
1679        let join_handle_2 = run_in_thread(
1680            cancel.clone(),
1681            broadcast_once(
1682                secret.clone(),
1683                relay_map.clone(),
1684                node0_addr.clone(),
1685                topic_id,
1686                "msg2".to_string(),
1687            ),
1688        );
1689        // assert that we received the message on the receiver node.
1690        // this means that the reconnect with the same node id worked.
1691        let msg = timeout(max_wait, msgs_recv_rx.recv()).await.e()?.unwrap();
1692        assert_eq!(&msg, "msg2");
1693        info!("kill broadcast node");
1694        cancel.cancel();
1695
1696        info!("kill recv node");
1697        recv_task.abort();
1698        assert!(join_handle_1.join().unwrap().is_none());
1699        assert!(join_handle_2.join().unwrap().is_none());
1700
1701        Ok(())
1702    }
