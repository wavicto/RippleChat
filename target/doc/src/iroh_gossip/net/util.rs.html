57#[derive(Debug, Serialize, Deserialize)]
58pub(crate) struct StreamHeader {
59    pub(crate) topic_id: TopicId,
60}
61
62impl StreamHeader {
63    pub(crate) async fn read(
64        stream: &mut RecvStream,
65        buffer: &mut BytesMut,
66        max_message_size: usize,
67    ) -> Result<Self, ReadError> {
68        let header: Self = read_frame(stream, buffer, max_message_size)
69            .await?
70            .ok_or_else(|| {
71                ReadError::from(io::Error::new(
72                    io::ErrorKind::UnexpectedEof,
73                    "stream ended before header",
74                ))
75            })?;
76        Ok(header)
77    }
78
79    pub(crate) async fn write(
80        self,
81        stream: &mut SendStream,
82        buffer: &mut Vec<u8>,
83        max_message_size: usize,
84    ) -> Result<(), WriteError> {
85        write_frame(stream, &self, buffer, max_message_size).await?;
86        Ok(())
87    }
88}

90pub(crate) struct RecvLoop {
91    remote_node_id: NodeId,
92    conn: Connection,
93    max_message_size: usize,
94    in_event_tx: mpsc::Sender<InEvent>,
95}
96
97impl RecvLoop {
98    pub(crate) fn new(
99        remote_node_id: NodeId,
100        conn: Connection,
101        in_event_tx: mpsc::Sender<InEvent>,
102        max_message_size: usize,
103    ) -> Self {
104        Self {
105            remote_node_id,
106            conn,
107            max_message_size,
108            in_event_tx,
109        }
110    }
111
112    pub(crate) async fn run(&mut self) -> Result<(), ReadError> {
113        let mut read_futures = FuturesUnordered::new();
114        let mut conn_is_closed = false;
115        let closed = self.conn.closed();
116        tokio::pin!(closed);
117        while !conn_is_closed || !read_futures.is_empty() {
118            tokio::select! {
119                _ = &mut closed, if !conn_is_closed => {
120                    conn_is_closed = true;
121                }
122                stream = self.conn.accept_uni(), if !conn_is_closed => {
123                    let stream = match stream {
124                        Ok(stream) => stream,
125                        Err(_) => {
126                            conn_is_closed = true;
127                            continue;
128                        }
129                    };
130                    let state = RecvStreamState::new(stream, self.max_message_size).await?;
131                    debug!(topic=%state.header.topic_id.fmt_short(), "stream opened");
132                    read_futures.push(state.next());
133                }
134                Some(res) = read_futures.next(), if !read_futures.is_empty() => {
135                    let (state, msg) = match res {
136                        Ok((state, msg)) => (state, msg),
137                        Err(err) => {
138                            debug!("recv stream closed with error: {err:#}");
139                            continue;
140                        }
141                    };
142                    match msg {
143                        None => debug!(topic=%state.header.topic_id.fmt_short(), "stream closed"),
144                        Some(msg) => {
145                            if self.in_event_tx.send(InEvent::RecvMessage(self.remote_node_id, msg)).await.is_err() {
146                                debug!("stop recv loop: actor closed");
147                                break;
148                            }
149                            read_futures.push(state.next());
150                        }
151                    }
152                }
153            }
154        }
155        debug!("recv loop closed");
156        Ok(())
157    }
158}

160#[derive(Debug)]
161struct RecvStreamState {
162    stream: RecvStream,
163    header: StreamHeader,
164    buffer: BytesMut,
165    max_message_size: usize,
166}
167
168impl RecvStreamState {
169    async fn new(mut stream: RecvStream, max_message_size: usize) -> Result<Self, ReadError> {
170        let mut buffer = BytesMut::new();
171        let header = StreamHeader::read(&mut stream, &mut buffer, max_message_size).await?;
172        Ok(Self {
173            buffer: BytesMut::new(),
174            max_message_size,
175            stream,
176            header,
177        })
178    }
179
180    /// Reads the next message from the stream.
181    ///
182    /// Returns `self` and the next message, or `None` if the stream ended gracefully.
183    ///
184    /// ## Cancellation safety
185    ///
186    /// This function is not cancellation-safe.
187    async fn next(mut self) -> Result<(Self, Option<ProtoMessage>), ReadError> {
188        let msg = read_frame(&mut self.stream, &mut self.buffer, self.max_message_size).await?;
189        let msg = msg.map(|msg| ProtoMessage {
190            topic: self.header.topic_id,
191            message: msg,
192        });
193        Ok((self, msg))
194    }
195}

197pub(crate) struct SendLoop {
198    conn: Connection,
199    streams: HashMap<TopicId, SendStream>,
200    buffer: Vec<u8>,
201    max_message_size: usize,
202    finishing: JoinSet<()>,
203    send_rx: mpsc::Receiver<ProtoMessage>,
204}
205
206impl SendLoop {
207    pub(crate) fn new(
208        conn: Connection,
209        send_rx: mpsc::Receiver<ProtoMessage>,
210        max_message_size: usize,
211    ) -> Self {
212        Self {
213            conn,
214            max_message_size,
215            buffer: Default::default(),
216            streams: Default::default(),
217            finishing: Default::default(),
218            send_rx,
219        }
220    }
221
222    pub(crate) async fn run(&mut self, queue: Vec<ProtoMessage>) -> Result<(), WriteError> {
223        for msg in queue {
224            self.write_message(&msg).await?;
225        }
226        let conn_clone = self.conn.clone();
227        let closed = conn_clone.closed();
228        tokio::pin!(closed);
229        loop {
230            tokio::select! {
231                biased;
232                _ = &mut closed => break,
233                Some(msg) = self.send_rx.recv() => self.write_message(&msg).await?,
234                _ = self.finishing.join_next(), if !self.finishing.is_empty() => {}
235                else => break,
236            }
237        }
238
239        // Close remaining streams.
240        for (topic_id, mut stream) in self.streams.drain() {
241            stream.finish().ok();
242            self.finishing.spawn(
243                async move {
244                    stream.stopped().await.ok();
245                    debug!(topic=%topic_id.fmt_short(), "stream closed");
246                }
247                .instrument(tracing::Span::current()),
248            );
249        }
250        if !self.finishing.is_empty() {
251            trace!(
252                "send loop closing, waiting for {} send streams to finish",
253                self.finishing.len()
254            );
255            // Wait for the remote to acknowledge all streams are finished.
256            if let Err(_elapsed) = n0_future::time::timeout(Duration::from_secs(5), async move {
257                while self.finishing.join_next().await.is_some() {}
258            })
259            .await
260            {
261                debug!("not all send streams finished within timeout, abort")
262            }
263        }
264        debug!("send loop closed");
265        Ok(())
266    }
267
268    /// Write a [`ProtoMessage`] as a length-prefixed, postcard-encoded message on its stream.
269    ///
270    /// If no stream is opened yet, this opens a new stream for the topic and writes the topic header.
271    ///
272    /// This function is not cancellation-safe.
273    pub async fn write_message(&mut self, message: &ProtoMessage) -> Result<(), WriteError> {
274        let ProtoMessage { topic, message } = message;
275        let topic_id = *topic;
276        let is_last = message.is_disconnect();
277
278        let mut entry = match self.streams.entry(topic_id) {
279            hash_map::Entry::Occupied(entry) => entry,
280            hash_map::Entry::Vacant(entry) => {
281                let mut stream = self.conn.open_uni().await?;
282                let header = StreamHeader { topic_id };
283                header
284                    .write(&mut stream, &mut self.buffer, self.max_message_size)
285                    .await?;
286                debug!(topic=%topic_id.fmt_short(), "stream opened");
287                entry.insert_entry(stream)
288            }
289        };
290        let stream = entry.get_mut();
291
292        write_frame(stream, message, &mut self.buffer, self.max_message_size).await?;
293
294        if is_last {
295            trace!(topic=%topic_id.fmt_short(), "stream closing");
296            let mut stream = entry.remove();
297            if stream.finish().is_ok() {
298                self.finishing.spawn(
299                    async move {
300                        stream.stopped().await.ok();
301                        debug!(topic=%topic_id.fmt_short(), "stream closed");
302                    }
303                    .instrument(tracing::Span::current()),
304                );
305            }
306        }
307
308        Ok(())
309    }
310}

312/// Errors related to message reading
313#[allow(missing_docs)]
314#[common_fields({
315    backtrace: Option<snafu::Backtrace>,
316})]
317#[derive(Debug, Snafu)]
318#[snafu(module)]
319#[non_exhaustive]
320pub(crate) enum ReadError {
321    /// Deserialization failed
322    #[snafu(transparent)]
323    De { source: postcard::Error },
324    /// IO error
325    #[snafu(transparent)]
326    Io { source: std::io::Error },
327    /// Message was larger than the configured maximum message size
328    #[snafu(display("message too large"))]
329    TooLarge {},
330}

332/// Read a length-prefixed frame and decode with postcard.
333pub async fn read_frame<T: DeserializeOwned>(
334    reader: &mut RecvStream,
335    buffer: &mut BytesMut,
336    max_message_size: usize,
337) -> Result<Option<T>, ReadError> {
338    match read_lp(reader, buffer, max_message_size).await? {
339        None => Ok(None),
340        Some(data) => {
341            let message = postcard::from_bytes(&data)?;
342            Ok(Some(message))
343        }
344    }
345}
346
347/// Reads a length prefixed buffer.
348///
349/// Returns the frame as raw bytes.  If the end of the stream is reached before
350/// the frame length starts, `None` is returned.
351pub async fn read_lp(
352    reader: &mut RecvStream,
353    buffer: &mut BytesMut,
354    max_message_size: usize,
355) -> Result<Option<Bytes>, ReadError> {
356    let size = match reader.read_u32().await {
357        Ok(size) => size,
358        Err(err) if err.kind() == io::ErrorKind::UnexpectedEof => return Ok(None),
359        Err(err) => return Err(err.into()),
360    };
361    let size = usize::try_from(size).map_err(|_| read_error::TooLargeSnafu.build())?;
362    if size > max_message_size {
363        return Err(read_error::TooLargeSnafu.build());
364    }
365    buffer.resize(size, 0u8);
366    reader
367        .read_exact(&mut buffer[..])
368        .await
369        .map_err(io::Error::other)?;
370    Ok(Some(buffer.split_to(size).freeze()))
371}
372
373/// Writes a length-prefixed frame.
374pub async fn write_frame<T: Serialize>(
375    stream: &mut SendStream,
376    message: &T,
377    buffer: &mut Vec<u8>,
378    max_message_size: usize,
379) -> Result<(), WriteError> {
380    let len = postcard::experimental::serialized_size(&message)?;
381    if len >= max_message_size {
382        return Err(write_error::TooLargeSnafu.build());
383    }
384    buffer.clear();
385    buffer.resize(len, 0u8);
386    let slice = postcard::to_slice(&message, buffer)?;
387    stream.write_u32(len as u32).await?;
388    stream.write_all(slice).await.map_err(io::Error::other)?;
389    Ok(())
390}

392/// A [`TimerMap`] with an async method to wait for the next timer expiration.
393#[derive(Debug)]
394pub struct Timers<T> {
395    next: Option<(Instant, Pin<Box<Sleep>>)>,
396    map: TimerMap<T>,
397}
398
399impl<T> Default for Timers<T> {
400    fn default() -> Self {
401        Self {
402            next: None,
403            map: TimerMap::default(),
404        }
405    }
406}
407
408impl<T> Timers<T> {
409    /// Creates a new timer map.
410    pub fn new() -> Self {
411        Self::default()
412    }
413
414    /// Inserts a new entry at the specified instant
415    pub fn insert(&mut self, instant: Instant, item: T) {
416        self.map.insert(instant, item);
417    }
418
419    fn reset(&mut self) {
420        self.next = self
421            .map
422            .first()
423            .map(|instant| (*instant, Box::pin(sleep_until(*instant))))
424    }
425
426    /// Waits for the next timer to elapse.
427    pub async fn wait_next(&mut self) -> Instant {
428        self.reset();
429        match self.next.as_mut() {
430            Some((instant, sleep)) => {
431                sleep.await;
432                *instant
433            }
434            None => std::future::pending().await,
435        }
436    }
437
438    /// Pops the earliest timer that expires at or before `now`.
439    pub fn pop_before(&mut self, now: Instant) -> Option<(Instant, T)> {
440        self.map.pop_before(now)
441    }
