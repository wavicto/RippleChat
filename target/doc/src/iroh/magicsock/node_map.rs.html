35/// Map of the [`NodeState`] information for all the known nodes.
36///
37/// The nodes can be looked up by:
38///
39/// - The node's ID in this map, only useful if you know the ID from an insert or lookup.
40///   This is static and never changes.
41///
42/// - The [`NodeIdMappedAddr`] which internally identifies the node to the QUIC stack.  This
43///   is static and never changes.
44///
45/// - The nodes's public key, aka `PublicKey` or "node_key".  This is static and never changes,
46///   however a node could be added when this is not yet known.
47///
48/// - A public socket address on which they are reachable on the internet, known as ip-port.
49///   These come and go as the node moves around on the internet
50///
51/// An index of nodeInfos by node key, NodeIdMappedAddr, and discovered ip:port endpoints.
52#[derive(Debug, Default)]
53pub(super) struct NodeMap {
54    inner: Mutex<NodeMapInner>,
55}

80/// The origin or *source* through which an address associated with a remote node
81/// was discovered.
82///
83/// An aggregate of the [`Source`]s of all the addresses of a node describe the
84/// [`Source`]s of the node itself.
85///
86/// A [`Source`] helps track how and where an address was learned. Multiple
87/// sources can be associated with a single address, if we have discovered this
88/// address through multiple means.
89///
90/// Each time a [`NodeAddr`] is added to the node map, usually through
91/// [`crate::endpoint::Endpoint::add_node_addr_with_source`], a [`Source`] must be supplied to indicate
92/// how the address was obtained.
93///
94/// A [`Source`] can describe a variety of places that an address or node was
95/// discovered, such as a configured discovery service, the network itself
96/// (if another node has reached out to us), or as a user supplied [`NodeAddr`].
97
98#[derive(Serialize, Deserialize, strum::Display, Debug, Clone, Eq, PartialEq, Hash)]
99#[strum(serialize_all = "kebab-case")]
100pub enum Source {
101    /// Address was loaded from the fs.
102    Saved,
103    /// A node communicated with us first via UDP.
104    Udp,
105    /// A node communicated with us first via relay.
106    Relay,
107    /// Application layer added the address directly.
108    App,
109    /// The address was discovered by a discovery service.
110    #[strum(serialize = "{name}")]
111    Discovery {
112        /// The name of the discovery service that discovered the address.
113        name: String,
114    },
115    /// Application layer with a specific name added the node directly.
116    #[strum(serialize = "{name}")]
117    NamedApp {
118        /// The name of the application that added the node
119        name: String,
120    },
121}

387    /// Removes a direct address from a node.
388    fn remove_by_ipp(&mut self, ipp: IpPort, reason: ClearReason) {
389        if let Some(id) = self.by_ip_port.remove(&ipp) {
390            if let Entry::Occupied(mut entry) = self.by_id.entry(id) {
391                let node = entry.get_mut();
392                node.remove_direct_addr(&ipp, reason);
393                if node.direct_addresses().count() == 0 {
394                    let node_id = node.public_key();
395                    let mapped_addr = node.quic_mapped_addr();
396                    self.by_node_key.remove(node_id);
397                    self.by_quic_mapped_addr.remove(mapped_addr);
398                    debug!(node_id=%node_id.fmt_short(), ?reason, "removing node");
399                    entry.remove();
400                }
401            }
402        }
403    }

543    fn handle_ping(&mut self, sender: NodeId, src: SendAddr, tx_id: TransactionId) -> PingHandled {
544        #[cfg(any(test, feature = "test-utils"))]
545        let path_selection = self.path_selection;
546        let node_state = self.get_or_insert_with(NodeStateKey::NodeId(sender), || {
547            debug!("received ping: node unknown, add to node map");
548            let source = if src.is_relay() {
549                Source::Relay
550            } else {
551                Source::Udp
552            };
553            Options {
554                node_id: sender,
555                relay_url: src.relay_url(),
556                active: true,
557                source,
558                #[cfg(any(test, feature = "test-utils"))]
559                path_selection,
560            }
561        });
562
563        let handled = node_state.handle_ping(src.clone(), tx_id);
564        if let SendAddr::Udp(ref addr) = src {
565            if matches!(handled.role, PingRole::NewPath) {
566                self.set_node_key_for_ip_port(*addr, &sender);
567            }
568        }
569        handled

572    /// Inserts a new node into the [`NodeMap`].
573    fn insert_node(&mut self, options: Options) -> &mut NodeState {
574        info!(
575            node = %options.node_id.fmt_short(),
576            relay_url = ?options.relay_url,
577            source = %options.source,
578            "inserting new node in NodeMap",
579        );
580        let id = self.next_id;
581        self.next_id = self.next_id.wrapping_add(1);
582        let node_state = NodeState::new(id, options);
583
584        // update indices
585        self.by_quic_mapped_addr
586            .insert(*node_state.quic_mapped_addr(), id);
587        self.by_node_key.insert(*node_state.public_key(), id);
588
589        self.by_id.insert(id, node_state);
590        self.by_id.get_mut(&id).expect("just inserted")
591    }

593    /// Makes future node lookups by ipp return the same endpoint as a lookup by nk.
594    ///
595    /// This should only be called with a fully verified mapping of ipp to
596    /// nk, because calling this function defines the endpoint we hand to
597    /// WireGuard for packets received from ipp.
598    fn set_node_key_for_ip_port(&mut self, ipp: impl Into<IpPort>, nk: &PublicKey) {
599        let ipp = ipp.into();
600        if let Some(id) = self.by_ip_port.get(&ipp) {
601            if !self.by_node_key.contains_key(nk) {
602                self.by_node_key.insert(*nk, *id);
603            }
604            self.by_ip_port.remove(&ipp);
605        }
606        if let Some(id) = self.by_node_key.get(nk) {
607            trace!("insert ip -> id: {:?} -> {}", ipp, id);
608            self.by_ip_port.insert(ipp, *id);
609        }
610    }

620        let now = Instant::now();
621        let mut prune_candidates: Vec<_> = self
622            .by_id
623            .values()
624            .filter(|node| !node.is_active(&now))
625            .map(|node| (*node.public_key(), node.last_used()))
626            .collect();
627
628        let prune_count = prune_candidates.len().saturating_sub(MAX_INACTIVE_NODES);
629        if prune_count == 0 {
630            // within limits
631            return;
632        }
633
634        prune_candidates.sort_unstable_by_key(|(_pk, last_used)| *last_used);
635        prune_candidates.truncate(prune_count);
636        for (public_key, last_used) in prune_candidates.into_iter() {
637            let node = public_key.fmt_short();
638            match last_used.map(|instant| instant.elapsed()) {
639                Some(last_used) => trace!(%node, ?last_used, "pruning inactive"),
640                None => trace!(%node, last_used=%"never", "pruning inactive"),
641            }
642
643            let Some(id) = self.by_node_key.remove(&public_key) else {
644                debug_assert!(false, "missing by_node_key entry for pk in by_id");
645                continue;
646            };
647
648            let Some(ep) = self.by_id.remove(&id) else {
649                debug_assert!(false, "missing by_id entry for id in by_node_key");
650                continue;
651            };
652
653            for ip_port in ep.direct_addresses() {
654                self.by_ip_port.remove(&ip_port);
655            }
656
657            self.by_quic_mapped_addr.remove(ep.quic_mapped_addr());
658        }
659    }
660}

662/// An (Ip, Port) pair.
663///
664/// NOTE: storing an [`IpPort`] is safer than storing a [`SocketAddr`] because for IPv6 socket
665/// addresses include fields that can't be assumed consistent even within a single connection.
666#[derive(Debug, derive_more::Display, Clone, Copy, Hash, PartialEq, Eq, PartialOrd, Ord)]
667#[display("{}", SocketAddr::from(*self))]
668pub struct IpPort {
669    ip: IpAddr,
670    port: u16,
671}
672
673impl From<SocketAddr> for IpPort {
674    fn from(socket_addr: SocketAddr) -> Self {
675        Self {
676            ip: socket_addr.ip(),
677            port: socket_addr.port(),
678        }
679    }
680}
681
682impl From<IpPort> for SocketAddr {
683    fn from(ip_port: IpPort) -> Self {
684        let IpPort { ip, port } = ip_port;
685        (ip, port).into()
686    }
687}
688
689impl IpPort {
690    pub fn ip(&self) -> &IpAddr {
691        &self.ip
692    }
693
694    pub fn port(&self) -> u16 {
695        self.port
696    }
697}

788    #[test]
789    #[traced_test]
790    fn test_prune_direct_addresses() {
791        let node_map = NodeMap::default();
792        let public_key = SecretKey::generate(rand::thread_rng()).public();
793        let id = node_map
794            .inner
795            .lock()
796            .unwrap()
797            .insert_node(Options {
798                node_id: public_key,
799                relay_url: None,
800                active: false,
801                source: Source::NamedApp {
802                    name: "test".into(),
803                },
804                path_selection: PathSelection::default(),
805            })
806            .id();
807
808        const LOCALHOST: IpAddr = IpAddr::V4(std::net::Ipv4Addr::LOCALHOST);
809
810        // add [`MAX_INACTIVE_DIRECT_ADDRESSES`] active direct addresses and double
811        // [`MAX_INACTIVE_DIRECT_ADDRESSES`] that are inactive
812
813        info!("Adding active addresses");
814        for i in 0..MAX_INACTIVE_DIRECT_ADDRESSES {
815            let addr = SocketAddr::new(LOCALHOST, 5000 + i as u16);
816            let node_addr = NodeAddr::new(public_key).with_direct_addresses([addr]);
817            // add address
818            node_map.add_test_addr(node_addr);
819            // make it active
820            node_map.inner.lock().unwrap().receive_udp(addr);
821        }
822
823        info!("Adding offline/inactive addresses");
824        for i in 0..MAX_INACTIVE_DIRECT_ADDRESSES * 2 {
825            let addr = SocketAddr::new(LOCALHOST, 6000 + i as u16);
826            let node_addr = NodeAddr::new(public_key).with_direct_addresses([addr]);
827            node_map.add_test_addr(node_addr);
828        }
829
830        let mut node_map_inner = node_map.inner.lock().unwrap();
831        let endpoint = node_map_inner.by_id.get_mut(&id).unwrap();
832
833        info!("Adding alive addresses");
834        for i in 0..MAX_INACTIVE_DIRECT_ADDRESSES {
835            let addr = SendAddr::Udp(SocketAddr::new(LOCALHOST, 7000 + i as u16));
836            let txid = stun_rs::TransactionId::from([i as u8; 12]);
837            // Note that this already invokes .prune_direct_addresses() because these are
838            // new UDP paths.
839            endpoint.handle_ping(addr, txid);
840        }
841
842        info!("Pruning addresses");
843        endpoint.prune_direct_addresses();
844
845        // Half the offline addresses should have been pruned.  All the active and alive
846        // addresses should have been kept.
847        assert_eq!(
848            endpoint.direct_addresses().count(),
849            MAX_INACTIVE_DIRECT_ADDRESSES * 3
850        );
851
852        // We should have both offline and alive addresses which are not active.
853        assert_eq!(
854            endpoint
855                .direct_address_states()
856                .filter(|(_addr, state)| !state.is_active())
857                .count(),
858            MAX_INACTIVE_DIRECT_ADDRESSES * 2
859        )
860    }

866        let active_node = SecretKey::generate(rand::thread_rng()).public();
