1//! The [`BestAddr`] is the currently active best address for UDP sends.
2
3use std::net::SocketAddr;
4
5use n0_future::time::{Duration, Instant};
6use tracing::{debug, info};
7
8/// How long we trust a UDP address as the exclusive path (without using relay) without having heard a Pong reply.
9const TRUST_UDP_ADDR_DURATION: Duration = Duration::from_millis(6500);
10
11#[derive(Debug, Default)]
12pub(super) struct BestAddr(Option<BestAddrInner>);
13
14#[derive(Debug)]
15struct BestAddrInner {
16    addr: AddrLatency,
17    trust_until: Option<Instant>,
18    confirmed_at: Instant,
19}
20
21impl BestAddrInner {
22    fn is_trusted(&self, now: Instant) -> bool {
23        self.trust_until
24            .map(|trust_until| trust_until >= now)
25            .unwrap_or(false)
26    }
27
28    fn addr(&self) -> SocketAddr {
29        self.addr.addr
30    }
31}
32
33#[derive(Debug)]
34pub(super) enum Source {
35    ReceivedPong,
36    BestCandidate,
37    Udp,
38}
39
40impl Source {
41    fn trust_until(&self, from: Instant) -> Instant {
42        match self {
43            Source::ReceivedPong => from + TRUST_UDP_ADDR_DURATION,
44            // TODO: Fix time
45            Source::BestCandidate => from + Duration::from_secs(60 * 60),
46            Source::Udp => from + TRUST_UDP_ADDR_DURATION,
47        }
48    }
49}
50
51#[derive(Debug)]
52pub(super) enum State<'a> {
53    Valid(&'a AddrLatency),
54    Outdated(&'a AddrLatency),
55    Empty,
56}
57
58#[derive(Debug, Clone, Copy)]
59pub enum ClearReason {
60    Inactive,
61    PongTimeout,
62    MatchesOurLocalAddr,
63}
64
65impl BestAddr {
66    #[cfg(test)]
67    pub fn from_parts(
68        addr: SocketAddr,
69        latency: Duration,
70        confirmed_at: Instant,
71        trust_until: Instant,
72    ) -> Self {
73        let inner = BestAddrInner {
74            addr: AddrLatency { addr, latency },
75            confirmed_at,
76            trust_until: Some(trust_until),
77        };
78        Self(Some(inner))
79    }
80
81    pub fn is_empty(&self) -> bool {
82        self.0.is_none()
83    }
84
85    /// Unconditionally clears the best address.
86    pub fn clear(&mut self, reason: ClearReason, has_relay: bool) {
87        let old = self.0.take();
88        if let Some(old_addr) = old.as_ref().map(BestAddrInner::addr) {
89            info!(?reason, ?has_relay, %old_addr, "clearing best_addr");
90        }
91    }
92
93    /// Clears the best address if equal to `addr`.
94    pub fn clear_if_equals(&mut self, addr: SocketAddr, reason: ClearReason, has_relay: bool) {
95        if self.addr() == Some(addr) {
96            self.clear(reason, has_relay)
97        }
98    }
99
100    pub fn clear_trust(&mut self, why: &'static str) {
101        if let Some(state) = self.0.as_mut() {
102            info!(
103                %why,
104                prev_trust_until = ?state.trust_until,
105                "clearing best_addr trust",
106            );
107            state.trust_until = None;
108        }
109    }
110
111    pub fn insert_if_better_or_reconfirm(
112        &mut self,
113        addr: SocketAddr,
114        latency: Duration,
115        source: Source,
116        confirmed_at: Instant,
117    ) {
118        match self.0.as_mut() {
119            None => {
120                self.insert(addr, latency, source, confirmed_at);
121            }
122            Some(state) => {
123                let candidate = AddrLatency { addr, latency };
124                if !state.is_trusted(confirmed_at) || candidate.is_better_than(&state.addr) {
125                    self.insert(addr, latency, source, confirmed_at);
126                } else if state.addr.addr == addr {
127                    state.confirmed_at = confirmed_at;
128                    state.trust_until = Some(source.trust_until(confirmed_at));
129                }
130            }
131        }
132    }
133
134    /// Reset the expiry, if the passed in addr matches the currently used one.
135    #[cfg(not(wasm_browser))]
136    pub fn reconfirm_if_used(&mut self, addr: SocketAddr, source: Source, confirmed_at: Instant) {
137        if let Some(state) = self.0.as_mut() {
138            if state.addr.addr == addr {
139                state.confirmed_at = confirmed_at;
140                state.trust_until = Some(source.trust_until(confirmed_at));
141            }
142        }
143    }
144
145    fn insert(
146        &mut self,
147        addr: SocketAddr,
148        latency: Duration,
149        source: Source,
150        confirmed_at: Instant,
151    ) {
152        let trust_until = source.trust_until(confirmed_at);
153
154        if self
155            .0
156            .as_ref()
157            .map(|prev| prev.addr.addr == addr)
158            .unwrap_or_default()
159        {
160            debug!(
161                %addr,
162                latency = ?latency,
163                trust_for = ?trust_until.duration_since(Instant::now()),
164               "re-selecting direct path for node"
165            );
166        } else {
167            info!(
168               %addr,
169               latency = ?latency,
170               trust_for = ?trust_until.duration_since(Instant::now()),
171               "selecting new direct path for node"
172            );
173        }
174        let inner = BestAddrInner {
175            addr: AddrLatency { addr, latency },
176            trust_until: Some(trust_until),
177            confirmed_at,
178        };
179        self.0 = Some(inner);
180    }
181
182    pub fn state(&self, now: Instant) -> State<'_> {
183        match &self.0 {
184            None => State::Empty,
185            Some(state) => match state.trust_until {
186                Some(expiry) if now < expiry => State::Valid(&state.addr),
187                Some(_) | None => State::Outdated(&state.addr),
188            },
189        }
190    }
191
192    pub fn addr(&self) -> Option<SocketAddr> {
193        self.0.as_ref().map(BestAddrInner::addr)
194    }
195}

197/// A `SocketAddr` with an associated latency.
198#[derive(Debug, Clone)]
199pub struct AddrLatency {
200    pub addr: SocketAddr,
201    pub latency: Duration,
202}
203
204impl AddrLatency {
205    /// Reports whether `self` is a better addr to use than `other`.
206    fn is_better_than(&self, other: &Self) -> bool {
207        if self.addr == other.addr {
208            return false;
209        }
210        if self.addr.is_ipv6() && other.addr.is_ipv4() {
211            // Prefer IPv6 for being a bit more robust, as long as
212            // the latencies are roughly equivalent.
213            if self.latency / 10 * 9 < other.latency {
214                return true;
215            }
216        } else if self.addr.is_ipv4() && other.addr.is_ipv6() && other.is_better_than(self) {
217            return false;
218        }
219        self.latency < other.latency
220    }
221}
