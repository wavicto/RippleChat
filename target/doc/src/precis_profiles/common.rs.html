10pub fn is_space_separator(c: char) -> bool {
11    let cp = c as u32;
12    SPACE_SEPARATOR
13        .binary_search_by(|cps| cps.partial_cmp(&cp).unwrap())
14        .is_ok()
15}
16
17pub fn is_non_ascii_space(c: char) -> bool {
18    c != SPACE && is_space_separator(c)
19}

21pub fn normalization_form_nfkc<'a, T>(s: T) -> Result<Cow<'a, str>, Error>
22where
23    T: Into<Cow<'a, str>>,
24{
25    // Checking if this string is normalized can help us to save memory
26    // allocations but this will make this function a bit less efficient
27    // due to the extra iteration over the string
28    let s = s.into();
29    if unicode_normalization::is_nfkc(&s) {
30        Ok(s)
31    } else {
32        Ok(s.nfkc().collect::<String>().into())
33    }
34}

51pub fn case_mapping_rule<'a, T>(s: T) -> Result<Cow<'a, str>, Error>
52where
53    T: Into<Cow<'a, str>>,
54{
55    let s = s.into();
56    match s.find(char::is_uppercase) {
57        None => Ok(s),
58        Some(pos) => {
59            let mut res = String::from(&s[..pos]);
60            res.reserve(s.len() - res.len());
61            for c in s[pos..].chars() {
62                if c.is_lowercase() {
63                    res.push(c);
64                } else {
65                    c.to_lowercase().for_each(|x| res.push(x));
66                }
67            }
68            Ok(res.into())
69        }
70    }
71}

77    #[test]
78    fn test_normalization_rule() {
79        let res = normalization_form_nfc("");
80        assert_eq!(res, Ok(Cow::from("")));
81
82        let res = normalization_form_nfc("\u{212b}");
83        assert_eq!(res, Ok(Cow::from("\u{00c5}")));
84
85        let res = normalization_form_nfc("a\u{212b}");
86        assert_eq!(res, Ok(Cow::from("a\u{00c5}")));
87
88        let res = normalization_form_nfc("\u{212b}a");
89        assert_eq!(res, Ok(Cow::from("\u{00c5}a")));
90
91        let res = normalization_form_nfc("\u{212b}\u{2126}\u{1e0b}\u{0323}");
92        assert_eq!(res, Ok(Cow::from("\u{00c5}\u{03a9}\u{1e0d}\u{0307}")));
93    }
