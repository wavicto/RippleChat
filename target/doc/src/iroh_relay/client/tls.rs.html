1//! Functionality related to lower-level tls-based connection establishment.
2//!
3//! Primarily to support [`ClientBuilder::connect_relay`].
4//!
5//! This doesn't work in the browser - thus separated into its own file.
6//!
7//! `connect_relay` uses a custom HTTP upgrade header value (see [`HTTP_UPGRADE_PROTOCOL`]),
8//! as opposed to [`WEBSOCKET_UPGRADE_PROTOCOL`].
9//!
10//! `connect_ws` however reuses websockets for framing.
11//!
12//! [`HTTP_UPGRADE_PROTOCOL`]: crate::http::HTTP_UPGRADE_PROTOCOL
13//! [`WEBSOCKET_UPGRADE_PROTOCOL`]: crate::http::WEBSOCKET_UPGRADE_PROTOCOL
14

76    pub async fn connect(self) -> Result<MaybeTlsStream<ProxyStream>, ConnectError> {
77        let roots = rustls::RootCertStore {
78            roots: webpki_roots::TLS_SERVER_ROOTS.to_vec(),
79        };
80        let mut config = rustls::client::ClientConfig::builder_with_provider(Arc::new(
81            rustls::crypto::ring::default_provider(),
82        ))
83        .with_safe_default_protocol_versions()
84        .expect("protocols supported by ring")
85        .with_root_certificates(roots)
86        .with_no_client_auth();
87        #[cfg(any(test, feature = "test-utils"))]
88        if self.insecure_skip_cert_verify {
89            warn!("Insecure config: SSL certificates from relay servers not verified");
90            config
91                .dangerous()
92                .set_certificate_verifier(Arc::new(NoCertVerifier));
93        }
94        config.resumption = Resumption::default();
95        let tls_connector: tokio_rustls::TlsConnector = Arc::new(config).into();
96
97        let tcp_stream = self.dial_url(&tls_connector).await?;
98
99        let local_addr = tcp_stream
100            .local_addr()
101            .map_err(|_| NoLocalAddrSnafu.build())?;
102
103        debug!(server_addr = ?tcp_stream.peer_addr(), %local_addr, "TCP stream connected");
104
105        if self.use_tls() {
106            debug!("Starting TLS handshake");
107            let hostname = self
108                .tls_servername()
109                .ok_or_else(|| InvalidTlsServernameSnafu.build())?;
110
111            let hostname = hostname.to_owned();
112            let tls_stream = tls_connector
113                .connect(hostname, tcp_stream)
114                .await
115                .context(TlsSnafu)?;
116            debug!("tls_connector connect success");
117            Ok(MaybeTlsStream::Tls(tls_stream))
118        } else {
119            debug!("Starting handshake");
120            Ok(MaybeTlsStream::Raw(tcp_stream))
121        }
122    }
123
124    fn use_tls(&self) -> bool {
125        // only disable tls if we are explicitly dialing a http url
126        #[allow(clippy::match_like_matches_macro)]
127        match self.url.scheme() {
128            "http" => false,
129            "ws" => false,
130            _ => true,
131        }
132    }
133
134    fn tls_servername(&self) -> Option<rustls::pki_types::ServerName<'_>> {
135        let host_str = self.url.host_str()?;
136        let servername = rustls::pki_types::ServerName::try_from(host_str).ok()?;
137        Some(servername)
138    }

140    async fn dial_url(
141        &self,
142        tls_connector: &tokio_rustls::TlsConnector,
143    ) -> Result<ProxyStream, DialError> {
144        if let Some(ref proxy) = self.proxy_url {
145            let stream = self.dial_url_proxy(proxy.clone(), tls_connector).await?;
146            Ok(ProxyStream::Proxied(stream))
147        } else {
148            let stream = self.dial_url_direct().await?;
149            Ok(ProxyStream::Raw(stream))
150        }
151    }
152
153    async fn dial_url_direct(&self) -> Result<tokio::net::TcpStream, DialError> {
154        use tokio::net::TcpStream;
155        debug!(%self.url, "dial url");
156        let dst_ip = self
157            .dns_resolver
158            .resolve_host(&self.url, self.prefer_ipv6, DNS_TIMEOUT)
159            .await?;
160
161        let port = url_port(&self.url).context(InvalidTargetPortSnafu)?;
162        let addr = SocketAddr::new(dst_ip, port);
163
164        debug!("connecting to {}", addr);
165        let tcp_stream = time::timeout(
166            DIAL_NODE_TIMEOUT,
167            async move { TcpStream::connect(addr).await },
168        )
169        .await??;
170
171        tcp_stream.set_nodelay(true)?;
172
173        Ok(tcp_stream)
174    }

176    async fn dial_url_proxy(
177        &self,
178        proxy_url: Url,
179        tls_connector: &tokio_rustls::TlsConnector,
180    ) -> Result<util::Chain<std::io::Cursor<Bytes>, MaybeTlsStream<tokio::net::TcpStream>>, DialError>
181    {
182        use hyper_util::rt::TokioIo;
183        use tokio::net::TcpStream;
184        debug!(%self.url, %proxy_url, "dial url via proxy");
185
186        // Resolve proxy DNS
187        let proxy_ip = self
188            .dns_resolver
189            .resolve_host(&proxy_url, self.prefer_ipv6, DNS_TIMEOUT)
190            .await?;
191
192        let proxy_port = url_port(&proxy_url).context(ProxyInvalidTargetPortSnafu)?;
193        let proxy_addr = SocketAddr::new(proxy_ip, proxy_port);
194
195        debug!(%proxy_addr, "connecting to proxy");
196
197        let tcp_stream = time::timeout(DIAL_NODE_TIMEOUT, async move {
198            TcpStream::connect(proxy_addr).await
199        })
200        .await??;
201
202        tcp_stream.set_nodelay(true)?;
203
204        // Setup TLS if necessary
205        let io = if proxy_url.scheme() == "http" {
206            MaybeTlsStream::Raw(tcp_stream)
207        } else {
208            let hostname = proxy_url.host_str().context(ProxyInvalidUrlSnafu {
209                proxy_url: proxy_url.clone(),
210            })?;
211            let hostname =
212                rustls::pki_types::ServerName::try_from(hostname.to_string()).map_err(|_| {
213                    ProxyInvalidTlsServernameSnafu {
214                        proxy_hostname: hostname.to_string(),
215                    }
216                    .build()
217                })?;
218            let tls_stream = tls_connector.connect(hostname, tcp_stream).await?;
219            MaybeTlsStream::Tls(tls_stream)
220        };
221        let io = TokioIo::new(io);
222
223        let target_host = self.url.host_str().context(InvalidUrlSnafu {
224            url: self.url.clone(),
225        })?;
226
227        let port = url_port(&self.url).context(InvalidTargetPortSnafu)?;
228
229        // Establish Proxy Tunnel
230        let mut req_builder = Request::builder()
231            .uri(format!("{target_host}:{port}"))
232            .method("CONNECT")
233            .header("Host", target_host)
234            .header("Proxy-Connection", "Keep-Alive");
235        if !proxy_url.username().is_empty() {
236            // Passthrough authorization
237            // https://developer.mozilla.org/en-US/docs/Web/HTTP/Headers/Proxy-Authorization
238            debug!(
239                "setting proxy-authorization: username={}",
240                proxy_url.username()
241            );
242            let to_encode = format!(
243                "{}:{}",
244                proxy_url.username(),
245                proxy_url.password().unwrap_or_default()
246            );
247            let encoded = BASE64URL.encode(to_encode.as_bytes());
248            req_builder = req_builder.header("Proxy-Authorization", format!("Basic {encoded}"));
249        }
250        let req = req_builder
251            .body(Empty::<Bytes>::new())
252            .expect("fixed config");
253
254        debug!("Sending proxy request: {:?}", req);
255
256        let (mut sender, conn) = hyper::client::conn::http1::handshake(io)
257            .await
258            .context(ProxyConnectSnafu)?;
259        task::spawn(async move {
260            if let Err(err) = conn.with_upgrades().await {
261                error!("Proxy connection failed: {:?}", err);
262            }
263        });
264
265        let res = sender.send_request(req).await.context(ProxyConnectSnafu)?;
266        if !res.status().is_success() {
267            return Err(ProxyConnectInvalidStatusSnafu {
268                status: res.status(),
269            }
270            .build());
271        }
272
273        let upgraded = hyper::upgrade::on(res).await.context(ProxyConnectSnafu)?;
274        let Parts { io, read_buf, .. } = upgraded
275            .downcast::<TokioIo<MaybeTlsStream<tokio::net::TcpStream>>>()
276            .expect("only this upgrade used");
277
278        let res = util::chain(std::io::Cursor::new(read_buf), io.into_inner());
279
280        Ok(res)
281    }

284impl ClientBuilder {
285    /// Connects to configured relay using HTTP(S) with an upgrade header
286    /// set to [`HTTP_UPGRADE_PROTOCOL`].
287    ///
288    /// [`HTTP_UPGRADE_PROTOCOL`]: crate::http::HTTP_UPGRADE_PROTOCOL
289    pub(super) async fn connect_relay(&self) -> Result<(Conn, SocketAddr), ConnectError> {
290        #[allow(unused_mut)]
291        let mut builder =
292            MaybeTlsStreamBuilder::new(self.url.clone().into(), self.dns_resolver.clone())
293                .prefer_ipv6(self.prefer_ipv6())
294                .proxy_url(self.proxy_url.clone());
295
296        #[cfg(any(test, feature = "test-utils"))]
297        if self.insecure_skip_cert_verify {
298            builder = builder.insecure_skip_cert_verify(self.insecure_skip_cert_verify);
299        }
300
301        let stream = builder.connect().await?;
302        let local_addr = stream
303            .as_ref()
304            .local_addr()
305            .map_err(|_| NoLocalAddrSnafu.build())?;
306        let response = self.http_upgrade_relay(stream).await?;
307
308        if response.status() != hyper::StatusCode::SWITCHING_PROTOCOLS {
309            UnexpectedUpgradeStatusSnafu {
310                code: response.status(),
311            }
312            .fail()?;
313        }
314
315        debug!("starting upgrade");
316        let upgraded = hyper::upgrade::on(response).await.context(UpgradeSnafu)?;
317
318        debug!("connection upgraded");
319        let conn = downcast_upgrade(upgraded).expect("must use TcpStream or client::TlsStream");
320
321        let conn = Conn::new_relay(conn, self.key_cache.clone(), &self.secret_key).await?;
322
323        Ok((conn, local_addr))
324    }

326    pub(super) async fn connect_ws(&self) -> Result<(Conn, SocketAddr), ConnectError> {
327        let mut dial_url = (*self.url).clone();
328        dial_url.set_path(RELAY_PATH);
329        // The relay URL is exchanged with the http(s) scheme in tickets and similar.
330        // We need to use the ws:// or wss:// schemes when connecting with websockets, though.
331        dial_url
332            .set_scheme(match self.url.scheme() {
333                "http" => "ws",
334                "ws" => "ws",
335                _ => "wss",
336            })
337            .map_err(|_| {
338                InvalidWebsocketUrlSnafu {
339                    url: dial_url.clone(),
340                }
341                .build()
342            })?;
343
344        debug!(%dial_url, "Dialing relay by websocket");
345
346        #[allow(unused_mut)]
347        let mut builder = MaybeTlsStreamBuilder::new(dial_url.clone(), self.dns_resolver.clone())
348            .prefer_ipv6(self.prefer_ipv6())
349            .proxy_url(self.proxy_url.clone());
350
351        #[cfg(any(test, feature = "test-utils"))]
352        if self.insecure_skip_cert_verify {
353            builder = builder.insecure_skip_cert_verify(self.insecure_skip_cert_verify);
354        }
355
356        let stream = builder.connect().await?;
357        let local_addr = stream
358            .as_ref()
359            .local_addr()
360            .map_err(|_| NoLocalAddrSnafu.build())?;
361        let (conn, response) = tokio_websockets::ClientBuilder::new()
362            .uri(dial_url.as_str())
363            .map_err(|_| {
364                InvalidRelayUrlSnafu {
365                    url: dial_url.clone(),
366                }
367                .build()
368            })?
369            .connect_on(stream)
370            .await?;
371
372        if response.status() != hyper::StatusCode::SWITCHING_PROTOCOLS {
373            UnexpectedUpgradeStatusSnafu {
374                code: response.status(),
375            }
376            .fail()?;
377        }
378
379        let conn = Conn::new_ws(conn, self.key_cache.clone(), &self.secret_key).await?;
380
381        Ok((conn, local_addr))
382    }

384    /// Sends the HTTP upgrade request to the relay server.
385    async fn http_upgrade_relay<T>(&self, io: T) -> Result<hyper::Response<Incoming>, ConnectError>
386    where
387        T: AsyncRead + AsyncWrite + Send + Unpin + 'static,
388    {
389        use hyper_util::rt::TokioIo;
390        let host_header_value =
391            host_header_value(self.url.clone()).context(InvalidRelayUrlSnafu {
392                url: Url::from(self.url.clone()),
393            })?;
394
395        let io = TokioIo::new(io);
396        let (mut request_sender, connection) = hyper::client::conn::http1::Builder::new()
397            .handshake(io)
398            .await
399            .context(UpgradeSnafu)?;
400        task::spawn(
401            // This task drives the HTTP exchange, completes once connection is upgraded.
402            async move {
403                debug!("HTTP upgrade driver started");
404                if let Err(err) = connection.with_upgrades().await {
405                    error!("HTTP upgrade error: {err:#}");
406                }
407                debug!("HTTP upgrade driver finished");
408            }
409            .instrument(info_span!("http-driver")),
410        );
411        debug!("Sending upgrade request");
412        let req = Request::builder()
413            .uri(RELAY_PATH)
414            .header(UPGRADE, Protocol::Relay.upgrade_header())
415            // https://datatracker.ietf.org/doc/html/rfc2616#section-14.23
416            // > A client MUST include a Host header field in all HTTP/1.1 request messages.
417            // This header value helps reverse proxies identify how to forward requests.
418            .header(HOST, host_header_value)
419            .body(http_body_util::Empty::<hyper::body::Bytes>::new())
420            .expect("fixed config");
421        request_sender.send_request(req).await.context(UpgradeSnafu)
422    }

424    /// Reports whether IPv4 dials should be slightly
425    /// delayed to give IPv6 a better chance of winning dial races.
426    /// Implementations should only return true if IPv6 is expected
427    /// to succeed. (otherwise delaying IPv4 will delay the connection
428    /// overall)
429    fn prefer_ipv6(&self) -> bool {
430        match self.address_family_selector {
431            Some(ref selector) => selector(),
432            None => false,
433        }
434    }

437/// Returns none if no valid url host was found.
438fn host_header_value(relay_url: RelayUrl) -> Option<String> {
439    // grab the host, turns e.g. https://example.com:8080/xyz -> example.com.
440    let relay_url_host = relay_url.host_str()?;
441
442    // strip the trailing dot, if present: example.com. -> example.com
443    let relay_url_host = relay_url_host.strip_suffix('.').unwrap_or(relay_url_host);
444    // build the host header value (reserve up to 6 chars for the ":" and port digits):
445    let mut host_header_value = String::with_capacity(relay_url_host.len() + 6);
446    host_header_value += relay_url_host;
447    if let Some(port) = relay_url.port() {
448        host_header_value += ":";
449        host_header_value += &port.to_string();
450    }
451    Some(host_header_value)
452}
453
454fn url_port(url: &Url) -> Option<u16> {
455    if let Some(port) = url.port() {
456        return Some(port);
457    }
458
459    match url.scheme() {
460        "http" | "ws" => Some(80),
461        "https" | "wss" => Some(443),
462        _ => None,
463    }
464}
