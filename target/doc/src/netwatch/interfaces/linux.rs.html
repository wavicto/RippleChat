63const PROC_NET_ROUTE_PATH: &str = "/proc/net/route";
64
65async fn default_route_proc() -> Result<Option<DefaultRouteDetails>, Error> {
66    const ZERO_ADDR: &str = "00000000";
67    let file = File::open(PROC_NET_ROUTE_PATH).await.context(IoSnafu)?;
68
69    // Explicitly set capacity, this is min(4096, DEFAULT_BUF_SIZE):
70    // https://github.com/google/gvisor/issues/5732
71    // On a regular Linux kernel you can read the first 128 bytes of /proc/net/route,
72    // then come back later to read the next 128 bytes and so on.
73    //
74    // In Google Cloud Run, where /proc/net/route comes from gVisor, you have to
75    // read it all at once. If you read only the first few bytes then the second
76    // read returns 0 bytes no matter how much originally appeared to be in the file.
77    //
78    // At the time of this writing (Mar 2021) Google Cloud Run has eth0 and eth1
79    // with a 384 byte /proc/net/route. We allocate a large buffer to ensure we'll
80    // read it all in one call.
81    let reader = BufReader::with_capacity(8 * 1024, file);
82    let mut lines_iter = reader.lines();
83    while let Some(line) = lines_iter.next_line().await.context(IoSnafu)? {
84        if !line.contains(ZERO_ADDR) {
85            continue;
86        }
87        let mut fields = line.split_ascii_whitespace();
88        let iface = fields.next().context(MissingIfaceFieldSnafu)?;
89        let destination = fields.next().context(MissingDestinationFieldSnafu)?;
90        let mask = fields.nth(5).context(MissingMaskFieldSnafu)?;
91        // if iface.starts_with("tailscale") || iface.starts_with("wg") {
92        //     continue;
93        // }
94        if destination == ZERO_ADDR && mask == ZERO_ADDR {
95            return Ok(Some(DefaultRouteDetails {
96                interface_name: iface.to_string(),
97            }));
98        }
99    }
100    Ok(None)
101}

109    /// Try find the default route by parsing the "ip route" command output.
110    ///
111    /// We use this on Android where /proc/net/route can be missing entries or have locked-down
112    /// permissions.  See also comments in <https://github.com/tailscale/tailscale/pull/666>.
113    pub async fn default_route() -> Result<Option<DefaultRouteDetails>, Error> {
114        let output = Command::new("/system/bin/ip")
115            .args(["route", "show", "table", "0"])
116            .kill_on_drop(true)
117            .output()
118            .await
119            .context(IoSnafu)?;
120        let stdout = std::string::String::from_utf8_lossy(&output.stdout);
121        let details = parse_android_ip_route(&stdout).map(|iface| DefaultRouteDetails {
122            interface_name: iface.to_string(),
123        });
124        Ok(details)
125    }
126}

143    type Handle = netlink_proto::ConnectionHandle<RouteNetlinkMessage>;

145    macro_rules! try_rtnl {
146        ($msg: expr, $message_type:path) => {{
147            use netlink_packet_core::NetlinkPayload;
148            use netlink_packet_route::RouteNetlinkMessage;
149
150            let (_header, payload) = $msg.into_parts();
151            match payload {
152                NetlinkPayload::InnerMessage($message_type(msg)) => msg,
153                NetlinkPayload::Error(err) => {
154                    return Err(NetlinkErrorMessageSnafu { message: err }.build())
155                }
156                _ => return Err(UnexpectedNetlinkMessageSnafu.build()),
157            }
158        }};
159    }
160
161    pub async fn default_route() -> Result<Option<DefaultRouteDetails>, Error> {
162        let (connection, handle, _receiver) =
163            netlink_proto::new_connection::<RouteNetlinkMessage>(NETLINK_ROUTE).context(IoSnafu)?;
164
165        let task = tokio::spawn(connection.instrument(info_span!("netlink.conn")));
166
167        let default = default_route_netlink_family(&handle, AddressFamily::Inet).await?;
168        let default = match default {
169            Some(default) => Some(default),
170            None => {
171                default_route_netlink_family(&handle, netlink_packet_route::AddressFamily::Inet6)
172                    .await?
173            }
174        };
175        task.abort();
176        task.await.ok();
177        Ok(default.map(|(name, _index)| DefaultRouteDetails {
178            interface_name: name,
179        }))
180    }
181
182    fn get_route(
183        handle: Handle,
184        message: RouteMessage,
185    ) -> impl TryStream<Ok = RouteMessage, Err = Error> {
186        let mut req = NetlinkMessage::from(RouteNetlinkMessage::GetRoute(message));
187        req.header.flags = NLM_F_REQUEST | NLM_F_DUMP;
188
189        match handle.request(req, netlink_proto::sys::SocketAddr::new(0, 0)) {
190            Ok(response) => Either::Left(
191                response.map(move |msg| Ok(try_rtnl!(msg, RouteNetlinkMessage::NewRoute))),
192            ),
193            Err(e) => Either::Right(n0_future::stream::once::<Result<RouteMessage, Error>>(Err(
194                NetlinkSnafu.into_error(e),
195            ))),
196        }
197    }
198
199    fn create_route_message(family: netlink_packet_route::AddressFamily) -> RouteMessage {
200        let mut message = RouteMessage::default();
201        message.header.table = RouteHeader::RT_TABLE_MAIN;
202        message.header.protocol = RouteProtocol::Static;
203        message.header.scope = RouteScope::Universe;
204        message.header.kind = RouteType::Unicast;
205        message.header.address_family = family;
206        message
207    }
208
209    /// Returns the `(name, index)` of the interface for the default route.
210    async fn default_route_netlink_family(
211        handle: &Handle,
212        family: netlink_packet_route::AddressFamily,
213    ) -> Result<Option<(String, u32)>, Error> {
214        let msg = create_route_message(family);
215        let mut routes = get_route(handle.clone(), msg);
216
217        while let Some(route) = routes.try_next().await? {
218            let route_attrs = route.attributes;
219
220            if !route_attrs
221                .iter()
222                .any(|attr| matches!(attr, RouteAttribute::Gateway(_)))
223            {
224                // A default route has a gateway.
225                continue;
226            }
227
228            if route.header.destination_prefix_length > 0 {
229                // A default route has no destination prefix length because it needs to route all
230                // destinations.
231                continue;
232            }
233
234            let index = route_attrs.iter().find_map(|attr| match attr {
235                RouteAttribute::Oif(index) => Some(*index),
236                _ => None,
237            });
238
239            if let Some(index) = index {
240                if index == 0 {
241                    continue;
242                }
243                let name = iface_by_index(handle, index).await?;
244                return Ok(Some((name, index)));
245            }
246        }
247        Ok(None)
248    }
249
250    fn get_link(
251        handle: Handle,
252        message: LinkMessage,
253    ) -> impl TryStream<Ok = LinkMessage, Err = Error> {
254        let mut req = NetlinkMessage::from(RouteNetlinkMessage::GetLink(message));
255        req.header.flags = NLM_F_REQUEST;
256
257        match handle.request(req, netlink_proto::sys::SocketAddr::new(0, 0)) {
258            Ok(response) => Either::Left(
259                response.map(move |msg| Ok(try_rtnl!(msg, RouteNetlinkMessage::NewLink))),
260            ),
261            Err(e) => Either::Right(n0_future::stream::once::<Result<LinkMessage, Error>>(Err(
262                NetlinkSnafu.into_error(e),
263            ))),
264        }
265    }
266
267    fn create_link_get_message(index: u32) -> LinkMessage {
268        let mut message = LinkMessage::default();
269        message.header.index = index;
270        message
271    }
272
273    async fn iface_by_index(handle: &Handle, index: u32) -> Result<String, Error> {
274        let message = create_link_get_message(index);
275        let mut links = get_link(handle.clone(), message);
276        let msg = links.try_next().await?.context(NoResponseSnafu)?;
277
278        for nla in msg.attributes {
279            if let LinkAttribute::IfName(name) = nla {
280                return Ok(name);
281            }
282        }
283        Err(InterfaceNotFoundSnafu.build())
284    }

301/// Parses the output of the android `/system/bin/ip` command for the default route.
302///
303/// Searches for line like `default via 10.0.2.2. dev radio0 table 1016 proto static mtu
304/// 1500`
305#[cfg(any(target_os = "android", test))]
306fn parse_android_ip_route(stdout: &str) -> Option<&str> {
307    for line in stdout.lines() {
308        if !line.starts_with("default via") {
309            continue;
310        }
311        let mut fields = line.split_ascii_whitespace();
312        if let Some(_dev) = fields.find(|s: &&str| *s == "dev") {
313            return fields.next();
314        }
315    }
316    None
317}
